047    //if arg is not null, then it's the first element in the returned list
048    private List<String[]> categories(LogType logType) {
049        List<String[]> ret = new ArrayList<String[]>();
050        for(LogType t : LogType.values()) {
051                String s[] = new String[2];
052                s[0] = t.toString();
053                s[1] = "";
054                if(t==logType) ret.add(0, s); else ret.add(s);
055        }       
056        return ret;
057    }     
058    
059        @RequestMapping("/events")
060    public @ResponseBody List<String[]> logEvents() {
061        List<String[]> ret = categories(null);
062        return ret;
063    } 
064
065    @RequestMapping("/{logType}/events")
066    public @ResponseBody List<String[]> logEvents(@PathVariable LogType logType) {        
067        //add all the events in the same category (type); 
068        //but make given type the first element in the list (selected)
069        List<String[]> ret = categories(logType);
070        for(LogEvent e : service.log().logEvents(logType)) {
071                ret.add(new String[]{e.getType().toString(), e.getName()});
072        }
073        return ret;
074    }    
075    
076    @RequestMapping("/{logType}/{name}/events")
077    public @ResponseBody List<String[]> logEvents(@PathVariable LogType logType, 
078                @PathVariable String name) {
079        List<String[]> ret = categories(logType);
080        //add all the events in the same category (type); 
081        // but make the one with given type,name the first element in the list
082        for(LogEvent e : service.log().logEvents(logType)) {
083                if(e.getName().equals(name))
084                        ret.add(0, new String[]{e.getType().toString(), e.getName()});
085                else 
086                        ret.add(new String[]{e.getType().toString(), e.getName()});
087        }
088        return ret;
089    }

228        /*
229         * Access log summary - 
230         * a json object (map) that contains the list of rows ('data'),
231         * each row is object (map) with three columns (key-val pairs);
232         * this can be then read by a jQuery DataTable 
233         * via "ajaxSource" and "columns" parameters
234         * (this method here is experimental).
235         * 
236         * @return object
237         */
238        @RequestMapping("/totals")
239    public @ResponseBody Map<String, List<Map<String,?>>> totalTable() {      
240                Map<String, List<Map<String,?>>> data = new HashMap<String,List<Map<String,?>>>();
241                List<Map<String,?>> l = new ArrayList<Map<String,?>>();
242                data.put("data", l); //there will be only one data entry (the list)
243                
244                Map<String, Object> m;
245                Long totalPathways = 0L;
246                Long totalInteractions = 0L;
247                
248                //also - for each provider -
249                int numProviders = 0; //not warehouse data
250                for(Metadata mtda : service.metadata().findAll()) {
251                        if(mtda.isNotPathwayData())
252                                continue; //skip warehouse data sources
253                        
254                        m = new HashMap<String, Object>();
255                        m.put("identifier", mtda.identifier);
256                        m.put("name", mtda.standardName());
257                        m.put("totalok", service.log().downloads(mtda.standardName()));//actually, incl. errors
258                        m.put("totalip", service.log().uniqueIps(mtda.standardName()));
259                        m.put("pathways", mtda.getNumPathways());
260                        m.put("interactions", mtda.getNumInteractions());
261                        l.add(m);
262                        totalPathways += mtda.getNumPathways();
263                        totalInteractions += mtda.getNumInteractions();
264                        
265                        numProviders++;
266                }
267
268                // add total requests/IPs counts to the list
269                m = new HashMap<String, Object>();
270                m.put("identifier", LogType.TOTAL.description); //"All"
271                m.put("name", "PC2 v" + CPathSettings.getInstance().getVersion() + " web service");
272                m.put("datasources", numProviders);
273                m.put("totalok", totalRequests());
274                m.put("totalip", totalUniqueIps());
275                m.put("pathways", totalPathways);
276                m.put("interactions", totalInteractions);
277                l.add(0, m);
278                
279                return data;
280    }

120    @RequestMapping("/metadata/logo/{identifier}")
121    public  @ResponseBody byte[] queryForLogo(@PathVariable String identifier) 
122                throws IOException 
123    {   
124        Metadata ds = service.metadata().findByIdentifier(identifier);
125        byte[] bytes = null;
126        
127        if(ds != null) {
128                        BufferedImage bufferedImage = ImageIO
129                                .read(CPathUtils.LOADER.getResource(ds.getIconUrl()).getInputStream());
130                        
131                        //resize (originals are around 125X60)
132                        bufferedImage = scaleImage(bufferedImage, 100, 50, null);
133                                                
134                        ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
135                        ImageIO.write(bufferedImage, "gif", byteArrayOutputStream);
136                        bytes = byteArrayOutputStream.toByteArray();
137                }
138        
139        return bytes;
140    }

144    @RequestMapping("/metadata/datasources")
145    public  @ResponseBody List<Metadata> queryForDatasources() {
146                log.debug("Getting pathway datasources info.");
147        //pathway/interaction data sources
148                List<Metadata> ds = new ArrayList<Metadata>();
149                //warehouse data sources
150                List<Metadata> wh = new ArrayList<Metadata>();
151                
152                for(Metadata m : service.metadata().findAll()) {
153                        //set dynamic extra fields
154                        if(m.isNotPathwayData()) {
155                                m.setUploaded((new File(m.getDataArchiveName()).exists()));
156                                wh.add(m);
157                        } else {                
158                                m.setNumAccessed(service.log().downloads(m.standardName()));
159                                m.setUploaded((new File(m.getDataArchiveName()).exists()));
160                                m.setPremerged(!m.getContent().isEmpty());
161                                m.setNumUniqueIps(service.log().uniqueIps(m.standardName()));
162                                ds.add(m);
163                        }
164                }
165                
166                //add warehouse data sources to the end of the list
167                ds.addAll(wh);
168                
169        return ds;
170    }

194    @RequestMapping(value = "/admin/datasources", consumes="application/json", method = RequestMethod.POST)
195    public void update(@RequestBody @Valid Metadata metadata, 
196                BindingResult bindingResult, HttpServletResponse response) throws IOException 
197    {   
198        if(bindingResult != null &&  bindingResult.hasErrors()) {
199                log.error(Status.BAD_REQUEST.getErrorCode() + "; " +  
200                                Status.BAD_REQUEST.getErrorMsg() + "; " + errorFromBindingResult(bindingResult));
201                response.sendError(Status.BAD_REQUEST.getErrorCode(), 
202                        Status.BAD_REQUEST.getErrorMsg() + "; " + errorFromBindingResult(bindingResult));
203        }
204        
205                service.save(metadata);
206    }

209    @RequestMapping(value = "/admin/datasources", consumes="application/json", method = RequestMethod.PUT)
210    public void put(@RequestBody @Valid Metadata metadata, 
211                BindingResult bindingResult, HttpServletResponse response) throws IOException 
212    {   
213        if(bindingResult != null &&  bindingResult.hasErrors()) {
214                log.error(Status.BAD_REQUEST.getErrorCode() + "; " +  
215                                Status.BAD_REQUEST.getErrorMsg() + "; " + errorFromBindingResult(bindingResult));
216                response.sendError(Status.BAD_REQUEST.getErrorCode(), 
217                        Status.BAD_REQUEST.getErrorMsg() + "; " + errorFromBindingResult(bindingResult));
218        }
219        
220        Metadata existing = service.metadata().findByIdentifier(metadata.identifier);
221        if(existing == null) {
222                service.save(metadata);
223        } else {
224                response.sendError(Status.BAD_REQUEST.getErrorCode(), 
225                "PUT failed: Metadata already exists for pk: " + metadata.identifier
226                        + " (use POST to update instead)");
227        }
228    }

231    public void delete(@PathVariable String identifier, HttpServletResponse response) throws IOException 
232    {   
233        Metadata existing = service.metadata().findByIdentifier(identifier);

235                service.delete(existing);//clear files
236        } else {
237                response.sendError(Status.NO_RESULTS_FOUND.getErrorCode(), 
238                "DELETE failed: no Metadata record found for pk: " + identifier);
239        }
240    }

243    //Upload a data archive
244    @RequestMapping(value = "/admin/datasources/{identifier}/file", method = RequestMethod.POST)
245    public void uploadDataArchive(@PathVariable String identifier, MultipartHttpServletRequest multiRequest, 
246                HttpServletResponse response) throws IOException
247    {           
248        Metadata m = service.metadata().findByIdentifier(identifier);
249        if(m==null) {
250                response.sendError(Status.NO_RESULTS_FOUND.getErrorCode(), Status.NO_RESULTS_FOUND.getErrorMsg() + 
251                        "; Metadata object with identifier: " + identifier + " not found.");
252        }
253        
254                Map<String, MultipartFile> files = multiRequest.getFileMap();
255                Assert.state(!files.isEmpty(), "No files to validate");
256                String filename = files.keySet().iterator().next();
257                MultipartFile file = files.get(filename);
258                String origFilename = file.getOriginalFilename();                       
259                if(file.getBytes().length==0 || filename==null || "".equals(filename) || !origFilename.endsWith(".zip")) {
260                        log.error("uploadDataArchive(), empty data file or null: " + origFilename);
261                        response.sendError(Status.BAD_REQUEST.getErrorCode(), 
262                        "File (" + origFilename + ") UPLOAD failed; id:" + identifier);
263                } else {
264                        //create or update the input source data file (must be ZIP archive!)
265                        CPathUtils.write(file.getBytes(), m.getDataArchiveName());
266                        log.info("uploadDataArchive(), saved uploaded file:" 
267                                        + origFilename + " as " + m.getDataArchiveName());
268                }
269    }

272    @RequestMapping("/idmapping")
273    public @ResponseBody Map<String, String> idMapping(@RequestParam String[] id, 
274                HttpServletRequest request, HttpServletResponse response) throws IOException
275    {                   
276        //log events: command, format
277        Set<LogEvent> events = new HashSet<LogEvent>();
278        events.add(LogEvent.IDMAPPING);
279        events.add(LogEvent.FORMAT_OTHER);
280
281        if(id == null || id.length == 0) {
282                errorResponse(Status.NO_RESULTS_FOUND, "No ID(s) specified.", 
283                                request, response, events);
284                return null;
285        }
286
287        Map<String, String> res = new TreeMap<String, String>();
288
289        for(String i : id) {                                                    
290                Set<String> im = service.map(i);
291                if(im == null) {
292                        res.put(i, null);
293                } else {
294                        for(String ac : im)
295                                res.put(i, ac);
296                }                       
297        }               
298
299        //log to db (for usage reports)
300        service.log(events, clientIpAddress(request));
301
302        return res;
303        }

306    private List<ValInfo> validationInfo() {
307        final List<ValInfo> list = new ArrayList<ValInfo>();
308        
309                for(Metadata m : service.metadata().findAll()) {
310                        if(m.isNotPathwayData())
311                                continue;
312                        
313                        ValInfo vi = new ValInfo();
314                        vi.setIdentifier(m.getIdentifier());
315                        
316                        for(Content pd : m.getContent())
317                                vi.getFiles().put(pd.getFilename(), pd.toString());
318                        
319                        list.add(vi);
320                }
321                
322                return list;
323        }

076         * This configures the web request parameters binding, i.e., 
077         * conversion to the corresponding java types; for example,
078         * "neighborhood" is recognized as {@link GraphType#NEIGHBORHOOD},  
079         *  "protein" - {@link Protein} , etc.
080         *  Depending on the editor, illegal query parameters may result 
081         *  in an error or just NULL value.
082         * 
083         * @param binder
084         */
085        @InitBinder
086    public void initBinder(WebDataBinder binder) {
087        binder.registerCustomEditor(GraphType.class, new GraphTypeEditor());
088        binder.registerCustomEditor(Direction.class, new GraphQueryDirectionEditor());
089        binder.registerCustomEditor(LimitType.class, new GraphQueryLimitEditor());
090        binder.registerCustomEditor(OutputFormat.class, new OutputFormatEditor());
091        binder.registerCustomEditor(Class.class, new BiopaxTypeEditor());
092    }

095        /**
096         * This is a convenience method is to make cpath2 data 
097         * more suitable for LinkedData / Semantic Web, by 
098         * resolving cpath2-generated URIs (created in the data 
099         * warehouse and during validation and normalization). 
100         * Ideally, all URIs are to be resolvable URLs.
101         * 
102         * Normally, one should use #elementById(Get, BindingResult, Writer, HttpServletRequest, HttpServletResponse).
103         * 
104         * @param localId - the part of URI following xml:base
105         * 
106         * TODO return a summary page (type, name, some numbers and links to get the biopax/sif/gsea data) instead of the BioPAX content.
107         */
108        @RequestMapping(method=RequestMethod.GET, value="/{localId}")
109        public void cpathIdInfo(@PathVariable String localId, Writer writer, 
110                        HttpServletRequest request, HttpServletResponse response) 
111                                        throws Exception 
112        {
113                        /* A hack (specific to our normalizer and also 
114                         * might not work for all client links/browsers...
115                         * a better solution would be never generate tricky URIs,
116                         * containing encoded sharps, colons, spaces, etc.): 
117                         * the 'localId' parameter is usually un-encoded by the frameworks;
118                         * so we need to encode ":","#"," " back to 
119                         * %3A, %23, and "+" respectively, to get the original URI
120                         * (otherwise, if we simply combine localId and xml:base, the 
121                         * the resulting "URI" will be non-existing or wrong one)
122                         */
123                        if(localId.startsWith("#"))
124                                localId = localId.substring(1);
125                                
126                        if(localId.contains(":") || localId.contains("#") || localId.contains(" "))
127                                localId = localId
128                                                .replaceAll(":", "%3A")
129                                                        .replaceAll("#", "%23")
130                                                                .replaceAll(" ", "+");
131                        Get get = new Get();                    
132                        get.setUri(new String[]{xmlBase + localId});
133                        elementById(get, null, writer, request, response);                      
134        }

137        // Get by ID (URI) command
138    @RequestMapping("/get")
139    public void elementById(@Valid Get get, BindingResult bindingResult, 
140        Writer writer, HttpServletRequest request, HttpServletResponse response) 
141                throws IOException
142    {
143                //log events: command, format
144        Set<LogEvent> events = new HashSet<LogEvent>();
145        events.add(LogEvent.from(Cmd.GET));
146        
147        if(bindingResult != null &&  bindingResult.hasErrors()) {
148                errorResponse(Status.BAD_REQUEST, 
149                                errorFromBindingResult(bindingResult), request, response, events);
150        } else {
151                        OutputFormat format = get.getFormat();
152                        String[] uri = get.getUri();
153                        ServiceResponse result = service.fetch(format, uri);
154                        events.add(LogEvent.from(format));
155                        stringResponse(result, writer, request, response, events);
156                }
157    }

160        @RequestMapping("/top_pathways")
161    public @ResponseBody SearchResponse topPathways(
162                @RequestParam(required=false) String[] datasource, @RequestParam(required=false) String[] organism, 
163                HttpServletRequest request, HttpServletResponse response) throws IOException
164    {
165        Set<LogEvent> events = new HashSet<LogEvent>();
166        events.add(LogEvent.from(Cmd.TOP_PATHWAYS));
167        events.add(LogEvent.FORMAT_OTHER);
168                
169                ServiceResponse results = service.topPathways(organism, datasource);
170                
171                if(results instanceof ErrorResponse) {
172                        errorResponse(((ErrorResponse) results).getStatus(), 
173                                        ((ErrorResponse) results).toString(), request, response, events);
174                } else if(results.isEmpty()) {
175                        errorResponse(Status.NO_RESULTS_FOUND, 
176                                        "no hits", request, response, events);
177                } else {//return results
178                        //log to db
179                        SearchResponse hits = (SearchResponse) results;
180                events.addAll(LogEvent.fromProviders(hits.getProviders()));
181                service.log(events, clientIpAddress(request));
182                        return hits;
183                }
184                
185                return null;
186    }

189    @RequestMapping("/traverse")
190    public @ResponseBody ServiceResponse traverse(@Valid Traverse query, 
191        BindingResult bindingResult, HttpServletRequest request, HttpServletResponse response) 
192                throws IOException 
193    {
194        Set<LogEvent> events = new HashSet<LogEvent>();
195        events.add(LogEvent.from(Cmd.TRAVERSE));
196        events.add(LogEvent.FORMAT_OTHER);
197        
198        if(bindingResult.hasErrors()) {
199                errorResponse(Status.BAD_REQUEST, 
200                                errorFromBindingResult(bindingResult), request, response, events);
201        } else {
202                ServiceResponse sr = service.traverse(query.getPath(), query.getUri());
203                if(sr instanceof ErrorResponse) {
204                                errorResponse(((ErrorResponse) sr).getStatus(), 
205                                                ((ErrorResponse) sr).toString(), request, response, events);
206                        }
207                else if(sr == null || sr.isEmpty()) {
208                        errorResponse(Status.NO_RESULTS_FOUND, "no results found", request, response, events);
209                }
210                else {
211                        //log to db
212                        service.log(events, clientIpAddress(request));
213                        
214                        return sr;
215                        }
216        }
217        return null;
218    }

221        @RequestMapping("/graph")
222        public void graphQuery(@Valid Graph graph, BindingResult bindingResult, 
223                        Writer writer, HttpServletRequest request, HttpServletResponse response) throws IOException
224    {
225
226        Set<LogEvent> events = new HashSet<LogEvent>();
227        events.add(LogEvent.from(Cmd.GRAPH));
228                
229                //check for binding errors
230                if(bindingResult.hasErrors()) {
231                        errorResponse(Status.BAD_REQUEST, 
232                                        errorFromBindingResult(bindingResult), request, response, events);
233                        return;
234                } 
235                
236        // on parameter binding success, add a few more events to log/count -
237                events.add(LogEvent.from(graph.getKind()));
238        events.add(LogEvent.from(graph.getFormat()));
239                
240                ServiceResponse result;
241                
242                switch (graph.getKind()) {
243                case NEIGHBORHOOD:
244                        result = service.getNeighborhood(graph.getFormat(), graph.getSource(), 
245                                graph.getLimit(), graph.getDirection(), graph.getOrganism(), graph.getDatasource());
246                        break;
247                case PATHSBETWEEN:
248                        result = service.getPathsBetween(graph.getFormat(), graph.getSource(), 
249                                graph.getLimit(), graph.getOrganism(), graph.getDatasource());
250                        break;
251                case PATHSFROMTO:
252                        result = service.getPathsFromTo(graph.getFormat(), graph.getSource(), 
253                                graph.getTarget(), graph.getLimit(), graph.getOrganism(), graph.getDatasource());
254                        break;
255                case COMMONSTREAM:
256                        result = service.getCommonStream(graph.getFormat(), graph.getSource(), 
257                                graph.getLimit(), graph.getDirection(), graph.getOrganism(), graph.getDatasource());
258                        break;
259                default:
260                        // impossible (should have failed earlier)
261                        errorResponse(Status.INTERNAL_ERROR, 
262                                getClass().getCanonicalName() + " does not support " 
263                                        + graph.getKind(), request, response, events);                  
264                        return;
265                }
266                
267                stringResponse(result, writer, request, response, events);
268    }

271    @RequestMapping(value="/search")
272    public @ResponseBody ServiceResponse search(@Valid Search search, 
273                BindingResult bindingResult, HttpServletRequest request, HttpServletResponse response) 
274                        throws IOException
275    {           
276                //prepare to count following service assess events
277        Set<LogEvent> events = new HashSet<LogEvent>();
278        events.add(LogEvent.from(Cmd.SEARCH));
279        events.add(LogEvent.FORMAT_OTHER); //XML or JSON (not BioPAX, SIF, etc.)
280        //do NOT add yet for the 'datasource' filter values can be anything (hard to analyze)           
281                
282        if(bindingResult.hasErrors()) {
283                        errorResponse(Status.BAD_REQUEST, 
284                                errorFromBindingResult(bindingResult), 
285                                        request, response, events);
286                        return null;
287                } else {
288                        // get results from the service
289                        ServiceResponse results = service.search(
290                                        search.getQ(), search.getPage(), search.getType(),
291                                        search.getDatasource(), search.getOrganism());
292
293                        if(results instanceof ErrorResponse) {
294                                errorResponse(((ErrorResponse) results).getStatus(), 
295                                        ((ErrorResponse) results).toString(), 
296                                                request, response, events);
297                        } else if(results.isEmpty()) {
298                                errorResponse(Status.NO_RESULTS_FOUND, 
299                                                "no hits", request, response, events);
300                        } else {
301                                //count for all unique provider names from the ServiceResponse
302                                //TODO ? not sure we have to count these...
303                        events.addAll(LogEvent.fromProviders(
304                                        ((SearchResponse)results).getProviders()
305                                ));
306                                //log to db
307                        service.log(events, clientIpAddress(request));
308                                return results;
309                        }
310                        return null;
311                }
312        }

393 	private static List<Model> initPathwayModels() throws IOException {
394 		final List<Model> pathwayModels = new ArrayList<Model>();
395 		
396 		SimpleIOHandler reader = new SimpleIOHandler();
397 		Normalizer normalizer = new Normalizer();
398 		normalizer.setXmlBase(XML_BASE);
399 		reader.mergeDuplicates(true);
400 		Model model;
401 
402 		model = reader.convertFromOWL(resourceLoader
403 			.getResource("classpath:merge/pathwaydata1.owl").getInputStream());
404 		normalizer.normalize(model);
405 		pathwayModels.add(model);
406 		
407 		model = null;
408 		model = reader.convertFromOWL(resourceLoader
409 			.getResource("classpath:merge/pathwaydata2.owl").getInputStream());
410 		normalizer.normalize(model);
411 		pathwayModels.add(model);
412 		model = null;
413 		model = reader.convertFromOWL(resourceLoader
414 			.getResource("classpath:merge/pid_60446.owl").getInputStream());
415 		normalizer.normalize(model);
416 		pathwayModels.add(model); //PR P22932 caused the trouble
417 		model = null;
418 		model = reader.convertFromOWL(resourceLoader
419 			.getResource("classpath:merge/pid_6349.owl").getInputStream());
420 		normalizer.normalize(model);
421 		pathwayModels.add(model); //Xref for P01118 caused the trouble
422 		model = null;
423 		model = reader.convertFromOWL(resourceLoader
424 			.getResource("classpath:merge/hcyc.owl").getInputStream());
425 		normalizer.normalize(model);
426 		pathwayModels.add(model);
427 		model = null;
428 		model = reader.convertFromOWL(resourceLoader
429 			.getResource("classpath:merge/hcyc2.owl").getInputStream());
430 //		normalizer.normalize(model);
431 		pathwayModels.add(model);	
432 		
433 		return pathwayModels;
434 	}

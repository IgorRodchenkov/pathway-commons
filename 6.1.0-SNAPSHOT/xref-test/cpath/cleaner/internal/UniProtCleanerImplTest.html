38  import cpath.dao.CPathUtils;
39  import cpath.importer.Cleaner;

41  /**
42   * Test UniProt data cleaner.
43   */
44  @Ignore
45  public class UniProtCleanerImplTest {
46  
47      // some statics of accessions before cleaning
48      private static final String CALR3_HUMAN_BEFORE = "AC   Q96L12; Q96LN3; Q53TS5;";
49      private static final String CALRL_HUMAN_BEFORE = "AC   Q16602; A8K6G5; A8KAD3; Q53S02; Q53TS5;";
50  
51      // some statics of accessions after cleaning
52      // note, Q53TS5 is shared accession that will be removed after cleaning
53      private static final String CALR3_HUMAN_AFTER = "AC   Q96L12; Q96LN3;";
54      private static final String CALRL_HUMAN_AFTER = "AC   Q16602; A8K6G5; A8KAD3; Q53S02;";
55  
56  	/**
57  	 * Test method for {@link cpath.converter.internal.UniprotCleanerImpl#cleane(java.lang.String)}.
58  	 * @throws IOException 
59  	 */
60  	@Test
61  	public void testCleaner() throws IOException {
62          // read data from file and look for accessions before being cleaned
63          ByteArrayOutputStream os = new ByteArrayOutputStream();
64  		CPathUtils.unzip(new ZipInputStream(CPathUtils.LOADER
65  				.getResource("/test_uniprot_data.dat.zip").getInputStream()), os);
66          byte[] bytes = os.toByteArray();
67          String data = new String(bytes);
68          assertTrue(data.indexOf(CALR3_HUMAN_BEFORE) != -1);
69          assertTrue(data.indexOf(CALRL_HUMAN_BEFORE) != -1);
70  
71  		Cleaner cleaner = new UniProtCleanerImpl();
72  		os.reset();
73  		cleaner.clean(new ByteArrayInputStream(bytes), os);
74  		bytes = os.toByteArray();
75  		
76  		data = new String(bytes);
77          assertTrue(data.indexOf(CALR3_HUMAN_AFTER) != -1);
78          assertTrue(data.indexOf(CALRL_HUMAN_AFTER) != -1);
79  		       
80  		// dump owl for review
81  		String outFilename = getClass().getClassLoader().getResource("").getPath() 
82  			+ File.separator + "testCleanUniProt.out.dat";
83          Writer out = new OutputStreamWriter(new FileOutputStream(outFilename));
84          out.write(data);
85          out.close();
86  	}

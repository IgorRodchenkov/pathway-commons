62  public class UniprotConverterImplTest {
63  
64  	/**
65  	 * Test method for {@link cpath.converter.internal.UniprotConverterImpl#convert(java.io.InputStream)}.
66  	 * @throws IOException 
67  	 */
68  	@Test
69  	public void testConvert() throws IOException {
70  		
71  		//test the tricky FT pattern first
72  		assertEquals("AA-(test test)test", "AA-(test test)test (Bysimilarity)".replaceFirst("\\([^()]+?\\)$","").trim());	
73  		
74  		ByteArrayOutputStream bos = new ByteArrayOutputStream();
75  		CPathUtils.unzip(new ZipInputStream(new FileInputStream(
76  				getClass().getResource("/test_uniprot_data.dat.zip").getFile())), bos);
77  		bos.close();	
78  		String outFilename = getClass().getClassLoader().getResource("").getPath() 
79  				+ File.separator + "testConvertUniprot.out.owl";
80  		Converter converter = ImportFactory.newConverter("cpath.converter.internal.UniprotConverterImpl");
81  		converter.setXmlBase(null);
82  		converter.convert(new ByteArrayInputStream(bos.toByteArray()), new FileOutputStream(outFilename));
83  
84  		// read Model
85  		Model model = new SimpleIOHandler().convertFromOWL(new FileInputStream(outFilename));
86  		Set<ProteinReference> proteinReferences = model.getObjects(ProteinReference.class);
87  		assertEquals(10, proteinReferences.size());
88  		assertTrue(proteinReferences.iterator().next().getXref().iterator().hasNext());
89  				
90  		//
91  		ProteinReference pr = (ProteinReference) model.getByID("http://identifiers.org/uniprot/P27797");
92  		assertEquals(10, pr.getName().size()); //make sure this one is passed (important!)
93  		assertEquals("CALR_HUMAN", pr.getDisplayName());
94  		assertEquals("Calreticulin", pr.getStandardName());
95  //		System.out.println("CALR_HUMAN xrefs: " + pr.getXref().toString());
96  		assertEquals(10, pr.getXref().size()); //10, no duplicates
97  		assertEquals("http://identifiers.org/taxonomy/9606", pr.getOrganism().getRDFId());		
98  		
99  		// test MOD_RES features are created
100 		pr = (ProteinReference) model.getByID("http://identifiers.org/uniprot/P62158");
101 		assertTrue(pr.getName().contains("CALM2"));
102 		assertTrue(pr.getName().contains("CALM3"));
103 		assertNotNull(pr);
104 		
105 		assertEquals(9, pr.getEntityFeature().size());
106 		
107 		//check for a feature object by using URI generated the same way as it's in the converter:
108 		String mfUri = Normalizer.uri(model.getXmlBase(), null, pr.getDisplayName() + "_1", ModificationFeature.class);
109 		ModificationFeature mf = (ModificationFeature) model.getByID(mfUri);
110 		assertNotNull(mf);
111 		assertTrue(pr.getEntityFeature().contains(mf));
112 		SequenceLocation sl = mf.getFeatureLocation();
113 		assertTrue(sl instanceof SequenceSite);
114 		assertEquals(2, ((SequenceSite)sl).getSequencePosition());
115 		assertEquals("MOD_RES N-acetylalanine", mf.getModificationType().getTerm().iterator().next());
116 		
117 		//this is just to test for a bug in the DR text format parser...
118 		boolean rel = false;
119 		for(Xref x : pr.getXref()) {
120 			if("NCBI GENE".equalsIgnoreCase(x.getDb())) {
121 				rel = true;
122 				break;
123 			}
124 		}
125 		assertTrue(rel);
126 		//test if the correct RefSeq xrefs was created from the DR line despite having something ([bla-bla]) after the '.' at the end
127 		String uri = Normalizer.uri(model.getXmlBase(), "REFSEQ", "NP_001734_identity", RelationshipXref.class);
128 		assertNotNull(model.getByID(uri));
129 		//test if the correct Ensembl ENSG* xrefs was created from the DR line despite having something ([bla-bla]) after the '.' at the end
130 		uri = Normalizer.uri(model.getXmlBase(), "ENSEMBL", "ENSG00000143933_identity", RelationshipXref.class);
131 		assertNotNull(model.getByID(uri));
132 		uri = Normalizer.uri(model.getXmlBase(), "NCBI GENE", "801_identity", RelationshipXref.class);
133 		assertNotNull(model.getByID(uri));
134 		
135 		uri = Normalizer.uri(model.getXmlBase(), "REFSEQ", "NP_619650_identity", RelationshipXref.class);
136 		assertNotNull(model.getByID(uri));
137 		//but the parser should not create xrefs from for the last parts in DR like "...; -.", "; Homo sapiens.\n", etc.
138 		uri = Normalizer.uri(model.getXmlBase(), "REFSEQ", "-_identity", RelationshipXref.class);
139 		assertNull(model.getByID(uri));	
140 		uri = Normalizer.uri(model.getXmlBase(), "ENSEMBL", "Homo_sapiens_identity", RelationshipXref.class);
141 		assertNull(model.getByID(uri));	
142 		
143 		//total xrefs generated for P62158
144 		assertEquals(32, pr.getXref().size());
145 		
146 		//test for the following FT entry (two-line) was correctly parsed/converted:
147 		//FT   MOD_RES      45     45       Phosphothreonine; by CaMK4 (By
148 		//FT                                similarity).
149 		EntityFeature f = null;
150 		EntityFeature g = null;
151 		for(EntityFeature ef : pr.getEntityFeature()) {
152 			assertEquals(1, ef.getComment().size());
153 			if(ef.getComment().iterator().next().contains("Phosphothreonine; by CaMK4"))
154 				f = ef;
155 			if(ef.getComment().iterator().next().contains("AA-(test"))
156 				g = ef;
157 		}
158 		assertNotNull(f);
159 		assertTrue(f instanceof ModificationFeature);
160 		assertTrue(((ModificationFeature)f).getModificationType() instanceof SequenceModificationVocabulary);
161 		assertEquals("MOD_RES Phosphothreonine", ((ModificationFeature)f).getModificationType().getTerm().iterator().next());
162 		assertTrue(((ModificationFeature)f).getFeatureLocation() instanceof SequenceSite);
163 			
164 		//another special test for records like this one:
165 		//FT   MOD_RES       1      1       AA-(test test)test (By
166 		//FT                                similarity).
167 		assertNotNull(g);
168 		assertTrue(g instanceof ModificationFeature);
169 		assertTrue(((ModificationFeature)g).getModificationType() instanceof SequenceModificationVocabulary);
170 		assertEquals("MOD_RES AA-(test test)test", ((ModificationFeature)g).getModificationType().getTerm().iterator().next());
171 		assertTrue(((ModificationFeature)g).getFeatureLocation() instanceof SequenceSite);
172 	}

055/**
056 * A utility class to convert a BioPAX 
057 * L3 RDF/XML data stream or {@link Model} 
058 * to one of {@link OutputFormat}s  
059 * (including - to BioPAX L3 RDF/XML)
060 * 
061 * @author rodche
062 */
063public class BiopaxConverter {
064        private static final Logger log = LoggerFactory.getLogger(BiopaxConverter.class);
065        
066        private final Blacklist blacklist;
067                
068        /**
069         * Constructor.
070         * 
071         * @param blacklist of ubiquitous molecules to exclude (in some algorithms)
072         */
073        public BiopaxConverter(Blacklist blacklist)
074        {
075                this.blacklist = blacklist;
076        }
077
078 
079        /**
080     * Converts the BioPAX data into the other format.
081     * 
082     * @param m paxtools model
083     * @param format output format
084     * @param os
085     * @param args optional format-specific parameters
086     * 
087         * @throws IOException 
088     */
089    public void convert(Model m, OutputFormat format, OutputStream os, Object... args) 
090                throws IOException 
091    {
092                        switch (format) {
093                        case BIOPAX: //to OWL (RDF/XML)
094                                new SimpleIOHandler().convertToOWL(m, os);
095                                break;
096                        case BINARY_SIF:
097                                String db = null; //default: will use HGNC (Symbol)
098                                if (args.length > 0 && args[0] instanceof String)
099                                        db = (String) args[0];
100                                convertToBinarySIF(m, os, false, db);
101                                break;
102                        case EXTENDED_BINARY_SIF:
103                                db = null; //default: will use HGNC (Symbol)
104                                if (args.length > 0 && args[0] instanceof String)
105                                        db = (String) args[0];
106                                convertToBinarySIF(m, os, true, db);
107                                break;
108                        case GSEA:
109                                db = "uniprot"; //default
110                                if (args.length > 0 && args[0] instanceof String)
111                                        db = (String) args[0];
112                                convertToGSEA(m, os, db);
113                                break;
114            case SBGN:
115                                boolean doLayout = true;
116                                if (args.length > 0) {
117                                        doLayout = (args[0] instanceof Boolean)
118                                                        ? ((Boolean)args[0]).booleanValue() 
119                                                                : Boolean.parseBoolean(String.valueOf(args[0]));
120                                }
121                convertToSBGN(m, os, blacklist, doLayout);
122                break;
123                        default: throw new UnsupportedOperationException(
124                                        "convert, yet unsupported format: " + format);
125                        }
126    }    
127    
128    
129        /**
130     * Converts not too large BioPAX model 
131     * (e.g., a graph query result) to another format.
132     * 
133     * @param m a sub-model (not too large), e.g., a get/graph query result
134     * @param format output format
135     * @param args optional format-specific parameters
136     * 
137     * @return data response with the converted data (up to 1Gb utf-8 string) 
138     *                  or {@link ErrorResponse}.
139     */
140    public ServiceResponse convert(Model m, OutputFormat format, Object... args) 
141    {
142        if(m == null || m.getObjects().isEmpty()) {
143                        return new ErrorResponse(NO_RESULTS_FOUND, "Empty BioPAX Model");
144                }
145        
146                // otherwise, convert, return a new DataResponse
147        // (can contain up to ~ 1Gb unicode string data)
148        ByteArrayOutputStream baos = new ByteArrayOutputStream();
149        try {
150                convert(m, format, baos, args);                 
151                DataResponse dataResponse = new DataResponse();
152                        dataResponse.setData(baos.toString());          
153                        // extract and save data provider names
154                        dataResponse.setProviders(providers(m));                        
155                        return dataResponse;
156                }
157        catch (Exception e) {
158                return new ErrorResponse(INTERNAL_ERROR, e);
159                }
160    }
161
162
163    /**
164     * Converts a BioPAX Model to SBGN format.
165     *
166     * @param m
167     * @param stream
168     * @param blackList
169     * @param doLayout
170     * 
171     * @throws IOException
172     */
173    public void convertToSBGN(Model m, OutputStream stream, Blacklist blackList, boolean doLayout)
174                throws IOException
175        {
176        
177        L3ToSBGNPDConverter converter = new L3ToSBGNPDConverter(
178                        new ListUbiqueDetector(blackList.getListed()), null, doLayout);
179
180                converter.writeSBGN(m, stream);
181    }

183
184    /**
185         * Converts service results that contain 
186         * a not empty BioPAX Model to GSEA format.
187         * 
188     * @param m paxtools model
189     * @param stream
190         * @param outputIdType output identifiers type (db name, is data-specific, the default is UniProt)
191         * 
192         * @throws IOException 
193         */
194        public void convertToGSEA(Model m, OutputStream stream, String outputIdType) 
195                        throws IOException 
196        {       
197                if(outputIdType==null || outputIdType.isEmpty())
198                        outputIdType = "uniprot";
199                
200                // convert, replace DATA
201                GSEAConverter gseaConverter = new GSEAConverter(outputIdType, true);
202                gseaConverter.writeToGSEA(m, stream);
203        }
204
205        
206        /**
207         * Converts a not empty BioPAX Model (contained in the service bean) 
208         * to the SIF or <strong>single-file</strong> extended SIF format.
209         * 
210         * This method is primarily designed for the web service.
211         * 
212     * @param m biopax paxtools to convert
213     * @param out
214     * @param extended if true, calls SIFNX else - SIF
215         * @param db - either 'uniprot' or null (then HGNC symbols will be used by default)
216         * 
217         * @throws IOException 
218         */
219        public void convertToBinarySIF(Model m, OutputStream out, boolean extended, String db) 
220                        throws IOException 
221        {
222                SIFSearcher searcher;
223                if("uniprot".equalsIgnoreCase(db)) {
224                        CommonIDFetcher idFetcher = new CommonIDFetcher();
225                        idFetcher.setUseUniprotIDs(true);       
226                        searcher =  new SIFSearcher(idFetcher, SIFEnum.values());
227                } else { //hgnc
228                        searcher =  new SIFSearcher(SIFEnum.values());
229                }
230                
231                searcher.setBlacklist(blacklist);
232                                
233                if (extended) {
234                        Set<SIFInteraction> binaryInts = searcher.searchSIF(m);
235                        OldFormatWriter.write(binaryInts, out);
236                } else {
237                        searcher.searchSIF(m, out);
238                }
239                
240        }
241        
242        /**
243         * The list of datasources (data providers)
244         * the BioPAX model contains.
245         * 
246         * @param m
247         */
248        @SuppressWarnings("unchecked")
249        private Set<String> providers(Model m) {
250                Set<String> names = null;
251                
252                if(m != null) {
253                        Set<Provenance> provs = m.getObjects(Provenance.class);           
254                        if(provs!= null && !provs.isEmpty()) {
255                                names = new TreeSet<String>();
256                                for(Provenance prov : provs) {
257                                        String name = prov.getStandardName();
258                                        if(name != null)
259                                                names.add(name);
260                                        else {
261                                                name = prov.getDisplayName();
262                                                if(name != null)
263                                                        names.add(name);
264                                                else 
265                                                        log.warn("No standard|display name found for " + prov);
266                                        }
267                                }
268                        }
269                }
270                
271                return (names != null && !names.isEmpty()) 
272                                ? names : Collections.EMPTY_SET;
273        }

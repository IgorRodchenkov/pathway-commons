075    public static enum RelTypeVocab {
076        IDENTITY("identity", "http://identifiers.org/psimi/MI:0356", "MI", "MI:0356"),
077        SECONDARY_ACCESSION_NUMBER("secondary-ac", "http://identifiers.org/psimi/MI:0360", "MI", "MI:0360"),
078        ADDITIONAL_INFORMATION("see-also", "http://identifiers.org/psimi/MI:0361", "MI", "MI:0361"),
079        //next should work for rel. xrefs pointing to a protein but attached to a Gene, Dna*, Rna* biopax objects
080        GENE_PRODUCT("gene product", "http://identifiers.org/psimi/MI:0251", "MI", "MI:0251"),
081        SET_MEMBER("set member", "http://identifiers.org/psimi/MI:1341", "MI", "MI:1341"),
082        //next one is probably for chebi "is_a" relationships (when parent is a chemical class/concept rather than compound)
083        MULTIPLE_PARENT_REFERENCE("multiple parent reference", "http://identifiers.org/psimi/MI:0829", "MI", "MI:0829"),
084        ISOFORM_PARENT("isoform-parent", "http://identifiers.org/psimi/MI:0243", "MI", "MI:0243"),
085        ;

088        public final String uri;
089        public final String db;
090        public final String id;

095                this.db = db;

130    /**
131         * (non-Javadoc)
132         * @see cpath.importer.Premerger#premerge
133         */
134    public void premerge() {
135                // grab all metadata (initially, there are no pathway data files yet;
136                // but if premerge was already called, there are can be not empty dataFile 
137                // and result files for the corresp. metadata objects, which will be cleared anyway.)
138                // iterate over all metadata
139                for (Metadata metadata : service.metadata().findAll()) {
140                        // use filter if set (identifier and version)
141                        if(identifier != null) {
142                                if(!metadata.getIdentifier().equals(identifier))
143                                        continue;
144                        }
145                        
146                        try {   
147                                        log.info("premerge(), now processing " + metadata.getIdentifier() );
148                                        
149                                        // Try to instantiate the Cleaner now, and exit if it fails!
150                                        Cleaner cleaner = null; //reset to null!
151                                        String cl = metadata.getCleanerClassname();
152                                        if(cl != null && cl.length()>0) {
153                                                cleaner = ImportFactory.newCleaner(cl);
154                                                if (cleaner == null) {
155                                                        log.error("premerge(), failed to create the Cleaner: " + cl
156                                                                + "; skipping for this data source...");
157                                                        return; // skip this data entirely due to the error
158                                                }                       
159                                        } else {
160                                                log.info("premerge(), no Cleaner class was specified; continue...");    
161                                        }
162                                        
163                                        Converter converter = null;
164                                        cl = metadata.getConverterClassname();
165                                        if(cl != null && cl.length()>0) {
166                                                converter = ImportFactory.newConverter(cl);
167                                                if (converter == null) {
168                                                        log.error("premerge(), failed to create the Converter: " + cl
169                                                                + "; skipping for this data source...");
170                                                        return; // skip due to the error
171                                                } 
172                                                
173                                                // initialize
174                                                converter.setXmlBase(xmlBase);
175                                                
176                                        } else {
177                                                log.info("premerge(), no Converter class was specified; continue...");          
178                                        }
179                                                                                
180                                        // clear all existing output files, parse input files, reset counters, save.
181                                        log.debug("num. of data files before init, " + metadata.getIdentifier() + ": " + metadata.getContent().size());
182                                        metadata = service.init(metadata);
183                                        metadata.setPremerged(null);
184                                        
185                                        //load/re-pack/save orig. data
186                                        CPathUtils.analyzeAndOrganizeContent(metadata);
187                                        
188                                        // Premerge for each pathway data: clean, convert, validate, 
189                                        // and then update premergeData, validationResults db fields.
190                                        for (Content content : new HashSet<Content>(metadata.getContent())) {
191                                                try {                                   
192                                                        pipeline(metadata, content, cleaner, converter);
193                                                } catch (Exception e) {
194                                                        metadata.getContent().remove(content);
195                                                        log.warn("premerge(), removed " + content 
196                                                                        + " due to error", e);
197                                                }               
198                                        }
199                                        
200                                        // save/update validation status
201                                        metadata = service.save(metadata);
202                                        log.debug("premerge(), for " + metadata.getIdentifier() + 
203                                                ", saved " + metadata.getContent().size() + " files");
204                                
205                        } catch (Exception e) {
206                                log.error("premerge(): failed", e);
207                                e.printStackTrace();
208                        }
209                }
210        }

212        /**
213         * Builds a BioPAX Warehouse model using all available
214         * WAREHOUSE type data sources, builds id-mapping tables from 
215         * MAPPING type data sources, generates extra xrefs, and saves the 
216         * result model.
217         */
218        public void buildWarehouse() {
219                
220                Model warehouse = BioPAXLevel.L3.getDefaultFactory().createModel();
221                warehouse.setXmlBase(xmlBase);
222                
223                // iterate over all metadata
224                for (Metadata metadata : service.metadata().findAll()) 
225                {
226                        //skip not warehouse data
227                        if (metadata.getType() != METADATA_TYPE.WAREHOUSE)
228                                continue; 
229                        
230                        log.info("buildWarehouse(), adding data: " + metadata.getUri());                        
231                        InputStream inputStream;
232                        for(Content content : metadata.getContent()) {
233                                try {
234                                        inputStream = new GZIPInputStream(new FileInputStream(content.normalizedFile()));
235                                        Model m = new SimpleIOHandler(BioPAXLevel.L3).convertFromOWL(inputStream);
236                                        m.setXmlBase(xmlBase);
237                                        warehouse.merge(m);
238                                } catch (IOException e) {
239                                        log.error("buildWarehouse, skip for " + content.toString() + 
240                                                "; failed to read/merge from " + content.convertedFile(), e);
241                                        continue;
242                                }
243                        }
244                }
245                warehouse.repair();
246                
247                //clear all id-mapping tables
248                service.mapping().deleteAll();
249                
250                // Using the Warehouse, generate the id-mapping tables
251                // from that BioPAX model:
252                buildIdMappingFromWarehouse(warehouse);
253                        
254                // Next, process all extra MAPPING data files, build, save tables
255                for (Metadata metadata : service.metadata().findAll()) 
256                {
257                        //skip not id-mapping data
258                        if (metadata.getType() != METADATA_TYPE.MAPPING)
259                                continue; 
260                        
261                        log.info("buildWarehouse(), adding id-mapping: " + metadata.getUri());
262                        for(Content content : metadata.getContent()) {
263                                try {
264                                        Set<Mapping> mappings = simpleMapping(content, new GZIPInputStream(
265                                                        new FileInputStream(content.originalFile())));
266                                        saveIgnoringDuplicates(mappings);
267                                } catch (Exception e) {
268                                        log.error("buildWarehouse(), failed to get id-mapping, " +
269                                                        "using: " + content.toString(), e);
270                                        continue;
271                                }
272                        }
273                }
274                
275                // add more relationship xrefs to the warehouse ERs; 
276                // this makes full-text search for ER by some secondary IDs possible
277                log.info("buildWarehouse(), adding more Xrefs to ERs using id-mapping...");
278                for(EntityReference er : new HashSet<EntityReference>(warehouse.getObjects(EntityReference.class))) 
279                {
280                        Assert.isTrue(er.getRDFId().contains("/chebi/") || er.getRDFId().contains("/uniprot/"),
281                                        er + " - warehouse ER is neither PR nor SMR (bug)!");
282                        addRelXrefsToWarehouseEntityRef(warehouse, er);
283                }
284                
285                // save to compressed file
286                String whFile  = CPathSettings.getInstance().warehouseModelFile();
287                log.info("buildWarehouse(), creating Warehouse BioPAX archive: " + whFile);             
288                try {
289                        new SimpleIOHandler(BioPAXLevel.L3).convertToOWL(warehouse, 
290                                        new GZIPOutputStream(new FileOutputStream(whFile)));
291                } catch (IOException e) {
292                        log.error("buildWarehouse(), failed", e);
293                }
294                
295                //Don't persist (do later after Merger)
296                log.info("buildWarehouse(), done.");
297        }

300        private void saveIgnoringDuplicates(Set<Mapping> mappings) {
301                for(Mapping mapping : mappings) {
302                        try {
303                                service.saveIfUnique(mapping);
304                        } catch (DataIntegrityViolationException e) {} //ignore same entries
305                }
306        }

309        private void addRelXrefsToWarehouseEntityRef(Model warehouse, EntityReference er) 
310        {
311                final String primaryDb = (er instanceof ProteinReference) ? "UNIPROT" : "CHEBI"; 
312                final String primaryId = CPathUtils.idfromNormalizedUri(er.getRDFId());
313                
314                //reverse id-mapping (from the primary db/id to all db/id entries that map to the primary pair)
315                List<Mapping> map = service.mapping().findByDestIgnoreCaseAndDestId(primaryDb, primaryId);                
316                for(Mapping m : map) 
317                {
318                        Assert.isTrue(m.getDest().equals(primaryDb) && m.getDestId().equals(primaryId), 
319                                "findByDestIgnoreCaseAndDestId result contains mappings with different primary db/id (bug!)");
320                                        
321                        //find the unif.xref by the normalized URI, if exists
322                        final String uxUri = Normalizer.uri(xmlBase, m.getSrc(), m.getSrcId(), UnificationXref.class);
323                        UnificationXref ux = (UnificationXref) warehouse.getByID(uxUri);                        
324                        if(ux != null && er.getXref().contains(ux)) 
325                                continue; //skip existing equivalet unif. xref
326
327                        //otherwise - find/make special rel.xref
328                        RelationshipXref rx = findOrCreateRelationshipXref(
329                                        RelTypeVocab.IDENTITY, m.getSrc(), m.getSrcId(), warehouse);                    
330                        er.addXref(rx); 
331                }
332        }

335        /**
336         * Creates mapping objects
337         * from a simple two-column (tab-separated) text file, 
338         * where the first line contains standard names of 
339         * the source and target ID types, and on each next line -
340         * source and target IDs, respectively.
341         * Currently, only ChEBI and UniProt are supported
342         * (valid) as the target ID type.
343         * 
344         * This is a package-private method, mainly for jUnit testing
345         * (not API).
346         * 
347         * @param content
348         * @param inputStream
349         * @return
350         * @throws IOException 
351         */
352        Set<Mapping> simpleMapping(Content content, InputStream inputStream) 
353                        throws IOException 
354        {
355                Set<Mapping> mappings = new HashSet<Mapping>();
356                
357                BufferedReader reader = new BufferedReader(new InputStreamReader(inputStream));                         
358                String line = reader.readLine(); //get the first, title line
359                String head[] = line.split("\t");
360                assert head.length == 2 : "bad header";
361                String from = head[0].trim().toUpperCase();
362                String to = head[1].trim().toUpperCase();
363                while ((line = reader.readLine()) != null) {
364                        String pair[] = line.split("\t");
365                        String srcId = pair[0].trim();
366                        String tgtId = pair[1].trim();
367                        mappings.add(new Mapping(from, srcId, to, tgtId));
368                }
369                reader.close();
370                
371                return mappings;
372        }

375        /**
376         * Extracts id-mapping information (name/id -> primary id) 
377         * from the Warehouse entity references's xrefs to the mapping tables.
378         * 
379         * @param warehouse target model
380         * @param exportAmbiguousMappings whether to file errors or not (ambiguous id mappings)
381         */
382        private void buildIdMappingFromWarehouse(Model warehouse) {             
383                log.info("buildIdMappingFromWarehouse(), updating id-mapping " +
384                                "tables by analyzing the warehouse data...");
385                                
386                //Generates Mapping tables (objects) using ERs:
387                //a. ChEBI secondary IDs, PUBCHEM Compound, InChIKey, chem. name - to primary CHEBI AC;
388                //b. UniProt secondary IDs, RefSeq, NCBI Gene, etc. - to primary UniProt AC.
389                final Set<Mapping> mappings = new HashSet<Mapping>();
390                
391                // for each ER, using xrefs, map other identifiers to the primary accession
392                for(EntityReference er : warehouse.getObjects(EntityReference.class)) 
393                {       
394                        String destDb = null;
395                        if(er instanceof ProteinReference)
396                                destDb = "UNIPROT";
397                        else if(er instanceof SmallMoleculeReference)
398                                destDb = "CHEBI";
399                        
400                        //extract the primary id from the standard (identifiers.org) URI
401                        final String ac = CPathUtils.idfromNormalizedUri(er.getRDFId());
402                        
403                        for(Xref x : er.getXref()) {
404                                // By design (warehouse), there are unif. and rel. xrefs added 
405                                // by the Uniprot Converter, and we will uses those, 
406                                // skipping publication and illegal xrefs:
407                                if(x.getDb() != null && x.getId() != null 
408                                        && 
409                                        (// any unification xref
410                                                (x instanceof UnificationXref) 
411                                                || // or for any typed relationship xref - 
412                                                ( (x instanceof RelationshipXref) 
413                                                          && ((RelationshipXref)x).getRelationshipType()!=null
414                                                          && 
415                                                          (     //use any identity type relationship xref
416                                                                ((RelationshipXref)x).getRelationshipType().getRDFId().endsWith(RelTypeVocab.IDENTITY.id)
417                                                                || // or any secondary accession type relationship xref
418                                                                ((RelationshipXref)x).getRelationshipType().getRDFId().endsWith(RelTypeVocab.SECONDARY_ACCESSION_NUMBER.id)
419                                                          )
420                                                )
421                                        )
422                                ) {
423                                        String id = x.getId();
424                                        String srcDb = x.getDb().toUpperCase();
425                                        
426                                        if(srcDb.startsWith("UNIPROT") || srcDb.startsWith("SWISSPROT"))
427                                                srcDb = "UNIPROT"; //consider 'uniprot*' source IDs/names as 'UNIPROT' for simplicity
428                                        
429                                        mappings.add(new Mapping(srcDb, id, destDb, ac));
430                                }
431                        }
432
433                        if(er instanceof SmallMoleculeReference) {
434                                SmallMoleculeReference smr = (SmallMoleculeReference) er;
435                                //map some names (display and std.)
436                                String name = smr.getDisplayName().toLowerCase();
437                                mappings.add(new Mapping("CHEMICAL NAME", name, destDb, ac));   
438                                //skip other names (and standardName) as they can be too long...
439                        }
440                        
441                        if(er instanceof ProteinReference) {
442                                ProteinReference pr = (ProteinReference) er;
443                                //map unofficial IDs, e.g., CALM_HUMAN, too
444                                String name = pr.getDisplayName().toUpperCase();
445                                mappings.add(new Mapping("UNIPROT", name, destDb, ac)); 
446                        }
447                }
448
449                saveIgnoringDuplicates(mappings);
450                
451                log.info("buildIdMappingFromWarehouse(), exitting...");
452        }

455        /**
456         * Given Content undergoes clean/convert/validate/normalize data pipeline.
457         * 
458         * @param metadata about the data provider
459         * @param content provider's pathway data (file) to be processed and modified
460         * @param cleaner data specific cleaner class (to apply before the validation/normalization)
461         * @param converter data specific to BioPAX L3 converter class
462         * @throws IOException 
463         */
464        private void pipeline(final Metadata metadata, final Content content, 
465                        Cleaner cleaner, Converter converter) throws IOException 
466        {       
467                final String info = content.toString();
468                
469                InputStream dataStream = new GZIPInputStream(new FileInputStream(content.originalFile()));
470                
471                //Clean the data, i.e., apply data-specific "quick fixes".
472                if(cleaner != null) {
473                        log.info("pipeline(), cleaning " + info + " with " + cleaner.getClass());
474                        OutputStream os = new GZIPOutputStream(new FileOutputStream(content.cleanedFile()));
475                        cleaner.clean(dataStream, os); //os must be closed inside
476                        //re-assign the input data stream
477                        dataStream = new GZIPInputStream(new FileInputStream(content.cleanedFile())); 
478                }
479                
480                if(metadata.getType() == METADATA_TYPE.MAPPING) {
481                        dataStream.close();
482                        return; //all done about the id-mappingdata (no need to convert/normalize)
483                }
484                
485                //Convert data to BioPAX L3 if needed (generate the 'converted' output file in any case)
486                if (converter != null) {
487                        log.info("pipeline(), converting " + info + " with " + converter.getClass());                                   
488                        OutputStream os = new GZIPOutputStream(new FileOutputStream(content.convertedFile()));
489                        converter.convert(dataStream, os);//os must be closed inside
490                        dataStream = new GZIPInputStream(new FileInputStream(content.convertedFile()));                 
491                }
492                
493                //here, the 'dataStream' will read either from the orig., cleaned, or converted file,
494                //(depending on cleaner/converter availability above)
495                
496                
497                //Go validate and normalize the pathway data
498                log.info("pipeline(), validating pathway data " + info);
499
500                /* Validate, auto-fix, and normalize (incl. convesion to L3): 
501                 * e.g., synonyms in xref.db may be replaced 
502                 * with the primary db name, as in Miriam, etc.
503                 */
504                //use a file instead of String for the RDF/XML data (which can be >2Gb and fail!)
505                Validation v = checkAndNormalize(info, dataStream, metadata, content.normalizedFile());
506
507                //save report data
508                content.saveValidationReport(v);
509
510                // count critical not fixed error cases (ignore warnings and fixed ones)
511                int noErrors = v.countErrors(null, null, null, null, true, true);
512                log.info("pipeline(), summary for " + info
513                                + ". Critical errors found:" + noErrors + ". " 
514                                + v.getComment().toString() + "; " + v.toString());
515
516                if(noErrors > 0) 
517                        content.setValid(false); 
518                else 
519                        content.setValid(true);
520        }

523        /**
524         * Validates, fixes, and normalizes given pathway data.
525         *
526         * @param title short description
527         * @param biopaxStream BioPAX OWL
528         * @param metadata data provider's metadata
529         * @param
530         * @return the object explaining the validation/normalization results
531         */
532        private Validation checkAndNormalize(String title, InputStream biopaxStream, Metadata metadata, String outFileName) 
533        {       
534                // create a new empty validation (options: auto-fix=true, report all) and associate with the model
535                Validation validation = new Validation(new IdentifierImpl(), 
536                                title, true, Behavior.WARNING, 0, null); // sets the title
537                
538                // configure Normalizer
539                Normalizer normalizer = new Normalizer();
540                // set cpath2 xml:base for the normalizer to use instead of the model's one (important!)
541                normalizer.setXmlBase(xmlBase);
542                // to infer/auto-fix biopax properties
543                normalizer.setFixDisplayName(true); // important
544                normalizer.setInferPropertyDataSource(false); // not important since we started generating Provenance from Metadata
545                normalizer.setInferPropertyOrganism(true); // important (for filtering by organism)
546                normalizer.setDescription(title);
547                
548                // because errors are also reported during the import (e.g., syntax)
549                try {
550                        validator.importModel(validation, biopaxStream);                        
551                        validator.validate(validation);
552                        // unregister the validation object 
553                        validator.getResults().remove(validation);
554                        
555                        // normalize
556                        log.info("checkAndNormalize, now normalizing pathway data "     + title);
557                        Model model = (Model) validation.getModel();
558                        normalizer.normalize(model);
559                        
560                        // (in addition to normalizer's job) find existing or create new Provenance 
561                        // from the metadata to add it explicitly to all entities -
562                        metadata.setProvenanceFor(model);
563                        
564                        OutputStream out = new GZIPOutputStream(new FileOutputStream(outFileName));
565                        (new SimpleIOHandler(model.getLevel())).convertToOWL(model, out);
566                        
567                } catch (Exception e) {
568                        throw new RuntimeException("checkAndNormalize(), " +
569                                "Failed " + title, e);
570                }
571                
572                return validation;
573        }

576        /**
577         * Given relationship type CV 'term' and target biological 'db' and 'id', 
578         * finds or creates a new relationship xref (and its controlled vocabulary) in the model.
579         * 
580         * Note: the corresponding CV does not have a unification xref
581         * (this method won't validate; so, non-standard CV terms can be used).
582         * 
583         * @param vocab relationship xref type
584         * @param model a biopax model where to find/add the xref
585         */
586        public static RelationshipXref findOrCreateRelationshipXref(
587                        RelTypeVocab vocab, String db, String id, Model model) 
588        {
589                Assert.notNull(vocab);
590                
591                RelationshipXref toReturn = null;
592
593                String uri = Normalizer.uri(model.getXmlBase(), db, id + "_" + vocab.toString(), RelationshipXref.class);
594                if (model.containsID(uri)) {
595                        return (RelationshipXref) model.getByID(uri);
596                }
597
598                // create a new relationship xref
599                toReturn = model.addNew(RelationshipXref.class, uri);
600                toReturn.setDb(db);
601                toReturn.setId(id);
602
603                // create/add the relationship type vocabulary
604                String relTypeCvUri = vocab.uri; //identifiers.org standard URI
605                RelationshipTypeVocabulary rtv = (RelationshipTypeVocabulary) model.getByID(relTypeCvUri);              
606                if (rtv == null) {
607                        rtv = model.addNew(RelationshipTypeVocabulary.class, relTypeCvUri);
608                        rtv.addTerm(vocab.term);
609                        //add the unif.xref
610                        uri = Normalizer.uri(model.getXmlBase(), vocab.db, vocab.id, UnificationXref.class);
611                        UnificationXref rtvux = (UnificationXref) model.getByID(uri);

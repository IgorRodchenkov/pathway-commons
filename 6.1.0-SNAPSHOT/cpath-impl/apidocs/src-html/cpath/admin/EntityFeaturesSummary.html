001package cpath.admin;
002
003import java.util.HashSet;
004import java.util.Set;
005
006import org.biopax.paxtools.model.Model;
007import org.biopax.paxtools.model.level3.BindingFeature;
008import org.biopax.paxtools.model.level3.EntityFeature;
009import org.biopax.paxtools.model.level3.ExperimentalForm;
010import org.biopax.paxtools.model.level3.ModificationFeature;
011import org.biopax.paxtools.model.level3.PhysicalEntity;
012import org.biopax.paxtools.model.level3.Provenance;
013
014import cpath.service.Analysis;
015
016/**
017 * Prints out Ð° summary of the main merged BioPAX model
018 * (just for development and debugging).
019 * 
020 * This class is not an essential part of this system; 
021 * it's to be optionally called using the cpath-cli.sh script,
022 * i.e., via {@link Admin} '-run-analysis' console command.
023 * 
024 * This uses a Java option: cpath.analysis.filter.metadataids=id1,id2,.. 
025 * (if not defined, all data are analyzed; IDs are cpath2 metadata/datasource identifiers,
026 * i.e., not providers' names)
027 * 
028 * @author rodche
029 *
030 */
031public final class EntityFeaturesSummary implements Analysis<Model> {
032        
033        public static final String JAVA_OPTION_DATASOURCES = "cpath.analysis.filter.metadataids";
034        
035
036        public void execute(Model model) {
037                
038                final Set<String> providerURIs = new HashSet<String>();
039                final String javaPropertyDatasources = System.getProperty(JAVA_OPTION_DATASOURCES);
040                if(javaPropertyDatasources!=null && !javaPropertyDatasources.isEmpty())
041                        for(String id : javaPropertyDatasources.split(","))
042                                providerURIs.add(model.getXmlBase()+id);
043                
044                final Set<ExperimentalForm> allExpForms = model.getObjects(ExperimentalForm.class);
045                                
046                for(EntityFeature ef : model.getObjects(EntityFeature.class)) {
047                        Set<String> providers = new HashSet<String>();
048                        Set<String> eforms = new HashSet<String>();
049                        Set<PhysicalEntity> pes = new HashSet<PhysicalEntity>();
050                        pes.addAll(ef.getFeatureOf());
051                        pes.addAll(ef.getNotFeatureOf());
052                        
053                        boolean passedProvidersFilter = (providerURIs.isEmpty())? true : false;
054                        for(PhysicalEntity pe : pes) {
055                                for(Provenance ds : pe.getDataSource()) {
056                                        providers.add(ds.getDisplayName());
057                                        if(providerURIs.contains(ds.getRDFId()))
058                                                passedProvidersFilter = true;
059                                }
060                        }
061                        if(!passedProvidersFilter && !providers.isEmpty()) 
062                                continue;
063                        
064                        //cannot easily get owner ExperimentalForm objects (no experimentalFeatureOf property exists...)
065                        for(ExperimentalForm xf : allExpForms) {
066                                if(xf.getExperimentalFeature().contains(ef))
067                                        eforms.add(xf.getRDFId());
068                        }       
069                        
070                        //output                        
071                        System.out.printf("%s\t%s\tpe providers: %s\tlocation: %s ", 
072                                        ef.getModelInterface().getSimpleName(), ef.getRDFId(),
073                                        providers, ef.getFeatureLocation(), ef.getFeatureLocationType());
074                        
075                        if(ef.getFeatureLocationType() != null)
076                                System.out.printf("%s", ef.getFeatureLocationType().getXref());
077                        
078                        System.out.print("\t");
079                        
080                        if(!eforms.isEmpty())
081                                System.out.printf("exp. forms: %s\t", eforms);
082                        
083                        if(ef instanceof BindingFeature) {
084                                BindingFeature bf = (BindingFeature) ef;
085                                System.out.printf("bindsTo: %s\t", bf.getBindsTo());
086                        }
087                        
088                        if(ef instanceof ModificationFeature) {
089                                ModificationFeature mf = (ModificationFeature) ef;
090                                System.out.printf("modif. type: %s\t", mf.getModificationType());
091                                if(mf.getModificationType() != null)
092                                        System.out.printf("%s\t", mf.getModificationType().getXref());
093                        }
094                        
095                        System.out.println();
096                }
097        }
098
099}

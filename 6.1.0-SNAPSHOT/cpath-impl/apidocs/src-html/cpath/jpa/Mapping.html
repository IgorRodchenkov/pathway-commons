001package cpath.jpa;
002
003import javax.persistence.*;
004
005import org.hibernate.annotations.DynamicInsert;
006import org.hibernate.annotations.DynamicUpdate;
007import org.hibernate.annotations.Index;
008import org.springframework.util.Assert;
009
010/**
011 * Id-mapping Entity.
012 * 
013 * @author rodche
014 */
015@Entity
016@DynamicUpdate
017@DynamicInsert
018@Table(name="mapping", uniqueConstraints = @UniqueConstraint(columnNames = {"src", "srcId", "dest", "destId"}))
019@org.hibernate.annotations.Table(appliesTo = "mapping",
020        indexes = {
021                @Index(name="dest_index", columnNames = "dest"),
022                @Index(name="dest_destId_index", columnNames = {"dest", "destId"}),
023                @Index(name="srcId_dest_index", columnNames = {"srcId", "dest"}),
024                @Index(name="src_srcId_dest_index", columnNames = {"src", "srcId", "dest"}),
025        }
026)
027public final class Mapping {
028                
029        @Id
030        @GeneratedValue(strategy=GenerationType.AUTO)
031        private Long id;
032        
033        @Column(nullable=false)
034        private String src; 
035        
036        @Column(nullable=false)
037        private String dest;
038                
039        @Column(nullable=false)
040        private String srcId; 
041        
042        @Column(nullable=false)
043        private String destId;
044        
045        /**
046         * Default Constructor.
047         */
048        public Mapping() {}
049
050 
051    public Mapping(String src, String srcId, String dest, String destId) 
052    {
053        Assert.hasText(src);
054        Assert.hasText(srcId);
055        Assert.hasText(dest);
056        Assert.hasText(destId);
057        
058        this.src = src.toUpperCase();
059        this.dest = dest.toUpperCase();
060        this.srcId = srcId;
061        this.destId = destId;
062    }
063    
064    
065    Long getId() {
066        return id;
067    }
068    
069
070        public String getSrc() {
071                return src;
072        }
073        void setSrc(String src) {
074                this.src = src.toUpperCase();
075        }
076
077
078        public String getDest() {
079                return dest;
080        }
081        void setDest(String dest) {
082                this.dest = dest.toUpperCase();
083        }
084
085
086        public String getSrcId() {
087                return srcId;
088        }
089        void setSrcId(String srcId) {
090                this.srcId = srcId;
091        }
092
093
094        public String getDestId() {
095                return destId;
096        }
097        void setDestId(String destId) {
098                this.destId = destId;
099        }
100
101
102        @Override
103    public String toString() {
104        return "Mapping from " + src + ", " + srcId + " to " + dest + ", " + destId;
105    } 
106}

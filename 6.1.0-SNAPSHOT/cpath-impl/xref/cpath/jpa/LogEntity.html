4   import javax.persistence.AttributeOverride;
5   import javax.persistence.AttributeOverrides;
6   import javax.persistence.Column;
7   import javax.persistence.Embedded;
8   import javax.persistence.Entity;
9   import javax.persistence.Table;
10  import javax.persistence.UniqueConstraint;
11  
12  import org.hibernate.annotations.DynamicInsert;
13  import org.hibernate.annotations.DynamicUpdate;
14  import org.springframework.util.Assert;
15  
16  
17  
18  /**
19   * @author rodche
20   *
21   */
22  @Entity
23  @DynamicUpdate
24  @DynamicInsert
25  //uniqueConstraints: 'type' is not listed because 'name' is enough (should never use same log name with different types)
26  @Table(uniqueConstraints = @UniqueConstraint(columnNames = {"date", "name", "addr"}))
27  public class LogEntity extends AbstractEntity {
28  		
29  	@Embedded
30  	@AttributeOverrides({
31  	    @AttributeOverride(name="type", column=@Column(name="type")),
32  	    @AttributeOverride(name="name", column=@Column(name="name"))
33  	})
34  	private LogEvent event;
35  	
36  	@Column
37  	private String addr;
38  	
39  	@Embedded
40  	@AttributeOverrides({
41  	    @AttributeOverride(name="country", column=@Column(name="country")),
42  	    @AttributeOverride(name="region", column=@Column(name="region")),
43  	    @AttributeOverride(name="city", column=@Column(name="city"))
44  	})
45  	private Geoloc geoloc;
46  
47  	@Column(nullable=false)
48  	private String date;
49  	
50  	@Column(nullable=false)
51  	private Long count;	
52  	
53  	public LogEntity() {
54  	}
55  	
56  	/**
57  	 * 
58  	 * @param date ISO date (yyyy-MM-dd)
59  	 * @param event
60  	 * @param ipAddress
61  	 */
62  	public LogEntity(String date, LogEvent event, String ipAddress) {
63  		Assert.notNull(event);
64  		Assert.notNull(event.getType());
65  		Assert.notNull(event.getName());
66  		Assert.notNull(date);
67  		Assert.notNull(ipAddress);
68  		
69  		setEvent(event);
70  		setCount(0L);
71  		setDate(date);
72  			
73  		setAddr(ipAddress);
74  		geoloc = Geoloc.fromIpAddress(ipAddress);
75  		setGeoloc(geoloc);
76  	}
77  	
78  	
79  	public String getAddr() {
80  		return addr;
81  	}
82  	public void setAddr(String addr) {
83  		this.addr = addr;
84  	}	
85  	
86  	public Long getCount() {
87  		return count;
88  	}
89  	public void setCount(Long count) {
90  		this.count = count;
91  	}
92  
93  	public String getDate() {
94  		return date;
95  	}
96  	public void setDate(String date) {
97  		this.date = date;
98  	}
99  	
100 	public Geoloc getGeoloc() {
101 		return geoloc;
102 	}
103 	public void setGeoloc(Geoloc geoloc) {
104 		this.geoloc = geoloc;
105 	}
106 
107 	public LogEvent getEvent() {
108 		return event;
109 	}
110 	public void setEvent(LogEvent event) {
111 		this.event = event;
112 	}
113 	
114 }

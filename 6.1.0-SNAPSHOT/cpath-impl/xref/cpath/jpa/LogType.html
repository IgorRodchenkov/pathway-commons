6   /**
7    * Categories of the data download/access events. 
8    * <p/>
9    * In fact, these are not mutually
10   * exclusive types, for a web request/response usually
11   * counts more than once: once, for sure, in the "total" 
12   * category but might also - in "providers", "commands", 
13   * "formats", or "errors" at the same time.
14   * 
15   * @author rodche
16   *
17   */
18  public enum LogType {
19  	PROVIDER("All Providers"),
20  	COMMAND("All Web Commands"),
21  	FORMAT("All Output Formats"),
22  	FILE("All Files"),
23  	ERROR("All Errors"),
24  	TOTAL("All");
25  	
26  	public final String description;
27  	
28  	private LogType(String description) {
29  		this.description = description;
30  	}
31  }

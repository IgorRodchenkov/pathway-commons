31  /**
32   * Implementation of Cleaner interface for Reactome data. 
33   * 
34   * Can normalize URIs for Reactome Entity class objects 
35   * to http://identifiers.org/reactome/REACT_* form
36   * if the unification xref with the stable Reactome ID is attached. 
37   * TODO add/remove features as needed, as reactome versions change...
38   */
39  final class ReactomeCleanerImpl implements Cleaner {
40  	
41  	// logger
42      private static Logger log = LoggerFactory.getLogger(ReactomeCleanerImpl.class);
43  
44      public void clean(InputStream data, OutputStream cleanedData)
45  	{	
46  		// create bp model from dataFile
47  		SimpleIOHandler simpleReader = new SimpleIOHandler(BioPAXLevel.L3);
48  		Model model = simpleReader.convertFromOWL(data);
49  		log.info("Cleaning Reactome data, this may take some time, please be patient...");
50  
51  		// (why the data provider could not always do this?)	
52  		// Normalize Entity URIs using Reactome stable id, where possible
53  		// (not required for utility class objects)
54  		Set<Entity> entities = new HashSet<Entity>(model.getObjects(Entity.class));
55  		Map<String, Entity> newUriToEntityMap = new HashMap<String, Entity>();
56  		for(Entity ent : entities) {
57  			Set<UnificationXref> uxrefs = new ClassFilterSet<Xref, UnificationXref>(
58  					new HashSet<Xref>(ent.getXref()), UnificationXref.class);			
59  			for(UnificationXref x : uxrefs) {
60  				if(x.getId() != null && x.getId().startsWith("REACT_")) {
61  					String id = x.getId();
62  					if(x.getIdVersion() != null && !x.getId().contains(".")) 
63  						id += "." + x.getIdVersion();
64  					
65  					String uri = "http://identifiers.org/reactome/" + id;
66  					
67  					if(!model.containsID(uri) && !newUriToEntityMap.containsKey(uri)) {
68  						newUriToEntityMap.put(uri, ent); //collect to replace URIs later (below)
69  					} else { //shared unification xref bug
70  						log.warn("Fixing Reactome: " + x.getId() + 
71  							" unification xref is shared by several entities: "
72  								+ x.getXrefOf());
73  						
74  						String rxUri = model.getXmlBase() + RelationshipXref.class.getSimpleName() + id;
75  						RelationshipXref rx = (RelationshipXref) model.getByID(rxUri);
76  						if(rx == null) {
77  							rx = model.addNew(RelationshipXref.class, rxUri);
78  							rx.setDb(x.getDb());
79  							rx.setId(id);
80  							rx.setIdVersion(x.getIdVersion());
81  							rx.setDbVersion(x.getDbVersion());
82  						}
83  						
84  						for(XReferrable owner : new HashSet<XReferrable>(x.getXrefOf())) {
85  							if(owner.equals(newUriToEntityMap.get(uri)))
86  								continue; //keep the entity to be updated unchanged
87  							owner.removeXref(x);
88  							owner.addXref(rx);
89  						}						
90  					}
91  					break; //skip the rest of xrefs, if any (must not have >1 REACT_* unif. xref for the same entity)
92  				}
93  			}
94  		}
95  		
96  		// set standard URIs for selected entities
97  		for(String uri : newUriToEntityMap.keySet())
98  			CPathUtils.replaceID(model, newUriToEntityMap.get(uri), uri);
99  		
100 		// All Conversions in Reactome are LEFT-TO-RIGH, 
101 		// unless otherwise was specified (confirmed with Guanming Wu, 2013/12)
102 		Set<Conversion> conversions = new HashSet<Conversion>(model.getObjects(Conversion.class));
103 		for(Conversion ent : conversions) {
104 			if(ent.getConversionDirection() == null)
105 				ent.setConversionDirection(ConversionDirectionType.LEFT_TO_RIGHT);
106 		}
107 		
108 		// Remove unstable UnificationXrefs like "Reactome Database ID Release XX"
109 		// if there is a stable one in the same object
110 		Set<Xref> xrefsToRemove = new HashSet<Xref>();
111 		for(Xref xref: new HashSet<Xref>(model.getObjects(Xref.class))) {
112 			if(xref.getDb() != null && xref.getDb()
113 				.toLowerCase().startsWith("reactome database id")) 
114 			{
115 				//remove the long comment (save some RAM)
116 				if(!(xref instanceof PublicationXref))
117 					xref.getComment().clear();
118 				
119 				//proceed with a unification xref only...
120 				if(xref instanceof UnificationXref) {
121 					for(XReferrable owner :  new HashSet<XReferrable>(xref.getXrefOf())) {
122 						for(Xref x : new HashSet<Xref>(owner.getXref())) {
123 							if(!(x instanceof UnificationXref) || x.equals(xref)) 
124 								continue;
125 							//another unif. xref present in the same owner object
126 							if(x.getDb() != null && x.getDb().toLowerCase().startsWith("reactome")
127 									&& x.getId()!= null && x.getId().startsWith("REACT_")) {
128 								//remove the unstable ID ref from the object
129 								owner.removeXref(xref); 
130 								xrefsToRemove.add(xref);
131 								//(it's ok to keep only this stable x)
132 							}
133 						}
134 					}
135 				}
136 			}
137 		}
138 		log.info(xrefsToRemove.size() + " unstable unif. xrefs, where a stable one also exists, " +
139 			"were removed from the corresponding xref properties.");
140 		
141 		ModelUtils.removeObjectsIfDangling(model, UtilityClass.class);
142 		
143 		// convert model back to OutputStream for return
144 		try {
145 			simpleReader.convertToOWL(model, cleanedData);
146 		} catch (Exception e) {
147 			throw new RuntimeException("clean(), Exception thrown while saving cleaned Reactome data", e);
148 		}
149 		
150 	}
151

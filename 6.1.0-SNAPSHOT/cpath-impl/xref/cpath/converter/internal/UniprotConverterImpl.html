44  	public void convert(InputStream is, OutputStream os) {
45  		// ref to reader here so
46  		// we can close in finally clause
47          InputStreamReader reader= null;
48          Model model = BioPAXLevel.L3.getDefaultFactory().createModel();
49          model.setXmlBase(xmlBase);
50          
51          try {
52              reader = new InputStreamReader(is, "UTF-8");
53              BufferedReader bufferedReader = new BufferedReader(reader);
54              String line = bufferedReader.readLine();
55              final HashMap<String, StringBuilder> dataElements = new HashMap<String, StringBuilder>();
56             	log.info("convert(), starting to read data...");
57              long linesReadSoFar = 0;
58              while (line != null) {
59              	linesReadSoFar++;
60                  if (line.startsWith ("//")) //reached the end of a Uniprot entry
61                  {
62  					// grab properties from the map and prepare for parsing
63                      String deField = dataElements.get("DE").toString();
64                      String organismName = dataElements.get("OS").toString(); //mostly occurs once per entry 
65                      String organismTaxId = dataElements.get("OX").toString(); //occurs once per entry 
66                      StringBuilder comments = dataElements.get("CC");
67                      StringBuilder geneName = dataElements.get("GN");
68                      String acNames = dataElements.get("AC").toString();
69                      StringBuilder xrefs = dataElements.get("DR");
70                      String idParts[] = dataElements.get("ID").toString().split("\\s+");
71                      StringBuilder sq = dataElements.get("SQ"); //SEQUENCE SUMMARY
72                      StringBuilder sequence = dataElements.get("  "); //SEQUENCE
73                      StringBuilder features = dataElements.get("FT"); //strict format in 6-75 char in each FT line
74                      
75                      ProteinReference proteinReference = newUniProtWithXrefs(idParts[0], acNames, model);
76                      
77              		// add some external xrefs from DR fileds
78                      if (xrefs != null) {
79                          setXRefsFromDRs(xrefs.toString(), proteinReference, model);
80                      }
81                            
82                      setNameAndSynonyms(proteinReference, deField);
83                      
84                      setOrganism(organismName, organismTaxId, proteinReference, model);
85                      
86                      // GN gene symbols - to PR names and rel. xrefs
87                      if (geneName != null) {
88                      	Collection<String> geneNames = getGeneSymbols(geneName.toString(), proteinReference);
89                          // always use "HGNC Symbol" for rel. xrefs, despite it can be from MGI, RGD,.. (these are coordinated by HGNC)
90                      	// (cannot do this in setXRefsFromDRs: no gene synonyms there, and organism specific db names like MGI)
91                          for (String symbol : geneNames) {
92                          	// also add Gene Names to PR names (can be >1 due to isoforms)
93                          	proteinReference.addName(symbol);
94                          	RelationshipXref rXRef = PreMerger
95                          		.findOrCreateRelationshipXref(RelTypeVocab.IDENTITY, "HGNC Symbol", symbol, model);
96                          	proteinReference.addXref(rXRef);
97                          }
98                      }
99                      
100                     //synonyms from GN
101                     if (geneName != null) {
102                     	Collection<String> geneSyns = getGeneSynonyms(geneName.toString(), proteinReference);
103                         for (String symbol : geneSyns) {
104                         	// also add gene synonyms to PR names but do not create xrefs 
105                         	// (these are not necessarily official HGNC Symbols and ambiguous)
106                         	proteinReference.addName(symbol);
107                         }
108                     }
109                                       
110                     // add some info from CC fields to BioPAX comments
111                     if (comments != null) {
112                         setComments (comments.toString(), proteinReference);
113                     }
114 
115 // won't store canonical sequences (in practice, it does not help and may even mislead: 
116 // in fact, one usually needs to know an isoform sequence (variant) and its version exactly)
117 //                    if(sequence != null) { //set sequence (remove spaces)
118 //                    	String seq = sequence.toString().replaceAll("\\s", "");
119 //                    	proteinReference.setSequence(seq);
120 //                    	proteinReference.addComment(sq.toString()); //sequence summary
121 //                    }
122                     
123                     //create modified residue features
124                     if(features != null)
125                     	createModResFeatures(features.toString(), proteinReference, model);
126                     
127                     // debug: write the one-protein-reference model
128                     log.debug("convert(). so far line# " + linesReadSoFar);
129                     
130                     dataElements.clear();
131                 }
132 				else { //continue read and collect current Uniprot entry lines
133 					/* The two-character line-type code that begins each line is
134 					 * always followed by three blanks, so that the actual 
135 					 * information begins with the sixth character.
136 					 */
137                     String key = line.substring (0, 2);
138                     String data = line.substring(5);
139                     if (data.startsWith("-------") ||
140                     	data.startsWith("Copyrighted") ||
141                         	data.startsWith("Distributed")) 
142                     {
143                         //  do nothing
144                     } else {
145                     	//important for correct splitting DR rows
146                     	if(key.equals("DR"))
147                     		data += "\n"; 
148                         if (dataElements.containsKey(key)) {
149                         	//remove leading spaces from second and next lines in FT, CC, DE records
150                         	if(data.startsWith(" ")) //i.e, the sixth char on the line is space/blank
151                         		data = data.replaceAll("^\\s+", "");                       	
152                         	dataElements.get(key).append(data);
153                         } else {
154                             dataElements.put(key, new StringBuilder (data));
155 						}
156                     }
157                 }
158                 line = bufferedReader.readLine();//it removes EOLs
159             }
160         }
161 		catch(IOException e) {
162 			throw new RuntimeException("Failed to convert UniProt data to BioPAX", e);
163 		}
164 		finally {
165 			log.debug("convert(), closing reader.");
166             if (reader != null) {
167 				try { reader.close(); } catch (Exception e) {} //ignore
168             }
169         }       
170 
171         log.info("convert(), repairing.");
172         model.repair();
173         
174         log.info("convert(), writing.");
175         new SimpleIOHandler(BioPAXLevel.L3).convertToOWL(model, os);
176     }

179 	/*
180 	 * Sets name and synonyms on protein reference.
181 	 *
182 	 * @param proteinReference ProteinReference
183 	 * @param deField String
184 	 */
185     private void setNameAndSynonyms (ProteinReference proteinReference, String deField) {
186         //  With the latest UNIPROT Export, the DE Line contains multiple fields.
187         //  For example:
188         //  DE   RecName: Full=14-3-3 protein beta/alpha;
189         //  DE   AltName: Full=Protein kinase C inhibitor protein 1;
190         //  DE            Short=KCIP-1;
191         //  DE   AltName: Full=Protein 1054;
192         //  We only want DE  RecName: Full
193         if (deField != null && deField.length() > 0) {
194             String fields[] = deField.split(";");
195             for (String field: fields) {
196                 String parts[] = field.split("=");
197                 if (parts.length == 2) {
198                     String fieldName = parts[0]; //no trim() required here
199                     
200                     String fieldValue = parts[1].trim();
201                     //after 1 Oct 2014, remove evidence (e.g., {type|source} - {ECO:0000269|PubMed10433554}) at the name's end (see http://www.uniprot.org/changes/evidences)
202                     int idx = fieldValue.indexOf(" {");
203                     if(idx>0) fieldValue = fieldValue.substring(0, idx);
204                     
205                     if ("RecName: Full".equals(fieldName)) {
206 						proteinReference.setStandardName(fieldValue);
207                     }
208 					else {
209 						proteinReference.addName(fieldValue);
210                     }
211                 }
212             }
213         }
214     }

217      * Sets the Current Organism Information.
218 	 *
219 	 * @param organismName String
220      * @param organismTaxId String
221      * @param proteinReference ProteinReference
222      * @param model
223      */
224     private void setOrganism(String organismName, String organismTaxId, 
225     		ProteinReference proteinReference, Model model) {
226         String parts[] = organismTaxId.replaceAll(";", "").split("=");
227         String taxId = parts[1];
228         //since 1 Oct 2014, have to remove {evidence} after the taxId (after space); see http://www.uniprot.org/changes/evidences
229         int idx = taxId.indexOf(" {");
230         if(idx > 0)
231         	taxId = taxId.substring(0, idx);
232         
233         parts = organismName.split("\\("); // - by first occurrence of '('
234         String name = parts[0].trim();
235 		BioSource bioSource = getBioSource(taxId, name, model);
236 		proteinReference.setOrganism(bioSource);
237     }

239     /**
240      * Sets some BioPAX comments 
241      * ("INTERACTION" sections of UniProt CC fields, 
242      * gene symbols, and copyright.)
243 	 *
244 	 * @param comments String
245 	 * @param geneSynonyms String
246 	 * @param proteinReference ProteinReference
247      */
248     private void setComments (String comments, ProteinReference proteinReference) 
249     {
250         String commentParts[] = comments.split("-!- ");
251         StringBuilder reducedComments = new StringBuilder();
252         for (int i=0; i<commentParts.length; i++) {
253             String currentComment = commentParts[i];
254             //  Filter out the Interaction comments.
255             //  We don't want these, as cPath itself will contain the interactions.
256             if (!currentComment.startsWith("INTERACTION")) {
257                 currentComment = currentComment.replaceAll("     ", " ");
258                 reducedComments.append (currentComment);
259             }
260         }
261         if (reducedComments.length() > 0) {
262             reducedComments.append (" COPYRIGHT:  Protein annotation is derived from the "
263                     + "UniProt Consortium (http://www.uniprot.org/).  Distributed under "
264                     + "the Creative Commons Attribution-NoDerivs License.");
265         }
266         
267 		proteinReference.addComment(reducedComments.toString());
268     }

271     /**
272      * Sets Multiple Types of XRefs, e.g. Entrez Gene ID and RefSeq.
273 	 *
274 	 * @param dbRefs String (concatenated 'DR' lines)
275      * @param proteinReference
276      * @param model
277      */
278     private void setXRefsFromDRs (String dbRefs, ProteinReference proteinReference, Model model) {
279     	final String lines[] = dbRefs.split("\n"); 
280     	
281         for (String line : lines) {
282         	//remove everything after '.' (e.g., isoform refs, comments)
283         	String xref = line.replaceFirst("\\..*", "").trim();
284         	String parts[] = xref.split(";");
285         	
286         	String db = parts[0].trim();
287         	
288         	// skip for other, not identity, ID types,
289         	// e.g., refs to pathway databases, ontologies, etc.:
290         	if (!db.equalsIgnoreCase("GENEID") 
291         			&& !db.equalsIgnoreCase("REFSEQ") 
292         			&& !db.equalsIgnoreCase("ENSEMBL") 
293         			&& !db.equalsIgnoreCase("HGNC")) {
294 				continue;
295         	}
296 
297 			String fixedDb = db;	
298 			if (db.equalsIgnoreCase("GENEID"))
299 				fixedDb = "NCBI Gene"; // - preferred name
300 			
301 			//iterate over the ID tokens of the same DR line, skipping non-ID comments, etc. (ending)
302 			for (int j = 1; j < parts.length; j++) {
303 				String id = parts[j].trim();
304 
305 				//at the end of a DR line in some cases (e.g, GeneID or RefSeq)?
306 				if(id.equals("-"))
307 					break;
308 				//no more Ensembl IDs (skip comments)
309 				if (db.equalsIgnoreCase("ENSEMBL") && !id.startsWith("ENS"))
310 					break; 
311 				//last ID in a HGNC line is in fact gene name
312 				if(db.equalsIgnoreCase("HGNC") && !id.startsWith("HGNC:")) {
313 					fixedDb = "HGNC Symbol";
314 				}
315 				//remove .version from RefSeq IDs
316 				if (db.equalsIgnoreCase("REFSEQ")) {
317 					// extract only RefSeq AC from AC.Version ID form
318 					fixedDb = "RefSeq";
319 					id = id.replaceFirst("\\.\\d+", "");
320 				}
321 				
322 				//ok to create a new rel. xref with type "identity"
323 				RelationshipXref rXRef = PreMerger
324 						.findOrCreateRelationshipXref(RelTypeVocab.IDENTITY, 
325 								fixedDb, id, model);					
326 				proteinReference.addXref(rXRef);
327 				// this xref type is then used for id-mapping in the Merger and queries;
328 			}
329         }
330         	
331     }

335      * Gets Official Gene Symbols from UniProt record's GN fields.
336      * 
337      * @param geneName concatenated UniProt record's GN fields.
338      * @param proteinReference
339      */
340     private Collection<String> getGeneSymbols(String geneName, ProteinReference proteinReference) 
341     {
342     	Collection<String> symbls = new ArrayList<String>();
343         String parts[] = geneName.split(";\\s*(and)?");
344         for (int i=0; i<parts.length; i++) {
345             String subParts[] = parts[i].split("=");
346             // can be >1 due to isoforms
347             if (subParts[0].trim().equals("Name")) {
348             	//remove {evidence}; see http://www.uniprot.org/changes/evidences (GN)
349             	String gn = subParts[1];
350             	int idx = gn.indexOf(" {");
351             	if(idx>0)
352             		gn = gn.substring(0, idx);         	
353             	symbls.add(gn);
354             }
355         }
356         return symbls;
357     }

359     /**
360      * Gets Gene Synonyms from UniProt record's GN fields.
361      * 
362      * @param geneName concatenated UniProt record's GN fields 
363      * @param proteinReference
364      */
365     private Collection<String> getGeneSynonyms(String geneName, ProteinReference proteinReference) 
366     {
367     	Collection<String> syns = new ArrayList<String>();
368         String parts[] = geneName.split(";\\s*(and)?");
369         for (int i=0; i<parts.length; i++) {
370             String subParts[] = parts[i].split("=");
371 			if (subParts[0].trim().equals("Synonyms")) {
372                 String synList[] = subParts[1].split(",");
373                 for (int j=0; j<synList.length; j++) {
374                     String currentSynonym = synList[j].trim();
375                 	//remove {evidence}; see http://www.uniprot.org/changes/evidences (GN)
376                 	int idx = currentSynonym.indexOf(" {");
377                 	if(idx>0)
378                 		currentSynonym = currentSynonym.substring(0, idx); 
379                     syns.add(currentSynonym);
380                 }
381             }
382         }
383         return syns;
384     }

410 	/**
411 	 * Generates a new protein reference object (ProteinReference, BioPAX L3)
412 	 * from a pre-processed UniProt record: assigns the standard URI and 
413 	 * unification xrefs.
414 	 *
415 	 * @param shortName
416 	 * @param accessions AC field value - from the UniProt text format
417 	 * @param model 
418 	 * @return ProteinReference
419 	 */
420 	private ProteinReference newUniProtWithXrefs(String shortName, String accessions, Model model) 
421 	{	
422 		// accession numbers as array
423 		final String acList[] = accessions.split(";");
424 		// the first one, primary id, becomes the RDFId
425 		final String primaryId = acList[0].trim();
426 		final String uri = "http://identifiers.org/uniprot/" + primaryId;
427 		// create a new PR
428 		ProteinReference proteinReference = model.addNew(ProteinReference.class, uri);
429 		proteinReference.setDisplayName(shortName);
430 		
431 		// add all unification xrefs
432 		for (String acEntry : acList) {
433 			String ac = acEntry.trim();
434 			//use db='UniProt' for these xrefs instead of 'UniProt Knowledgebase' (preferred in MIRIAM)
435 			setUnificationXRef("UniProt", ac, proteinReference, model);
436 		}
437 		
438 		return proteinReference;
439 	}

442 	 * Gets a biosource
443 	 *
444 	 * @param taxId String
445 	 * @param name String
446 	 * @param model
447 	 * @return BioSource
448 	 */
449 	private BioSource getBioSource(String taxId, String name, Model model) 
450 	{
451 		// check taxonomy ID is integer value
452 		Integer taxonomy = null;
453 		try {
454 			taxonomy = Integer.valueOf(taxId);
455 		} catch (NumberFormatException e) {
456 			throw new RuntimeException("Faild to convert " + taxId
457 					+ " into integer taxonomy ID", e);
458 		}
459 
460 		BioSource toReturn = null;
461 
462 		// check the organism was previously used, re-use it
463 		if(taxonomy==null || taxonomy <= 0) {
464 			throw new RuntimeException("Illegal taxonomy ID: " + taxId);
465 		} else {
466 			String uri = "http://identifiers.org/taxonomy/" + taxonomy;
467 			if (model.containsID(uri)) {
468 				toReturn = (BioSource) model.getByID(uri);
469 			} else {
470 				toReturn = (BioSource) model
471 						.addNew(BioSource.class, uri);
472 				toReturn.setStandardName(name);
473 				toReturn.setDisplayName(name);
474 				UnificationXref taxonXref = (UnificationXref) model
475 					.addNew(UnificationXref.class, Normalizer
476 						.uri(model.getXmlBase(), "TAXONOMY", taxId, UnificationXref.class));
477 				taxonXref.setDb("taxonomy");
478 				taxonXref.setId(taxId);
479 				toReturn.addXref((UnificationXref) taxonXref);
480 			}
481 		}
482 		return toReturn;
483 	}

486 	/*
487 	 * Parses only "FT   MOD_RES   N    M   Term..." lines data and creates protein modification features and sites;
488 	 * original data line cannot exceed 70 chars, but can span multiple lines (usually just one or two),
489 	 * and ends with '.'; extra lines were originally like "FT                      end-of-term."
490 	 * (dot is used only on the last line), but "FT   " and all the leading spaces up to original position 
491 	 * no. 35 in the second etc. lines were already removed from the final 'features' text.
492 	 * In other words, here, 'features' string contains concatenated with '.' lines like 
493 	 * "MOD_RES   N    M   full-term-name" (and such strings can be longer than 65 chars)
494 	 */
495 	private void createModResFeatures(final String features, 
496 			final ProteinReference pr, Model model) 
497 	{
498 		// using a special "not greedy" regex!
499         Pattern pattern = Pattern.compile("MOD_RES.+?\\.");
500         Matcher matcher = pattern.matcher(features);
501         int mfIndex = 0;
502         while(matcher.find()) {
503         	String ftContent = matcher.group(); //i.e., not including "FT   ",
504         	//the term starts at 29th char (because "FT   " at the beginning of each line already's gone)
505 			String what = ftContent.substring(29, ftContent.length()-1);//excluding the final dot '.' 
506 			// split the result by ';' (e.g., it might now look like "Phosphothreonine; by CaMK4") 
507 			// to extract the modification type and create the standard PSI-MOD synonym; 
508 			String[] terms = what.toString().split(";");
509 			String mod = terms[0];
510 			
511 			//remove non-standard ending comment from the standard CV term,
512 			//right before the final dot, if present; i.e. it removes things like 
513 			//"...(By similarity).", or "...(Probable).", "...(Potential)." -
514 			// but should not be too greedy to left just "N6-" out of "N6-(pyridoxal phosphate)lysine (By similarity)."
515 			mod = mod.replaceFirst("\\([^()]+?\\)$","").trim();
516 			
517 			//official PSI-MOD synonym (see http://www.ebi.ac.uk/ontology-lookup)
518 			final String modTerm = "MOD_RES " + mod; 
519 			
520 			//PSI-MOD ID will be auto-added by the biopax-validator/normalizer
521 			
522 			// Create the feature with CV and location -
523 			mfIndex++;
524 			String uri = Normalizer.uri(model.getXmlBase(), 
525 					null, pr.getDisplayName() + "_" + mfIndex, ModificationFeature.class);
526 			ModificationFeature modificationFeature = model.addNew(ModificationFeature.class, uri);
527 			modificationFeature.addComment(ftContent);
528 			
529 			// get/create a new PSI-MOD SequenceModificationVocabulary (can be shared by many PRs)
530 			uri = Normalizer.uri(model.getXmlBase(), "MOD", modTerm, SequenceModificationVocabulary.class);
531 			// so, let's check if it exists in the temp. or target model:
532 			SequenceModificationVocabulary cv = (SequenceModificationVocabulary) model.getByID(uri);
533 			if(cv == null) {
534 				// create a new SequenceModificationVocabulary
535 				cv = model.addNew(SequenceModificationVocabulary.class, uri);
536 				cv.addTerm(modTerm);
537 				//add the name without MOD_RES prefix as well (sometimes it is the valid one)
538 				cv.addTerm(mod);
539 			}
540 			modificationFeature.setModificationType(cv);
541 			
542 			// create feature location (site or interval)
543 			final int start = Integer.parseInt(ftContent.substring(9, 15).trim());
544 			final int end = Integer.parseInt(ftContent.substring(16, 22).trim());
545 			// so, let's check if the site exists in the temp. model
546 			String idPart = pr.getDisplayName() + //e.g., CALM_HUMAN - from the ID line 
547 								"_" + start;
548 			uri = Normalizer.uri(model.getXmlBase(), null, idPart, SequenceSite.class);			
549 						
550 			SequenceSite startSite = (SequenceSite) model.getByID(uri);
551 			if(startSite == null) {
552 				startSite = model.addNew(SequenceSite.class, uri);
553 				startSite.setPositionStatus(PositionStatusType.EQUAL);
554 				startSite.setSequencePosition(start);
555 			}
556 			
557 			if(start == end) {
558 				modificationFeature.setFeatureLocation(startSite);
559 			} else {
560 				//create the second site (end) and sequence interval -
561 				idPart = pr.getDisplayName() + //e.g., CALM_HUMAN - from the ID line
562 							"_" + end;
563 				uri = Normalizer.uri(model.getXmlBase(), null, idPart, SequenceSite.class);
564 					
565 				SequenceSite endSite = (SequenceSite) model.getByID(uri);
566 				if(endSite == null) {
567 					endSite = model.addNew(SequenceSite.class, uri);
568 					endSite.setPositionStatus(PositionStatusType.EQUAL);
569 					endSite.setSequencePosition(end);
570 				}
571 				
572 				idPart = pr.getDisplayName() + 	"_" + start + "_" + end;
573 				uri = Normalizer.uri(model.getXmlBase(), null, idPart, SequenceInterval.class);
574 						
575 				SequenceInterval sequenceInterval = (SequenceInterval) model.getByID(uri);
576 				if(sequenceInterval == null) {

28   * Test ChEBI to BioPAX converter.
29   *
30   */
31  public class ChebiConvertersTest {	
32  	
33  	@Test
34  	public void testConvertObo() throws IOException {
35  		// convert test data
36  		ByteArrayOutputStream bos = new ByteArrayOutputStream();
37  		CPathUtils.unzip(new ZipInputStream(new FileInputStream(
38  				getClass().getResource("/chebi.obo.zip").getFile())), bos);
39  		byte[] data = bos.toByteArray();
40  		//run the new OBO converter
41  		bos.reset(); //re-use
42  		Converter converter = ImportFactory.newConverter("cpath.converter.internal.ChebiOboConverterImpl");
43  		converter.setXmlBase(CPathSettings.getInstance().getXmlBase());		
44  		converter.convert(new ByteArrayInputStream(data), bos);
45  		
46  		Model model = new SimpleIOHandler().convertFromOWL(new ByteArrayInputStream(bos.toByteArray()));
47  		assertNotNull(model);
48  		assertFalse(model.getObjects().isEmpty());
49  		
50  		bos.close(); data=null;
51  		
52  		// dump owl for review
53  		String outFilename = getClass().getClassLoader().getResource("").getPath() 
54  			+ File.separator + "testConvertChebiObo.out.owl";		
55  		(new SimpleIOHandler(BioPAXLevel.L3)).convertToOWL(model, new FileOutputStream(outFilename));
56  		
57  		// get all small molecule references out
58  		assertEquals(6, model.getObjects(SmallMoleculeReference.class).size());
59  
60  		// get lactic acid sm
61  		String rdfID = "http://identifiers.org/chebi/CHEBI:422";
62  		assertTrue(model.containsID(rdfID));
63  		SmallMoleculeReference smallMoleculeReference = (SmallMoleculeReference)model.getByID(rdfID);
64  
65  		// check some props
66  		assertEquals("(S)-lactic acid", smallMoleculeReference.getDisplayName());
67  		assertEquals(11, smallMoleculeReference.getName().size());
68  		assertEquals("C3H6O3", smallMoleculeReference.getChemicalFormula());
69  		System.out.println("CHEBI:422 xrefs (from OBO): " + smallMoleculeReference.getXref());
70  		int relationshipXrefCount = 0;
71  		int unificationXrefCount = 0;
72  		int publicationXrefCount = 0;
73  		for (Xref xref : smallMoleculeReference.getXref()) {
74  			if (xref instanceof RelationshipXref) ++relationshipXrefCount;
75  			if (xref instanceof UnificationXref) ++ unificationXrefCount;
76  			if (xref instanceof PublicationXref) ++ publicationXrefCount;
77  		}
78  		assertEquals(1, unificationXrefCount); //no secondary ChEBI IDs there (non-chebi are, if any, relationship xrefs)
79  		assertEquals(7, relationshipXrefCount); //only chebi, inchikey, cas, kegg, wikipedia are taken there (could be drugbank and chembl if present too)
80  		assertEquals(9, publicationXrefCount); //there are nine such xrefs
81  		
82  		// following checks work in this test only (using in-memory model); with DAO - use getObject...
83          assertTrue(model.containsID("http://identifiers.org/chebi/CHEBI:20"));
84          EntityReference er20 = (EntityReference) model.getByID("http://identifiers.org/chebi/CHEBI:20");
85          assertTrue(model.containsID("http://identifiers.org/chebi/CHEBI:28"));
86          EntityReference er28 = (EntityReference) model.getByID("http://identifiers.org/chebi/CHEBI:28");
87          assertTrue(model.containsID("http://identifiers.org/chebi/CHEBI:422"));
88          EntityReference er422 = (EntityReference) model.getByID("http://identifiers.org/chebi/CHEBI:422");
89          
90          // 28 has member - 422 has member - 20
91          assertTrue(er20.getMemberEntityReferenceOf().contains(er422));
92          assertEquals(er20, er422.getMemberEntityReference().iterator().next());
93          
94  		assertTrue(er422.getMemberEntityReferenceOf().contains(er28));
95          assertEquals(er422, er28.getMemberEntityReference().iterator().next());
96  
97          // check new elements (created by the OBO converter) exist in the model;
98          // (particularly, these assertions are important to test within the persistent model (DAO) session)
99          assertTrue(model.containsID(Normalizer.uri(model.getXmlBase(), "CHEBI", "CHEBI_20_see-also", RelationshipXref.class)));
100         assertTrue(model.containsID(Normalizer.uri(model.getXmlBase(), "CHEBI", "CHEBI_422_see-also", RelationshipXref.class)));		
101 		
102 	}
103 }

028package cpath.service;
029
030/**
031 * cPath2 web service command arguments.
032 * 
033 * @author rodche
034 *
035 */
036public enum CmdArgs {
037        uri("an identifier, usually a BioPAX element URI (default); multiple '&id=' arguments per query are supported"),
038    q("query string (full-text search)."),
039    page("search results page number (>=0)."),
040        type("a BioPAX class - see the <a href=\"#valid_biopax_parameter\">valid values for type parameter</a> below"),
041        kind("graph query type - see the <a href=\"#valid_graph_parameter\">valid values for kind parameter</a> below"),
042        format("output format - see the <a href=\"#valid_output_parameter\">valid values for format parameter</a> below"),
043        organism("filter by organism (multiple 'organism=' are allowed) - see the <a href=\"#valid_organism_parameter\">valid values for organism parameter</a> below"),
044        datasource("filter by data sources (multiple 'datasource=' are allowed) - see the <a href=\"#valid_datasource_parameter\">valid values for datasource parameter</a> below"),
045        source("graph query source URI (multiple 'source=' are allowed)"),
046        target("graph query destination URI (multiple 'target=' are allowed)"),
047    limit("graph query search distance limit"),
048    biopax("a BioPAX OWL to convert"),
049    path("a BioPAX property path expression (like xPath)"),
050    direction("graph query parameter 'direction'"),
051    //alg("a user-defined BioPAX data analysis (byte code) to run."), //TODO
052        ;
053        
054        private final String info;
055        
056        public String getInfo() {
057                return info;
058        }
059
060        private CmdArgs(String info) {
061                this.info = info;
062        }
063        
064        @Override
065        public String toString() {
066                return name().toLowerCase();
067        }       
068}

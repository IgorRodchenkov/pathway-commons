028package cpath.service;
029
030import static cpath.service.CmdArgs.*;
031
032
033/**
034 * cPath2 web service commands.
035 * 
036 * @author rodche
037 *
038 */
039public enum Cmd {
040        SEARCH("Full-text search for BioPAX objects. " +
041                        "It accepts up to five parameters " +
042                        "and returns the ordered list of search hits, " +
043                        "which are simplified description of BioPAX elements " +
044                        "matching the query and passing all filters. A hit's uri (same as the corresponding BioPAX " +
045                        "object's RDF ID) can be used with other webservice commands to " +
046                        "extract the corresponding sub-model to BioPAX or another supported format. ",
047                        "/search?q=brca*&organism=9606", //full PROVIDER_URL shouldn't be specified here (it depends on the server configuration)!
048                        "Search Response that lists Search Hits - XML (default) or JSON (when called as '/search.json?')",
049                        new CmdArgs[]{q, page, type, organism, datasource}),    
050        GET("Gets a BioPAX element or sub-model " +
051        "by ID(s).  This command has two parameters.",
052        "/get?uri=http://identifiers.org/uniprot/P38398",
053        "BioPAX by default, other formats as specified by the format parameter.  " +
054        "See the <a href=\"#valid_output_parameter\">valid values for format parameter</a> below.",
055        new CmdArgs[]{uri, format}),
056        GRAPH("Executes an advanced graph query on the data within pathway commons. " +
057          "Returns a sub-model as the result. This command has up to six parameters.",
058          "/graph?kind=neighborhood&source=URI1&source=URI2&...",
059          "BioPAX by default, other formats as specified by the format parameter. " +
060          "See the <a href=\"#valid_output_parameter\">valid values for format parameter</a> below.",
061          new CmdArgs[]{kind, source, target, format, limit, direction, organism, datasource}),
062    TOP_PATHWAYS("Gets Top Pathways. This command accepts optional filter by organism and by datasource values", 
063        "/top_pathways",
064        "Search Response - XML (JSON, when called as '/top_pathways.json?') contains the list of all top pathways.", 
065        new CmdArgs[]{organism, datasource}),   
066    TRAVERSE("Gets data property values (or elements's URIs) " +
067        "at the end of the property path.  This command has two parameters.",
068        "/traverse?uri=http://identifiers.org/uniprot/P38398&path=ProteinReference/organism/displayName",
069        "Traverse Response - XML (or JSON, when called as '/traverse.json?').", 
070        new CmdArgs[]{path, uri})
071          
072    ;
073        
074        private final CmdArgs[] args; //Array is better for use in json/jsp than List/Set
075        private final String info;
076    private final String example;
077    private final String output;
078        
079        public CmdArgs[] getArgs() {
080                return args;
081        }
082        
083        public String getInfo() {
084                return info;
085        }
086
087    public String getExample() {
088        return example;
089    }
090    
091    public String getOutput() {
092        return output;
093    }
094        
095        private Cmd(String info, String example, String output, CmdArgs... args) {
096                this.info = info;
097        this.example = example;
098        this.output = output;
099                this.args = args;
100        }
101
102        @Override
103        public String toString() {
104                return name().toLowerCase();
105        }
106}

089        public static final String FIELD_KEYWORD = "keyword"; //anything, e.g., names, terms, comments, incl. - from child elements

091        public static final String FIELD_XREFDB = "xrefdb"; //xref.db
092        public static final String FIELD_XREFID = "xrefid"; //xref.id
093        public static final String FIELD_PATHWAY = "pathway"; //pathways and parent pathways; to be inferred from the whole biopax model
094        public static final String FIELD_SIZE = "size";

103        //one can still search in other fields directly, e.g.,
104        //pathway:some_keywords datasource:"pid", etc.

110                        FIELD_KEYWORD, //includes data type properties (names, terms, comments), 
111                                                  //also from  child elements up to given depth (3), also stores but not indexes parent pathway uris and names;         
112//                      FIELD_PATHWAY, // this (if pathway) and PARENT pathway URIs are stored in the index, not analyzed/indexed; whereas names - indexed but not stored;                      
113// the following fields are for filtering only (thus excluded):
114//                      FIELD_ORGANISM, 
115//                      FIELD_DATASOURCE, 
116//                      FIELD_TYPE,

161        public SearchResponse search(String query, int page,
162                        Class<? extends BioPAXElement> filterByType, String[] datasources,
163                        String[] organisms) 
164        {
165                SearchResponse response = null;
166                
167                LOG.debug("search: " + query + ", page: " + page 
168                        + ", filterBy: " + filterByType
169                        + "; extra filters: ds in (" + Arrays.toString(datasources)
170                        + "), org. in (" + Arrays.toString(organisms) + ")");
171                
172                IndexSearcher searcher = null;
173        
174                try {   
175                        QueryParser queryParser = new MultiFieldQueryParser(DEFAULT_FIELDS, analyzer);
176                        queryParser.setAllowLeadingWildcard(true);//TODO do we really want leading wildcards (e.g. *sulin)?
177                        
178                        searcher = searcherManager.acquire();   
179                        
180                        //find and transform top docs to search hits (beans), considering pagination...
181                        if(!query.trim().equals("*")) { //if not "*" query, which is not supported out-of-the-box, then
182                                //create the lucene query
183                                Query luceneQuery = queryParser.parse(query);
184//do NOT (Lucene 4.1), or scoring/highlighting won't work for wildcard queries...                               
185//luceneQuery = searcher.rewrite(luceneQuery); 
186                                LOG.debug("parsed lucene query is " + luceneQuery.getClass().getSimpleName());
187                                
188                                //create filter: type AND (d OR d...) AND (o OR o...)
189                                Filter filter = createFilter(filterByType, datasources, organisms);
190                                
191                                //get the first page of top hits
192                                TopDocs topDocs = searcher.search(luceneQuery, filter, maxHitsPerPage);
193                                //get the required hits page if page>0
194                                if(page>0) {
195                                        TopScoreDocCollector collector = TopScoreDocCollector.create(maxHitsPerPage*(page+1), true);  
196                                        searcher.search(luceneQuery, filter, collector);
197                                        topDocs = collector.topDocs(page * maxHitsPerPage, maxHitsPerPage);
198                                }
199                                
200                                //transform docs to hits, use a highlighter to get excerpts
201                                response = transform(luceneQuery, searcher, true, topDocs);
202        
203                        } else { //find ALL objects of a particular BioPAX class (+ filters by organism, datasource)
204                                if(filterByType==null) 
205                                        filterByType = Level3Element.class;
206
207                                //replace q="*" with a search for the class or its sub-class name in the TYPE field
208                                BooleanQuery luceneQuery = new BooleanQuery();
209                                for(Class<? extends BioPAXElement> subType : SimpleEditorMap.L3.getKnownSubClassesOf(filterByType)) {
210                                        luceneQuery.add(new TermQuery(new Term(FIELD_TYPE, subType.getSimpleName().toLowerCase())), Occur.SHOULD);
211                                }
212                                Filter filter = createFilter(null, datasources, organisms);
213                                
214                                //get the first page of top hits
215                                TopDocs topDocs = searcher.search(luceneQuery, filter, maxHitsPerPage);
216                                //get the required hits page if page>0
217                                if(page>0) {
218                                        TopScoreDocCollector collector = TopScoreDocCollector.create(maxHitsPerPage*(page+1), true);  
219                                        searcher.search(luceneQuery, filter, collector);
220                                        topDocs = collector.topDocs(page * maxHitsPerPage, maxHitsPerPage);     
221                                }
222                                
223                                //convert
224                                response = transform(luceneQuery, searcher, false, topDocs);                    
225                        }       
226                        
227                } catch (ParseException e) {
228                        throw new RuntimeException("getTopDocs: failed to parse the query string.", e);
229                } catch (IOException e) {
230                        throw new RuntimeException("getTopDocs: failed.", e);
231                } finally {
232                        try {
233                                if(searcher!=null) {
234                                        searcherManager.release(searcher);
235                                        searcher = null;
236                                }
237                        } catch (IOException e) {}      
238                }
239        
240                response.setPageNo(page);
241                
242                return response;
243        }

246        // Transform Lucene docs to hits (xml/java beans)
247        private SearchResponse transform(Query query, IndexSearcher searcher, boolean highlight, TopDocs topDocs) 
248                        throws CorruptIndexException, IOException 
249        {       
250                if(topDocs == null)
251                        throw new IllegalArgumentException("topDocs is null");
252                
253                SearchResponse response = new SearchResponse();
254                response.setMaxHitsPerPage(maxHitsPerPage);
255                response.setNumHits(topDocs.totalHits); 
256                List<SearchHit> hits = response.getSearchHit();//empty list
257                assert hits!=null && hits.isEmpty();
258                LOG.debug("transform, no. TopDocs to process:" + topDocs.scoreDocs.length);
259                for(ScoreDoc scoreDoc : topDocs.scoreDocs) {                    
260                        SearchHit hit = new SearchHit();
261                        Document doc = searcher.doc(scoreDoc.doc);
262                        String uri = doc.get(FIELD_URI);
263                        BioPAXElement bpe = model.getByID(uri);                 
264                        LOG.debug("transform: doc:" + scoreDoc.doc + ", uri:" + uri);
265                        
266                        // use the highlighter (get matching fragments)
267                        // for this to work, all keywords were stored in the index field
268                        if (highlight && doc.get(FIELD_KEYWORD) != null) {                              
269                                // use a Highlighter (store.YES must be enabled for 'keyword' field)
270                                QueryScorer scorer = new QueryScorer(query, FIELD_KEYWORD);
271                                
272                                //the following fixes scoring/highlighting for all-field wildcard (like q=insulin*)
273                                //but not for term/prefix queries (q=name:insulin*, q=pathway:brca2)_.
274                                scorer.setExpandMultiTermQuery(true); //TODO    
275                                
276                                //TODO use PostingsHighlighter once it's stable (see http://lucene.apache.org/core/4_10_0/highlighter/org/apache/lucene/search/postingshighlight/PostingsHighlighter.html)                              
277                                SimpleHTMLFormatter formatter = new SimpleHTMLFormatter("<span class='hitHL'>", "</span>");
278                                Highlighter highlighter = new Highlighter(formatter, scorer);
279                                highlighter.setTextFragmenter(new SimpleSpanFragmenter(scorer, 80));
280                                                                                
281                                final String text = StringUtils.join(doc.getValues(FIELD_KEYWORD), " ");
282                                try {
283                                        TokenStream tokenStream = analyzer.tokenStream("", new StringReader(text));
284                                        String res = highlighter.getBestFragments(tokenStream, text, 7, "...");
285
286                                        if(res != null && !res.isEmpty())
287                                                hit.setExcerpt(res);
288
289                                } catch (Exception e) {throw new RuntimeException(e);}
290
291                        } else if(highlight) {
292                                LOG.warn("Highlighter skipped, because KEYWORD field was null; hit: " 
293                                                + uri + ", " + bpe.getModelInterface().getSimpleName());
294                        }
295                        
296                        hit.setUri(uri);
297                        hit.setBiopaxClass(bpe.getModelInterface().getSimpleName());
298                        
299                        // add standard and display names if any -
300                        if (bpe instanceof Named) {
301                                Named named = (Named) bpe;
302                                String std = named.getStandardName();
303                                if (std != null)
304                                        hit.setName(std);
305                                else
306                                        hit.setName(named.getDisplayName());
307                                
308                                // a hack for BioSource (store more info)
309                                if(bpe instanceof BioSource) {
310                                        for(String name : named.getName())
311                                                hit.getOrganism().add(name);
312                                        String txid = getTaxonId((BioSource)named);
313                                        if(txid != null)
314                                                hit.getOrganism().add(txid);
315                                }
316                                
317                                // a hack for Provenance: save/return other names 
318                                // (to be used as filter by data source values)
319                                if(bpe instanceof Provenance) {
320                                        for(String name : named.getName())
321                                                hit.getDataSource().add(name);
322                                }       
323                        }
324                                                
325                        // extract organisms (URI only) 
326                        if(doc.get(FIELD_ORGANISM) != null) {
327                                Set<String> uniqueVals = new TreeSet<String>();
328                                for(String o : doc.getValues(FIELD_ORGANISM)) {
329                                        //note: only URIS are stored in the index                                       
330                                        uniqueVals.add(o);
331                                }
332                                hit.getOrganism().addAll(uniqueVals);
333                        }
334                        
335                        // extract values form the index
336                        if(doc.get(FIELD_DATASOURCE) != null) {
337                                Set<String> uniqueVals = new TreeSet<String>();
338                                for(String d : doc.getValues(FIELD_DATASOURCE)) {
339                                        //note: only URIS are stored in the index
340                                        uniqueVals.add(d);
341                                }
342                                hit.getDataSource().addAll(uniqueVals);
343                        }       
344                        
345                        // extract only pathway URIs 
346                        //(because names and IDs used to be stored in the index field as well)
347                        if(doc.get(FIELD_PATHWAY) != null) {
348                                Set<String> uniqueVals = new TreeSet<String>();
349                                for(String d : doc.getValues(FIELD_PATHWAY)) {
350                                        //only URIs were stored (though all names/ids were indexed/analyzed)
351                                        if(!d.equals(uri)) //exclude itself
352                                                uniqueVals.add(d);
353                                }
354                                hit.getPathway().addAll(uniqueVals);
355                        }
356                        
357                        //no. processes in the sub-network
358                        if(doc.get(FIELD_SIZE)!=null)
359                                hit.setSize(Integer.valueOf(doc.get(FIELD_SIZE))); 
360                        
361                        //if cpath2 debugging, update hit.excerpt - add the Lucene's 'explain' using the query and doc.
362                        if(CPathSettings.getInstance().isDebugEnabled()) {
363                                String excerpt = hit.getExcerpt();
364                                if(excerpt == null) excerpt = "";
365                                hit.setExcerpt(excerpt + " -SCORE- " + scoreDoc.score 
366                                                + " -EXPLANATION- " + searcher.explain(query, scoreDoc.doc));
367                        }
368                        
369                        hits.add(hit);
370                }
371                                
372                //add the Provenance's standardName(s) to the search response
373                if(!hits.isEmpty()) {
374                        for(String puri : response.provenanceUris()) {
375                                Provenance p = (Provenance) model.getByID(puri);
376                                response.getProviders()
377                                        .add((p.getStandardName()!=null)?p.getStandardName():p.getDisplayName());
378                        }
379                }
380                
381                return response;
382        }

385        public void index() {
386                final int numObjects =  model.getObjects().size();
387                LOG.info("index(), there are " + numObjects + " BioPAX objects to be (re-)indexed.");           
388                IndexWriter iw;         
389                try {
390                        //close the searcher manager if the old index exists
391                        if(searcherManager != null) {
392                                searcherManager.close();
393                                searcherManager = null;
394                        }
395                        IndexWriterConfig conf = new IndexWriterConfig(Version.LATEST, analyzer);
396                        iw = new IndexWriter(FSDirectory.open(indexFile), conf);
397                        //cleanup
398                        iw.deleteAll();
399                        iw.commit();
400                } catch (IOException e) {
401                        throw new RuntimeException("Failed to create a new IndexWriter.", e);
402                }               
403                final IndexWriter indexWriter = iw;
404
405                ExecutorService exec = Executors.newFixedThreadPool(30);
406                
407                final AtomicInteger numLeft = new AtomicInteger(numObjects);
408                for(final BioPAXElement bpe : model.getObjects()) {     
409                        // prepare & index each element in a separate thread
410                        exec.execute(new Runnable() {
411                                public void run() {                                     
412                                        // get or infer some important values if possible from this, child or parent objects:
413                                        Set<String> keywords = ModelUtils.getKeywords(bpe, 3);
414                                        
415                                        // a hack to remove special (debugging) biopax comments
416                                        for(String s : new HashSet<String>(keywords)) {
417                                                //exclude additional comments generated by normalizer, merger, etc.
418                                                if(s.startsWith("REPLACED ") || s.contains("ADDED"))
419                                                        keywords.remove(s);
420                                        }
421                                        
422                                        bpe.getAnnotations().put(FIELD_KEYWORD, keywords);
423                                        bpe.getAnnotations().put(FIELD_DATASOURCE, ModelUtils.getDatasources(bpe));
424                                        bpe.getAnnotations().put(FIELD_ORGANISM, ModelUtils.getOrganisms(bpe));
425                                        bpe.getAnnotations().put(FIELD_PATHWAY, ModelUtils.getParentPathways(bpe));
426                                        
427                                        // for bio processes, also save the total no. member interactions and pathways:
428                                        if(bpe instanceof org.biopax.paxtools.model.level3.Process) {
429                                                int size = new Fetcher(SimpleEditorMap.L3, Fetcher.nextStepFilter)
430                                                                .fetch(bpe, Process.class).size(); //except itself                                              
431                                                bpe.getAnnotations().put(FIELD_SIZE, Integer.toString(size)); 
432                                        }
433
434                                        index(bpe, indexWriter);
435                                        
436                                        //count, log a progress message
437                                        int left = numLeft.decrementAndGet();
438                                        if(left % 10000 == 0)
439                                                LOG.info("index(), biopax objects left to index: " + left);
440                                }
441                        });
442                }
443                
444                exec.shutdown(); //stop accepting new tasks     
445                try { //wait
446                        exec.awaitTermination(Long.MAX_VALUE, TimeUnit.SECONDS);
447                } catch (InterruptedException e) {
448                        throw new RuntimeException("Interrupted!", e);
449                }
450                
451                try {
452                        indexWriter.close(); //wait for pending op., auto-commit, close.
453                } catch (IOException e) {
454                        throw new RuntimeException("Failed to close IndexWriter.", e);
455                } 
456                
457                //finally, create a new searcher manager
458                initSearcherManager();
459        }

464        /**
465         * Creates a new Lucene Document that corresponds to a BioPAX object.
466         * It does not check whether the document exists (should not be there,
467         * because the {@link #index(Model)} method cleans up the index)
468         * 
469         * Some fields also include biopax data type property values not only from 
470         * the biopax object but also from its child elements, up to some depth 
471         * (using key-value pairs in the pre-computed bpe.annotations map):
472         * 
473         *  'uri' - biopax object's absolute URI, index=no, analyze=no, store=yes;
474         * 
475         *  'name' - names, analyze=yes, store=yes; boosted;
476         * 
477         *  'keyword' - infer from this bpe and its child objects' data properties,
478         *            such as Score.value, structureData, structureFormat, chemicalFormula, 
479         *            availability, term, comment, patoData, author, source, title, url, published, 
480         *            up to given depth/level; and also all 'pathway' field values are included here; 
481         *            analyze=yes, store=yes;
482         *  
483         *  'datasource', 'organism' and 'pathway' - infer from this bpe and its child objects 
484         *                                                                              up to given depth/level, analyze=no, store=yes;
485         *  
486         *  'size' - number of child processes, an integer as string; analyze=no, store=yes
487         * 
488         * @param bpe
489         * @param indexWriter
490        */
491        void index(BioPAXElement bpe, IndexWriter indexWriter) {                
492                // create a new document
493                final Document doc = new Document();
494                
495                // save URI (not indexed field)
496                Field field = new StoredField(FIELD_URI, bpe.getRDFId());
497                doc.add(field);
498                
499                // index and store but not analyze/tokenize the biopax class name:
500                field = new StringField(FIELD_TYPE, bpe.getModelInterface().getSimpleName().toLowerCase(), Field.Store.YES);
501                doc.add(field);
502                
503                // make index fields from the annotations map (of pre-calculated/inferred values)
504                if(!bpe.getAnnotations().isEmpty()) {
505                        if(bpe.getAnnotations().containsKey(FIELD_PATHWAY)) {
506                                addPathways((Set<Pathway>)bpe.getAnnotations().get(FIELD_PATHWAY), doc);
507                        }
508                        if(bpe.getAnnotations().containsKey(FIELD_ORGANISM)) {
509                                addOrganisms((Set<BioSource>)bpe.getAnnotations().get(FIELD_ORGANISM), doc);
510                        }
511                        if(bpe.getAnnotations().containsKey(FIELD_DATASOURCE)) {
512                                addDatasources((Set<Provenance>)bpe.getAnnotations().get(FIELD_DATASOURCE), doc);
513                        }
514                        if(bpe.getAnnotations().containsKey(FIELD_KEYWORD)) {
515                                addKeywords((Set<String>)bpe.getAnnotations().get(FIELD_KEYWORD), doc);
516                        }
517                        if(bpe.getAnnotations().containsKey(FIELD_SIZE)) {
518                                field = new IntField(FIELD_SIZE, 
519                                        Integer.parseInt((String)bpe.getAnnotations()
520                                        .get(FIELD_SIZE)), Field.Store.YES);
521                                doc.add(field);
522                        }
523                }
524                bpe.getAnnotations().remove(FIELD_KEYWORD);
525                bpe.getAnnotations().remove(FIELD_DATASOURCE);
526                bpe.getAnnotations().remove(FIELD_ORGANISM);
527                bpe.getAnnotations().remove(FIELD_PATHWAY);
528                bpe.getAnnotations().remove(FIELD_SIZE);
529                        
530                // name
531                if(bpe instanceof Named) {
532                        Named named = (Named) bpe;
533                        if(named.getStandardName() != null) {
534                                field = new TextField(FIELD_NAME, named.getStandardName(), Field.Store.NO);
535                                field.setBoost(3.0f);
536                                doc.add(field);
537                        }
538                        if(named.getDisplayName() != null && !named.getDisplayName().equalsIgnoreCase(named.getStandardName())) {
539                                field = new TextField(FIELD_NAME, named.getDisplayName(), Field.Store.NO);
540                                field.setBoost(2.5f);
541                                doc.add(field);
542                        }
543                        for(String name : named.getName()) {
544                                if(name.equalsIgnoreCase(named.getDisplayName()) || name.equalsIgnoreCase(named.getStandardName()))
545                                        continue;
546                                field = new TextField(FIELD_NAME, name.toLowerCase(), Field.Store.NO);
547                                field.setBoost(2.0f);
548                                doc.add(field);
549                        }
550                }
551                
552                // XReferrable.xref - build 'xrefid' index field from all Xrefs)
553                if(bpe instanceof XReferrable) {
554                        XReferrable xr = (XReferrable) bpe;
555                        for(Xref xref : xr.getXref()) {
556                                if (xref.getId() != null) {
557                                        //the filed is not_analyzed; so in order to make search case-insensitive 
558                                        //(when searcher uses standard analyzer), we turn the value to lowercase.
559                                        field = new TextField(FIELD_XREFID, xref.getId().toLowerCase(), Field.Store.NO);
560                                        doc.add(field);
561                                }
562                        }
563                }
564                
565                // Xref db/id (these are for a precise search by standard bio ID)
566                if(bpe instanceof Xref) {
567                        Xref xref = (Xref) bpe;
568                        if (xref.getId() != null) {
569                                field = new TextField(FIELD_XREFID, xref.getId().toLowerCase(), Field.Store.NO);
570                                doc.add(field);
571                        }
572                        if (xref.getDb() != null) {
573                                field = new TextField(FIELD_XREFDB, xref.getDb().toLowerCase(), Field.Store.NO);
574                                doc.add(field);
575                        }
576                }
577                
578                // write
579                try {
580                        indexWriter.addDocument(doc);
581                } catch (IOException e) {
582                        throw new RuntimeException("Failed to index; " + bpe.getRDFId(), e);
583                }
584        }

624                        // include cell type terms
625                        if (bs.getCellType() != null) {
626                                for (String s : bs.getCellType().getTerm()) {
627                                        doc.add(new TextField(FIELD_ORGANISM, s.toLowerCase(), Field.Store.NO));
628                                }
629                        }
630                }

650                                        doc.add(new StoredField(FIELD_KEYWORD, x.getId().toLowerCase()));//don't index but store for highlighting
651                                }
652                        }
653                }

673        /** 
674         * Creates a search filter like 
675         * type AND (datasource OR datasource...) 
676         *      AND (organism OR organism OR...)
677         * 
678         * Both names (partial or full) and URIs should work as filter values. 
679         * 
680         * @param type
681         * @param datasources
682         * @param organisms
683         */     
684        private Filter createFilter(Class<? extends BioPAXElement> type, 
685                        String[] datasources, String[] organisms) {
686                
687                BooleanQuery filterQuery = new BooleanQuery();
688                
689                //AND datasources       
690                if (datasources != null && datasources.length > 0) {
691                        filterQuery.add(subQuery(datasources, FIELD_DATASOURCE), Occur.MUST);
692                }
693                //AND organisms
694                if (organisms != null && organisms.length > 0) {
695                        filterQuery.add(subQuery(organisms, FIELD_ORGANISM), Occur.MUST);
696                }               
697                //AND type      
698                if(type != null) { //add biopax class filter
699                        BooleanQuery query = new BooleanQuery();
700                        query.add(new TermQuery(new Term(FIELD_TYPE, type.getSimpleName().toLowerCase())), Occur.SHOULD);//OR
701                        //for each biopax subclass (interface), add the name to the filter query
702                        for(Class<? extends BioPAXElement> subType : SimpleEditorMap.L3.getKnownSubClassesOf(type)) {
703                                query.add(new TermQuery(new Term(FIELD_TYPE, subType.getSimpleName().toLowerCase())), Occur.SHOULD);//OR
704                        }               
705                        filterQuery.add(query, Occur.MUST);
706                }
707                
708                if(!filterQuery.clauses().isEmpty()) {
709                        LOG.debug("filterQuery: " + filterQuery.toString());
710                        return new CachingWrapperFilter( new QueryWrapperFilter(filterQuery) ); //TODO why CachingWrapperFilter, QueryWrapperFilter?
711                } else 
712                        return null;
713        }

715        /*
716         * Values are joint with OR, but if a value
717         * has whitespace symbols, it also make a sub-query,
718         * in which terms are joint with AND. This is to filter
719         * by datasource/organism's full name, partial name, uri,
720         * using multiple datasources/organisms.
721         * For example, 
722         * "search?q=*&datasource=intact complex&type..." - will get only IntAct Complex objects;
723         * "search?q=*&datasource=intact&type..." - will consider both IntAct and IntAct Complex
724         * "search?q=*&datasource=intact biogrid&type..." means "to occur in both intact and biogrid" 
725         *  (can be a canonical protein reference; it's is not equivalent to "search?q=*&datasource=intact&datasource=biogrid&...",
726         *  which means "to occur either in intact, incl. IntAct Complex, or in biogrid or in all these")
727         * "search?q=*&datasource=intact complex biogrid&..." - won't match anything.
728         * 
729         * @param filterValues
730         * @param filterField
731         * @return
732         */
733        private Query subQuery(String[] filterValues, String filterField) {
734                BooleanQuery query = new BooleanQuery();        
735                final Pattern pattern = Pattern.compile("\\s");         
736                for(String v : filterValues) {
737                        //if v has whitespace chars (several words), make a "word1 AND word2 AND..." subquery
738                        if(pattern.matcher(v).find()) {
739                                BooleanQuery bq = new BooleanQuery();
740//bug: text with spaces and 'of', 'and', 'for', etc., fails to match anything (fix: use the same analyzer here as when indexing!)
741//                              for(String w : v.split("\\s+")) {
742//                                      bq.add(new TermQuery(new Term(filterField, w.toLowerCase())), Occur.MUST);
743//                                      LOG.debug("subQuery, add part: " + w.toLowerCase());
744//                              }
745                                try {
746                                        //use the same analyser as when indexing
747                                        TokenStream tokenStream = analyzer.tokenStream(filterField, new StringReader(v));
748                                        CharTermAttribute chattr = tokenStream.addAttribute(CharTermAttribute.class);
749                                        tokenStream.reset();
750                                        while(tokenStream.incrementToken()) {
751                                                //'of', 'and', 'for',.. never occur as tokens (this is how the analyzer works)
752                                                String token = chattr.toString();
753                                                bq.add(new TermQuery(new Term(filterField, token)), Occur.MUST);
754//                                              LOG.debug("subQuery, add part: " + token);
755                                        }
756                                        tokenStream.end(); 
757                                        tokenStream.close();
758                                } catch (IOException e) {
759                                        //should never happen as we use StringReader
760                                        throw new RuntimeException("Failed to open a token stream; "
761                                                        + "field:" + filterField + ", value:" + v,e);
762                                }
763                                query.add(bq, Occur.SHOULD);
764                        } else {
765                                query.add(new TermQuery(new Term(filterField, v.toLowerCase())), Occur.SHOULD);
766                        }                       
767                }
768                
769                return query;
770        }

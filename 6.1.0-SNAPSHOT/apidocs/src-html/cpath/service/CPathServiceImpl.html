139        private void loadModel() {              
140                //fork the model loading (which takes quite a while)
141                ExecutorService executor = Executors.newSingleThreadExecutor();
142                executor.execute(
143                        new Runnable() {
144                        @Override
145                        public void run() {
146                                Model model = CPathUtils.loadMainBiopaxModel();
147                                // set for this service
148                                paxtoolsModel = model;
149                                if(paxtoolsModel != null) {
150                                        model.setXmlBase(CPathSettings.getInstance().getXmlBase());
151                                        log.info("Main BioPAX model (in-memory) is now ready for queries.");
152                                        searcher = new SearchEngine(paxtoolsModel, 
153                                                        CPathSettings.getInstance().indexDir());
154                                        ((SearchEngine) searcher).setMaxHitsPerPage(
155                                                Integer.parseInt(CPathSettings.getInstance().getMaxHitsPerPage()));
156                                }       
157                        }
158                });
159                executor.shutdown();
160                //won't wait (nothing else to do)
161        }

164        @Override
165        public ServiceResponse search(String queryStr, 
166                        int page, Class<? extends BioPAXElement> biopaxClass, String[] dsources, String[] organisms) 
167        {
168                if(!paxtoolsModelReady() || searcher == null) 
169                        return new ErrorResponse(MAINTENANCE,"Waiting for the initialization to complete (try later)...");
170                
171                try {
172                        // do search
173                        SearchResponse hits = searcher.search(queryStr, page, biopaxClass, dsources, organisms);
174                        
175                        hits.setComment("Search '" + queryStr  + "' in " + 
176                                ((biopaxClass == null) ? "all types" : biopaxClass.getSimpleName()) 
177                                + "; ds: " + Arrays.toString(dsources)+ "; org.: " + Arrays.toString(organisms));
178                        
179                        return hits;
180                        
181                } catch (Exception e) {
182                        log.error("search() failed", e);
183                        return new ErrorResponse(INTERNAL_ERROR, e);
184                }
185                
186        }

189        @Override
190        public ServiceResponse fetch(final OutputFormat format, final String... uris) {
191                if (uris.length == 0)
192                        return new ErrorResponse(NO_RESULTS_FOUND,
193                                        "No URIs were specified for the query");
194                
195                if(!paxtoolsModelReady()) 
196                        return new ErrorResponse(MAINTENANCE,"Waiting for the initialization to complete (try later)...");
197                
198                // extract/convert a sub-model
199                try {
200                        final String[] mappedUris = findUrisByIds(uris);
201                        Set<BioPAXElement> elements = urisToBpes(paxtoolsModel, mappedUris);
202                        
203                        if(elements.isEmpty()) {
204                                return new ErrorResponse(NO_RESULTS_FOUND,
205                                                "No BioPAX objects found by URI(s): " + Arrays.toString(uris));
206                        }
207                                        
208                        //auto-complete (add important child/parent elements)   
209                        elements = (new Completer(simpleIO.getEditorMap())).complete(elements, paxtoolsModel); 
210                        assert !elements.isEmpty() : "Completer.complete() produced empty set from not empty";
211
212                        Model m = cloner.clone(paxtoolsModel, elements);
213                        m.setXmlBase(paxtoolsModel.getXmlBase());
214                        return convert(m, format);
215                        
216                } catch (Exception e) {
217                        return new ErrorResponse(INTERNAL_ERROR, e);
218                }
219
220    }

223        private Filter[] createFilters(String[] organisms, String[] datasources) {
224                ArrayList<Filter> filters = new ArrayList<Filter>();
225                
226                if(getBlacklist() != null)
227                        filters.add(new UbiqueFilter(getBlacklist().getListed()));
228                
229                if(organisms != null && organisms.length > 0)
230                        filters.add(new OrganismFilter(organisms));
231                
232                if(datasources != null && datasources.length > 0)
233                        filters.add(new DataSourceFilter(datasources));
234                
235                return filters.toArray(new Filter[]{});
236        }

239        @Override
240        public ServiceResponse getNeighborhood(final OutputFormat format, 
241                final String[] sources, final Integer limit, Direction direction, 
242                final String[] organisms, final String[] datasources)
243        {
244                if(!paxtoolsModelReady()) 
245                        return new ErrorResponse(MAINTENANCE,"Waiting for the initialization to complete (try later)...");
246                
247                final String[] src = findUrisByIds(sources);
248
249                if(direction == null) {
250                        direction = Direction.UNDIRECTED;       
251                }
252                
253                // execute the paxtools graph query
254                try {
255                        // init source elements
256                        Set<BioPAXElement> elements = urisToBpes(paxtoolsModel, src);
257                        if(elements.isEmpty()) {
258                                return new ErrorResponse(NO_RESULTS_FOUND,
259                                                "No BioPAX objects found by URI(s): " + Arrays.toString(src));
260                        }
261
262                        // Execute the query, get result elements
263                        elements = QueryExecuter.runNeighborhood(elements, paxtoolsModel,
264                                        limit, direction, createFilters(organisms, datasources));
265
266                        if(elements != null) {
267                                // auto-complete (gets a reasonable size sub-model)
268                                elements = (new Completer(simpleIO.getEditorMap())).complete(elements, paxtoolsModel);
269                                Model m = cloner.clone(paxtoolsModel, elements);
270                                m.setXmlBase(paxtoolsModel.getXmlBase());
271                                return convert(m, format);
272                        } else {
273                                return new ErrorResponse(NO_RESULTS_FOUND,
274                                                "No results found by URI(s): " + Arrays.toString(src));
275                        }
276                } catch (Exception e) {
277                        return new ErrorResponse(INTERNAL_ERROR, e);
278                }

323        @Override
324        public ServiceResponse getPathsFromTo(final OutputFormat format, 
325                final String[] sources, final String[] targets, final Integer limit,
326                final String[] organisms, final String[] datasources)
327        {
328                if(!paxtoolsModelReady()) 
329                        return new ErrorResponse(MAINTENANCE,"Waiting for the initialization to complete (try later)...");
330                
331                final String[] src = findUrisByIds(sources);
332                final String[] tgt = findUrisByIds(targets);
333                
334                // execute the paxtools graph query     
335                try {
336                        // init source and target elements
337                        Set<BioPAXElement> source = urisToBpes(paxtoolsModel, src);
338                        if(source.isEmpty()) {
339                                return new ErrorResponse(NO_RESULTS_FOUND,
340                                                "No source BioPAX objects found by URI(s): " + Arrays.toString(src));
341                        }
342                        Set<BioPAXElement> target = urisToBpes(paxtoolsModel, tgt);
343                        if(target.isEmpty()) {
344                                return new ErrorResponse(NO_RESULTS_FOUND,
345                                                "No target BioPAX objects found by URI(s): " + Arrays.toString(tgt));
346                        }
347
348                        // Execute the query
349                        Set<BioPAXElement> elements = (target==null || target.isEmpty()) 
350                                        ? QueryExecuter.runPathsBetween(source, paxtoolsModel, limit,
351                                                        createFilters(organisms, datasources))
352                                                        : QueryExecuter.runPathsFromTo(source, target, 
353                                                                        paxtoolsModel, LimitType.NORMAL, limit,
354                                                                        createFilters(organisms, datasources));
355
356                                        if(elements != null) {
357                                                // auto-complete (gets a reasonable size sub-model)
358                                                elements = (new Completer(simpleIO.getEditorMap())).complete(elements, paxtoolsModel);
359                                                Model m = cloner.clone(paxtoolsModel, elements);
360                                                m.setXmlBase(paxtoolsModel.getXmlBase());
361                                                return convert(m, format);
362                                        } else {
363                                                return new ErrorResponse(NO_RESULTS_FOUND,
364                                                                "No results found; source: " + Arrays.toString(src)
365                                                                +  ", target: " + Arrays.toString(tgt));
366                                        }       
367                } catch (Exception e) {
368                        return new ErrorResponse(INTERNAL_ERROR, e);
369                }
370
371        }

429        /**
430         * Mapping to BioPAX URIs.
431         *
432         * It does not "understand" RefSeq Versions and UniProt Isoforms 
433         * (one has to submit canonical identifiers, i.e, ones without "-#" or ".#").
434         * 
435         * TODO map to either gene or chemical entities or entity references instead of any xrefs?..
436         * 
437         * @param identifiers - a list of genes/protein or molecules as: \
438         *              HGNC symbols, UniProt, RefSeq, ENS* and NCBI Gene identifiers; or\
439         *              CHEBI, InChIKey, ChEMBL, DrugBank, CID: (PubChem), SID: (PubChem), KEGG Compound, PharmGKB, or chem. name.
440         * @return URIs
441         */
442        private String[] findUrisByIds(String[] identifiers)
443        {
444                if (identifiers.length == 0)
445                        return identifiers;
446                
447                Set<String> uris = new TreeSet<String>();
448
449                // id-mapping: get primary IDs where possible; 
450                // build a Lucene query string (will be eq. to xrefid:"A" OR xrefid:"B" OR ...)
451                final StringBuilder q = new StringBuilder();
452                for (String identifier : identifiers) {
453                        
454                        String id = identifier;
455                        
456                        if(identifier.toLowerCase().startsWith("http://") 
457                                        || identifier.toLowerCase().startsWith("urn:")) 
458                        {
459                                // it must be an existing URI (a user hopes so)
460                                uris.add(identifier);
461                                
462                                if(identifier.startsWith("http://identifier.org/")) {
463                                        //also extract the id from the URI to map it (below) to the primary id/URI
464                                        id = CPathUtils.idfromNormalizedUri(identifier);
465                                } else //no id-mapping required
466                                        continue; //go to next identifier
467                        } 
468                        
469                        // do gene/protein/chemical id-mapping;
470                        // mapping can be ambiguous, but this is OK for queries (unlike when merging data)
471                        Set<String> m = map(id);
472                        if (!m.isEmpty()) {
473                                for(String ac : m) {
474                                        // add to the query string; 
475                                        // quotation marks around the query id are required
476                                        q.append("xrefid:\"").append(ac).append("\" "); 
477                                        log.debug("findUrisByIds, mapped " + id + " -> " + ac);
478                                }
479                        }
480                        
481                        // use the original id regardless the mapping results
482                        if(!q.toString().contains(id))
483                                q.append("xrefid:\"").append(id).append("\" "); 
484
485                }
486                
487                /* 
488                 * find existing Xref URIs by ids using cpath2 full-text search
489                 * and pagination; iterate until all hits/pages are read,
490                 * because our query is very specific - uses field and class -
491                 * we want all hits)
492                */
493                if (q.length() > 0) {
494                        final String query = q.toString().trim();
495                        log.debug("findUrisByIds, will run: " + query);
496                        int page = 0; // will use search pagination
497                        SearchResponse resp = (SearchResponse) search(query, page, Xref.class, null, null);
498                        log.debug("findUrisByIds, hits: " + resp.getNumHits());
499                        while (!resp.isEmpty()) {
500                                log.debug("Retrieving xref search results, page #" + page);
501                                for (SearchHit h : resp.getSearchHit()) 
502                                {
503                                        if("UnificationXref".equalsIgnoreCase(h.getBiopaxClass())
504                                                        || "RelationshipXref".equalsIgnoreCase(h.getBiopaxClass())) {
505                                                //exclude some RX types if the rel.type is set
506                                                if("RelationshipXref".equalsIgnoreCase(h.getBiopaxClass())) {
507                                                        RelationshipXref rx = null;
508                                                        rx = (RelationshipXref) paxtoolsModel.getByID(h.getUri());
509                                                        
510                                                        //TODO review/decide RX types to keep/exclude...
511                                                        //we created RXs with 'identity', 'see-also', etc. types when building the Warehouse and merging data
512                                                        if(rx.getRelationshipType()==null || 
513                                                                        rx.getRelationshipType().getTerm().contains("identity"))
514                                                                uris.add(h.getUri());
515                                                        
516                                                } else 
517                                                        uris.add(h.getUri());
518                                        }
519                                }
520                                // go next page
521                                resp = (SearchResponse) search(query, ++page, Xref.class, null, null);
522                        }
523                }
524                                
525                log.debug("findUrisByIds, seed Xrefs: " + uris + 
526                                " were mapped/found by orig. IDs: " + Arrays.toString(identifiers));
527                
528                return uris.toArray(new String[]{});
529        }

550                        return new ErrorResponse(INTERNAL_ERROR, e);
551                }

556        /**
557         * {@inheritDoc}
558         * 
559         * "Top pathway" can mean different things...
560         * 
561         * 1) One may want simply collect pathways which are not 
562         * values of any BioPAX property (i.e., a "graph-theoretic" approach, 
563         * used by {@link ModelUtils#getRootElements(org.biopax.paxtools.model.Model, Class)} method) and by
564         * BioPAX normalizer, which (in the cPath2 "premerge" stage),
565         * for all Entities, generates relationship xrefs to their "parent" pathways.
566         * 
567         * 2) Another approach would be to check whether specific (inverse) 
568         * properties, such as controlledOf, pathwayComponentOf and stepProcessOf, are empty.
569         * 
570         * Here we follow the second method.
571         */
572        @Override
573        public ServiceResponse topPathways(final String[] organisms, final String[] datasources) {
574                
575                if(!paxtoolsModelReady() || searcher == null) 
576                        return new ErrorResponse(MAINTENANCE,"Waiting for the initialization to complete (try later)...");
577                
578                SearchResponse topPathways = new SearchResponse();
579                final List<SearchHit> hits = topPathways.getSearchHit(); //empty list
580                int page = 0; // will use search pagination
581                
582                SearchResponse r = null;
583                try {
584                        r = searcher.search("*", page, Pathway.class, datasources, organisms);
585                        
586                } catch(Exception e) {
587                        log.error("topPathways() failed", e);
588                        return new ErrorResponse(INTERNAL_ERROR, e);
589                }
590                
591                //go over all hits, all pages
592                final int numPathways = r.getNumHits();
593                int processed = 0;
594                while(!r.isEmpty()) {
595                        log.debug("Retrieving top pathways search results, page #" + page);
596                        //keep only pathways where 'pathway' index field
597                        //is empty (no controlledOf and pathwayComponentOf values)
598                        for(SearchHit h : r.getSearchHit()) {
599                                if(h.getPathway().isEmpty() || 
600                                                (h.getPathway().size()==1 
601                                                        && h.getPathway().get(0).equalsIgnoreCase(h.getUri())
602                                                )
603                                        ) hits.add(h); //add to the list
604                                processed++;
605                        }
606                        
607                        if(processed >= numPathways)
608                                break; //may save us one uselss query
609                        
610                        // go next page
611                        try {
612                                r = searcher.search("*", ++page, Pathway.class, datasources, organisms);
613                        } catch(Exception e) {
614                                log.error("topPathways() failed", e);
615                                return new ErrorResponse(INTERNAL_ERROR, e);
616                        }
617                }
618                
619                // final touches...
620                topPathways.setNumHits(hits.size());
621                topPathways.setComment("Top Pathways (technically, each has empty index " +
622                                "field 'pathway'; that also means, they are neither components of " +
623                                "other pathways nor controlled of any process)");
624                topPathways.setMaxHitsPerPage(hits.size());
625                topPathways.setPageNo(0);
626                        
627                return topPathways;
628        }

641         * This utility method prepares the source or target 
642         * object sets for a graph query.
643         * 
644         * @param main source model
645         * @param ids specific source or target set of IDs
646         * @return related biopax elements
647         */
648        private Set<BioPAXElement> urisToBpes(Model model, String[] ids)
649        {
650                Set<BioPAXElement> elements = new HashSet<BioPAXElement>();
651
652                for(Object id : ids) {
653                        BioPAXElement e = model.getByID(id.toString());
654
655                        if(e != null)
656                                elements.add(e);
657                        else
658                                log.warn("urisToBpes: unknown/broken URI: " + id);
659                }
660
661                return elements;
662        }

691        @Override
692        public Set<String> map(String fromDb, String fromId, String toDb) {
693        Assert.hasText(fromId);
694        Assert.hasText(toDb);
695        
696        List<Mapping> maps;       
697        if(fromDb == null || fromDb.isEmpty()) {
698                maps = mappingsRepository.findBySrcIdAndDestIgnoreCase(fromId, toDb);
699        } else {        
700                //if possible, use a "canonical" id instead isoform, version, kegg gene...
701                // (e.g., uniprot.isoform, P04150-2 pair becomes uniprot, P04150)
702                String id = fromId;
703                String db = fromDb;
704                
705                //normalize the name
706                try {
707                                String stdDb = MiriamLink.getName(db);
708                                if(stdDb != null) // be safe
709                                        db = stdDb.toUpperCase();
710                        } catch (IllegalArgumentException e) {
711                        }               
712                
713                if(db.equalsIgnoreCase("uniprot isoform") 
714                                || db.equalsIgnoreCase("uniprot.isoform")) 
715                {
716                        int idx = id.lastIndexOf('-');
717                        if(idx > 0) {//using corr. UniProt ID instead
718                                id = id.substring(0, idx);
719                                db = "UNIPROT";
720                        }
721                }
722                else if(db.toUpperCase().startsWith("UNIPROT")) {
723                        //e.g., 'UNIPROT' instead of 'UniProt Knowledgebase'
724                        db = "UNIPROT";
725                }
726                else if(db.toUpperCase().startsWith("SWISSPROT")) {
727                        db = "UNIPROT";
728                }
729                else if(db.equalsIgnoreCase("refseq")) {
730                        //strip, e.g., refseq:NP_012345.2 to refseq:NP_012345
731                        int idx = id.lastIndexOf('.');
732                        if(idx > 0)
733                                id = id.substring(0, idx);
734                } 
735                else if(db.toLowerCase().startsWith("kegg") && id.matches(":\\d+$")) {
736                        int idx = id.lastIndexOf(':');
737                        if(idx > 0) {
738                                id = id.substring(idx + 1); //it's NCBI Gene ID;
739                                db = "NCBI GENE";
740                        }
741                }
742                else if(db.equalsIgnoreCase("GENEID") || db.equalsIgnoreCase("ENTREZ GENE")) {
743                        db = "NCBI GENE";
744                } else if(db.toUpperCase().contains("PUBCHEM") && 
745                                (db.toUpperCase().contains("SUBSTANCE") || db.toUpperCase().contains("SID"))) {
746                        db = "PUBCHEM-SUBSTANCE";
747                } else if(db.toUpperCase().contains("PUBCHEM") && 
748                                (db.toUpperCase().contains("COMPOUND") || db.toUpperCase().contains("CID"))) {
749                        db = "PUBCHEM-COMPOUND";
750                }
751                
752                maps = mappingsRepository
753                        .findBySrcIgnoreCaseAndSrcIdAndDestIgnoreCase(db, id, toDb);
754        }
755        
756        Set<String> results = new TreeSet<String>();        
757        for(Mapping m : maps) {
758                results.add(m.getDestId());
759        }
760        
761                return results;
762        }

765        @Override
766        public Set<String> map(String identifier) {
767                if(identifier.startsWith("http://"))
768                        throw new AssertionError("URI is not allowed here; use ID");
769                
770                if(identifier.toUpperCase().startsWith("CHEBI:")) {
771                        // chebi -> to primary chebi id
772                        return map("CHEBI", identifier, "CHEBI");
773                } else if(identifier.length() == 25 || identifier.length() == 27) {
774                        // InChIKey identifier (25 or 27 chars long) -> to primary chebi id
775                        return map(null, identifier, "CHEBI"); //null - for looking in InChIKey, names, etc.
776                } else if(identifier.toUpperCase().startsWith("CID:")) {
777                        // - a hack to tell PubChem ID from NCBI Gene ID in graph queries
778                        return map("PubChem-compound", identifier.substring(4), "CHEBI");
779                } else if(identifier.toUpperCase().startsWith("SID:")) {
780                        // - a hack to tell PubChem ID from NCBI Gene ID in graph queries
781                        return map("PubChem-substance", identifier.substring(4), "CHEBI");
782                } else if(identifier.toUpperCase().startsWith("PUBCHEM:")) { 
783                        // - a hack to tell PubChem ID from NCBI Gene ID in graph queries
784                        return map("PubChem-compound", identifier.substring(8), "CHEBI");       
785                } else {
786                        // gene/protein name, id, etc. -> to primary uniprot AC
787                        Set<String> ret = new TreeSet<String>();
788                        ret.addAll(map(null, identifier, "UNIPROT"));
789                        if(ret.isEmpty()) //ChEMBL, DrugBank, chem. names, etc to ChEBI
790                                ret.addAll(map(null, identifier, "CHEBI"));
791                        return ret;
792                }
793        }

828        @Override
829        public LogEntity count(String date, LogEvent event, String ipAddr) 
830        {               
831                // find or create a record, count+1
832                LogEntity t = null;
833                try {
834                        t = (LogEntity) logEntitiesRepository
835                                .findByEventNameIgnoreCaseAndAddrAndDate(event.getName(), ipAddr, date);
836                } catch (DataAccessException e) {
837                        log.error("count(), findByEventNameIgnoreCaseAndAddrAndDate " +
838                                "failed to update for event: " + event.getName() + 
839                                ", IP: " + ipAddr + ", date: " + date, e);
840                }
841                
842                if(t == null) {                 
843                        t = new LogEntity(date, event, ipAddr);
844                }
845                
846                t.setCount(t.getCount() + 1);
847                
848                return logEntitiesRepository.save(t);
849        }

852        @Override
853        public ValidatorResponse validationReport(String provider, String file) {
854                ValidatorResponse response = new ValidatorResponse();
855                Metadata metadata = metadataRepository.findByIdentifier(provider);
856                for (Content content : metadata.getContent()) {
857                        String current = content.getFilename();                 
858                        
859                        if(file != null && !file.equals(current))
860                                continue; 
861                        //file==null means all files                    
862                        
863                        try {
864                                // unmarshal and add
865                                ValidatorResponse resp = (ValidatorResponse) ValidatorUtils.getUnmarshaller()
866                                        .unmarshal(new GZIPInputStream(new FileInputStream(content.validationXmlFile())));
867                                assert resp.getValidationResult().size() == 1;                          
868                                response.getValidationResult().addAll(resp.getValidationResult());                              
869                        } catch (Exception e) {
870                                log.error("validationReport: failed converting the XML response to objects", e);
871                        }
872                        
873                        if(current.equals(file))
874                                break;
875                }
876
877                return response;
878        }

894        @Override
895        public Metadata save(Metadata metadata) {               
896                log.info("Saving metadata: " + metadata.getIdentifier());
897                
898                if(metadata.getId() != null) { //update
899                        metadata = metadataRepository.save(metadata);
900                } else {                
901                        Metadata existing = metadataRepository.findByIdentifier(metadata.getIdentifier());
902                        if(existing != null)  {//update (except for the Content list, which should not be touched unless in Premerge)
903                                existing.setAvailability(metadata.getAvailability());
904                                existing.setCleanerClassname(metadata.getCleanerClassname());
905                                existing.setConverterClassname(metadata.getConverterClassname());
906                                existing.setDescription(metadata.getDescription());
907                                existing.setName(metadata.getName());
908                                existing.setIconUrl(metadata.getIconUrl());
909                                existing.setNumInteractions(metadata.getNumInteractions());
910                                existing.setNumPathways(metadata.getNumPathways());
911                                existing.setNumPhysicalEntities(metadata.getNumPhysicalEntities());
912                                existing.setPubmedId(metadata.getPubmedId());
913                                existing.setType(metadata.getType());
914                                existing.setUrlToData(metadata.getUrlToData());
915                                existing.setUrlToHomepage(metadata.getUrlToHomepage());
916                                metadata = existing;
917                                //the jpa managed (persistent) entity will be auto-updated/flashed
918                        }
919                        
920                        metadata = metadataRepository.save(metadata);
921                }
922                
923                return metadata;
924    }

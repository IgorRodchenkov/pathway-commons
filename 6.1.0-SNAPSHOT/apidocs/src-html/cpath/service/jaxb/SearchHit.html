001package cpath.service.jaxb;
002
003import java.io.Serializable;
004import java.util.ArrayList;
005import java.util.List;
006import javax.xml.bind.annotation.XmlAccessType;
007import javax.xml.bind.annotation.XmlAccessorType;
008import javax.xml.bind.annotation.XmlElement;
009import javax.xml.bind.annotation.XmlType;
010
011import org.apache.commons.lang.StringEscapeUtils;
012
013/**
014 * Search hit java bean.
015 * 
016 * @author rodche
017 *
018 */
019@XmlAccessorType(XmlAccessType.FIELD)
020@XmlType(
021        name = "SearchHit", 
022        propOrder = {
023    "uri",
024    "biopaxClass",
025    "name",
026    "dataSource",
027    "organism",
028    "pathway",
029    "excerpt",
030    "size"
031        }
032)
033public class SearchHit implements Serializable {
034    @XmlElement(required = true)
035    private String uri;
036    @XmlElement(required = true)
037    private String biopaxClass;
038    private String name;
039    private List<String> dataSource;
040    private List<String> organism;
041    private List<String> pathway;
042    private String excerpt;
043    private Integer size;
044
045    public SearchHit() {
046        }
047
048    public String getUri() {
049        return uri;
050    }
051    public void setUri(String value) {
052        this.uri = value;
053    }
054        
055        public String getName() {
056        return this.name;
057    }
058        public void setName(String name) {
059                this.name = name;
060        }
061
062    /**
063     * The BioPAX type (short name),
064     * e.g., "ProteinReference". 
065     * @return
066     */
067    public String getBiopaxClass() {
068        return biopaxClass;
069    }
070
071    public void setBiopaxClass(String value) {
072        this.biopaxClass = value;
073    }
074
075    /**
076     * The list of data source (Provenance) URIs. 
077     * @return
078     */
079    public List<String> getDataSource() {
080        if (dataSource == null) {
081            dataSource = new ArrayList<String>();
082        }
083        return this.dataSource;
084    }
085    public void setDataSource(List<String> dataSource) {
086                this.dataSource = dataSource;
087        }
088
089    /**
090     * The list of organism (BioSource) URIs. 
091     * @return
092     */
093    public List<String> getOrganism() {
094        if (organism == null) {
095            organism = new ArrayList<String>();
096        }
097        return this.organism;
098    }
099    public void setOrganism(List<String> organism) {
100                this.organism = organism;
101        }
102   
103    /**
104     * The list of parent pathwaya' URIs. 
105     * @return
106     */
107    public List<String> getPathway() {
108        if (pathway == null) {
109            pathway = new ArrayList<String>();
110        }
111        return this.pathway;
112    }
113    public void setPathway(List<String> pathway) {
114                this.pathway = pathway;
115        }
116
117    
118    public String getExcerpt() {
119        return excerpt;
120    }
121    public void setExcerpt(String value) {
122        this.excerpt = value;
123    }
124
125    /**
126     * For a BioPAX Interaction or Pathway,
127     * this is a number of associated processes; in other words,
128     * estimated size of the sub-network.
129     * 
130     * @return
131     */
132    public Integer getSize() {
133        return size;
134    }
135    public void setSize(Integer value) {
136        this.size = value;
137    }
138    
139    
140    @Override
141    public String toString() {
142        return (name != null) ? StringEscapeUtils.unescapeHtml(name) : uri;
143    }
144}

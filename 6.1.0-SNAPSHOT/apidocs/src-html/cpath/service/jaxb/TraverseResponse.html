001package cpath.service.jaxb;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import javax.xml.bind.annotation.*;
007
008@XmlRootElement(name="traverseResponse")
009@XmlAccessorType(XmlAccessType.FIELD)
010@XmlType(name = "TraverseResponse")
011public class TraverseResponse extends ServiceResponse {
012    @XmlAttribute
013    private String propertyPath;
014
015    private List<TraverseEntry> traverseEntry;
016    
017        public TraverseResponse() {
018        }
019    
020    public String getPropertyPath() {
021                return propertyPath;
022        }
023        public void setPropertyPath(String propertyPath) {
024                this.propertyPath = propertyPath;
025        }
026
027        public List<TraverseEntry> getTraverseEntry() {
028                if(traverseEntry == null) {
029                        traverseEntry = new ArrayList<TraverseEntry>();
030                }
031                return traverseEntry;
032        }
033        public void setTraverseEntry(List<TraverseEntry> traverseEntry) {
034                this.traverseEntry = traverseEntry;
035        }
036        
037        @Override
038        @XmlTransient
039        public boolean isEmpty() {
040                if(getTraverseEntry().isEmpty())
041                        return true;
042                
043                for(TraverseEntry ent : traverseEntry) {
044                        if(!ent.isEmpty())
045                                return false;
046                }
047                
048                return  true; //empty
049        } 
050
051}

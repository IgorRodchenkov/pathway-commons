001package cpath.jpa;
002
003import java.util.Collection;
004import java.util.HashSet;
005import java.util.Set;
006
007import javax.persistence.Column;
008import javax.persistence.Embeddable;
009import javax.persistence.EnumType;
010import javax.persistence.Enumerated;
011
012import org.springframework.util.Assert;
013
014import cpath.config.CPathSettings;
015import cpath.service.Cmd;
016import cpath.service.ErrorResponse;
017import cpath.service.GraphType;
018import cpath.service.OutputFormat;
019import cpath.service.Status;
020
021/**
022 * A pair of type and name that describes log entry.
023 * 
024 * A convenience bean for logging
025 * a server access event from different 
026 * perspectives. A list of such value 
027 * pairs can be associated with a request IP address
028 * and date and used to increment corresponding access counts 
029 * all at once, e.g., to simultaneously log which data sources
030 * were used, format, command, filename, etc., from one web request.
031 * 
032 * @see LogEntity
033 * 
034 * @author rodche
035 *
036 */
037@Embeddable
038public class LogEvent {
039        
040        @Column(nullable=false)
041        @Enumerated(EnumType.STRING)
042        private LogType type;
043
044        @Column(nullable=false)
045        private String name;
046        
047        /**
048         * a special constant event type, because there is no DOWNLOAD command in the cpath2 api.
049         */
050        public static LogEvent DOWNLOAD = new LogEvent(LogType.COMMAND, "DOWNLOAD");
051        
052        /**
053         * a special constant event type, because there is no IDMAPPING command in the cpath2 api.
054         */
055        public static LogEvent IDMAPPING = new LogEvent(LogType.COMMAND, "IDMAPPING");
056        
057        /**
058         * a special constant event type for total access counts
059         */
060        public static LogEvent TOTAL = new LogEvent(LogType.TOTAL, LogType.TOTAL.description);
061        
062        /**
063         * a special constant event type for other formats access counts (e.g., XML, JSON search, id-mapping query results)
064         */
065        public static LogEvent FORMAT_OTHER = new LogEvent(LogType.FORMAT, "OTHER");
066        
067        
068        public LogEvent() {
069        }
070        
071        public LogEvent(LogType type, String name) {
072                Assert.notNull(type);
073                Assert.notNull(name);
074                
075                this.type = type;
076                this.name = name;
077        }
078        
079        
080        public static LogEvent from(GraphType graphType) {      
081                return new LogEvent(LogType.COMMAND, graphType.toString());
082        }
083        
084        public static LogEvent from(Cmd command) {      
085                return new LogEvent(LogType.COMMAND, command.toString());
086        }
087        
088        public static LogEvent from(OutputFormat outputFormat) {
089                return new LogEvent(LogType.FORMAT, outputFormat.toString());
090        }
091        
092        
093        /*for tests*/
094        static LogEvent fromStatusCode(int statusCode) {
095                return new LogEvent(LogType.ERROR, String.valueOf(statusCode));
096        }
097                
098        public static LogEvent from(Status status) {
099                return new LogEvent(LogType.ERROR, status.name());
100        }
101        
102        public static LogEvent from(ErrorResponse errorResponse) {
103                return from(errorResponse.getStatus());
104        }
105        
106        public static Set<LogEvent> fromProviders(Collection<String> providers) {
107                Set<LogEvent> set = new HashSet<LogEvent>();
108                for(String prov : providers)
109                        set.add(new LogEvent(LogType.PROVIDER, prov));
110                return set;
111        }
112        
113        
114        /**
115         * Creates a list of things to update counts for -
116         * name, format, provider - from the 
117         * auto-generated data archive file.
118         * 
119         * @param filename see {@link CPathSettings#biopaxExportFileName(String)}
120         *                      for how it's created.
121         * @return
122         */
123        public static Set<LogEvent> fromDownloads(String filename) {
124                Set<LogEvent> set = new HashSet<LogEvent>();
125
126                set.add(new LogEvent(LogType.FILE, filename));
127                set.add(LogEvent.DOWNLOAD);
128                
129                // extract the orig. data source's standard name -
130                // first, remove common prefix (incl. cPath2 instance name and ver.)
131                if(filename.startsWith(CPathSettings.getInstance().exportArchivePrefix())) {
132                        int idx = CPathSettings.getInstance().exportArchivePrefix().length();
133                        String s = filename.substring(idx);
134                        String[] parts = s.split("\\.");
135                        //a hack: in order to skip for by-organism and special archives
136                        if(Character.isUpperCase(parts[0].charAt(0)) 
137                                        && !"All".equalsIgnoreCase(parts[0])
138                                        && !"Warehouse".equalsIgnoreCase(parts[0])
139                                        && !"Detailed".equalsIgnoreCase(parts[0])) {
140                                set.add(new LogEvent(LogType.PROVIDER, parts[0]));
141                        }
142
143                        // extract the format
144                        OutputFormat format = null; 
145                        try { 
146                                format = OutputFormat.valueOf(parts[1]);
147                        } catch(Exception e) {
148
149                        }
150                        if(format != null)
151                                set.add(LogEvent.from(format));
152                        else {
153                                set.add(LogEvent.FORMAT_OTHER);
154                        }
155                } else {
156                        set.add(LogEvent.FORMAT_OTHER);
157                }
158                        
159                
160                return set;
161        }
162        
163
164        public LogType getType() {
165                return type;
166        }
167
168        public String getName() {//return copy
169                return name;
170        }
171
172        public void setType(LogType type) {
173                this.type = type;
174        }
175
176        public void setName(String name) {
177                this.name = name;
178        }
179        
180        
181        @Override
182        public String toString() {
183                return type + " " + name;
184        };
185}

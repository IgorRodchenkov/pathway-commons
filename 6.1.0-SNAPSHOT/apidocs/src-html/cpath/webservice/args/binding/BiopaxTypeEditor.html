028package cpath.webservice.args.binding;
029
030import java.beans.PropertyEditorSupport;
031
032import org.biopax.paxtools.controller.EditorMap;
033import org.biopax.paxtools.controller.SimpleEditorMap;
034import org.biopax.paxtools.model.BioPAXElement;
035import org.biopax.paxtools.model.BioPAXFactory;
036import org.biopax.paxtools.model.BioPAXLevel;
037
038
039/**
040 * Helps convert request PROVIDER_URL path values to a BioPAX type.
041 * 
042 * @author rodche
043 *
044 */
045public class BiopaxTypeEditor extends PropertyEditorSupport {
046        private static BioPAXFactory bioPAXFactory = BioPAXLevel.L3.getDefaultFactory();
047        private static EditorMap editorMap = SimpleEditorMap.L3;
048        
049        /* (non-Javadoc)
050         * @see java.beans.PropertyEditorSupport#setAsText(java.lang.String)
051         */
052        @Override
053        public void setAsText(String type) {
054                setValue(getSearchableBiopaxClassByName(type));
055        }
056        
057        /*
058         * Enables using arguments, BioPAX class names, in any case: 
059         *     ProteinReference, PROTEINREFERENCE, proteinreference, etc.
060         * 
061         */
062        private static Class<? extends BioPAXElement> getSearchableBiopaxClassByName(String type) 
063        {
064                // case sensitive -
065                //return bioPAXFactory.getImplClass(BioPAXLevel.L3.getInterfaceForName(type));
066                
067                // better - case insensitive way -
068                for(Class<? extends BioPAXElement> c : editorMap.getKnownSubClassesOf(BioPAXElement.class)) {
069                        if(c.getSimpleName().equalsIgnoreCase(type)) {
070                                if(c.isInterface() && bioPAXFactory.getImplClass(c) != null)
071                                        return c; // interface
072                        }
073                }
074                
075//              log.info("Illegal BioPAX class name '" +
076//                      type + "' (cannot this " +
077//                      "as a filter by type value in queries) ");
078                throw new IllegalArgumentException("Illegal BioPAX class name '" + type);
079        }
080}

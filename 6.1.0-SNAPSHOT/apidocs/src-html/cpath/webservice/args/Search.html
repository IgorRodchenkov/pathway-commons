001package cpath.webservice.args;
002
003import javax.validation.constraints.Min;
004
005import org.biopax.paxtools.model.BioPAXElement;
006import org.hibernate.validator.constraints.NotBlank;
007
008public class Search {
009        @NotBlank(message="Parameter 'q' (a Lucene query string) is blank (not specified).")
010        private String q;
011
012        private Class<? extends BioPAXElement> type;
013
014        private String[] organism;
015        private String[] datasource;
016        
017        @Min(0)
018        private Integer page;
019
020        public Search() {
021                page = 0;
022        }
023
024        public String getQ() {
025                return q;
026        }
027
028        public void setQ(String q) {
029                this.q = q;
030        }
031
032        public Class<? extends BioPAXElement> getType() {
033                return type;
034        }
035
036        public void setType(Class<? extends BioPAXElement> type) {
037                this.type = type;
038        }
039
040
041        public String[] getOrganism() {
042                return organism;
043        }
044
045        public void setOrganism(String[] organism) {
046                this.organism = organism;
047        }
048
049        public String[] getDatasource() {
050                return datasource;
051        }
052
053        public void setDatasource(String[] datasource) {
054                this.datasource = datasource;
055        }
056
057        public Integer getPage() {
058                return page;
059        }
060
061        public void setPage(Integer page) {
062                this.page = page;
063        }
064                
065}

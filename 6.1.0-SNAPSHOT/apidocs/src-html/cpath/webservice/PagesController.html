125    @RequestMapping("/downloads")
126    public String downloads(Model model, HttpServletRequest request) {
127
128        // get the sorted list of files to be shared on the web
129        String path = CPathSettings.getInstance().downloadsDir(); 
130        File[] list = new File(path).listFiles();
131        
132        Map<String,String> files = new TreeMap<String,String>();
133        
134        for(int i = 0 ; i < list.length ; i++) {
135                File f = list[i];
136                String name = f.getName();
137                long size = f.length();
138                
139                if(!name.startsWith(".")) {
140                        StringBuilder sb = new StringBuilder();
141                        sb.append("size: ").append(FileUtils.byteCountToDisplaySize(size));
142                        List<Object[]> dl = service.log().downloadsWorld(null, name);
143                        String topCountry = null;
144                        long topCount = 0;
145                        long total = 0;
146                        Iterator<Object[]> it = dl.iterator();
147                        it.next(); //skip title line
148                        while(it.hasNext()) {
149                                Object[] a = it.next();
150                                long count = (Long) a[1];
151                                total += count;
152                                if(count > topCount) {
153                                        topCount = count;
154                                        topCountry = (String) a[0];
155                                }                                       
156                        }
157                        
158                        sb.append("; downloads: ").append(total);
159                        if(topCount > 0) {
160                                sb.append(", mostly from ")
161                                .append((topCountry != null && !topCountry.isEmpty()) 
162                                                ? topCountry : "Local/Unknown")
163                                .append(" [").append(topCount).append("]");
164                        }
165                        
166                        files.put(name, sb.toString());
167                }
168        }
169        
170        model.addAttribute("files", files.entrySet());
171                
172                return "downloads";
173    }

232    public  @ResponseBody byte[] icon(HttpServletResponse response) 
233                throws IOException {
234        
235        String cpathLogoUrl = CPathSettings.getInstance().getLogoUrl();
236        
237                byte[] iconData = null;
238
239                try {
240                        BufferedImage image = ImageIO.read(new URL(cpathLogoUrl));
241                        if(image != null) {
242                                image = scaleImage(image, 16, 16, null);
243                                ByteArrayOutputStream baos = new ByteArrayOutputStream();
244                                ImageIO.write(image, "gif", baos);
245                                baos.flush();
246                                iconData = baos.toByteArray();
247                        }
248                } catch (IOException e) {
249                        log.error("Failed to load icon image from " +  cpathLogoUrl, e);
250                }
251                
252        return iconData;
253    }

260        /**
261         * Recursively gets the sorted filename->size map
262         * from the cpath2 home and dir and 'data' sub-directory.
263         * 
264         * TODO consider using a Tree object (set of nodes) in the future
265         * 
266     * @param path
267     * @param relativePath
268     * @param showDirNames
269     * @param goIntoDirs expand (list files in) the specified sub-directories, or all - if null.
270     * @return
271     */
272        private Map<String, String> files(String path, String relativePath, 
273                        boolean showDirNames, Collection<String> goIntoDirs) 
274        {
275                Map<String,String> files = new TreeMap<String,String>();
276                
277                String fullPath = (relativePath == null) ? path 
278                                : path + File.separator + relativePath;         
279                File[] list = new File(fullPath).listFiles();
280                
281        for(int i = 0 ; i < list.length ; i++) {
282                File f = list[i];
283                String name = f.getName();
284                
285                if(name.startsWith(".")) 
286                        continue; //skip system/hidden
287                
288                //add curr. rel. path to this filename
289                name = (relativePath == null) ? name 
290                                        : relativePath + File.separator + name;
291                
292                if(f.isDirectory() && (goIntoDirs==null || goIntoDirs.contains(name))) { 
293                        //deep traverse into the data dir. only
294                        files.putAll(files(path, name, showDirNames, goIntoDirs));
295                } else {        
296                        String size =  (f.isDirectory()) ? "directory"
297                                        : FileUtils.byteCountToDisplaySize(f.length());
298                        
299                        if(!f.isDirectory() || showDirNames)
300                                files.put(name, size);
301                }
302        }
303        
304        return files;
305        }

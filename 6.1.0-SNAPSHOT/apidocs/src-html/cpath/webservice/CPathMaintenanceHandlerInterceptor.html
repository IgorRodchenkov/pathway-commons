001package cpath.webservice;
002
003import javax.servlet.http.HttpServletRequest;
004import javax.servlet.http.HttpServletResponse;
005
006import org.springframework.stereotype.Component;
007import org.springframework.web.servlet.handler.HandlerInterceptorAdapter;
008
009import cpath.config.CPathSettings;
010
011/**
012 * @author rodche
013 *
014 */
015@Component
016public final class CPathMaintenanceHandlerInterceptor extends HandlerInterceptorAdapter
017{
018
019        @Override
020        public boolean preHandle(HttpServletRequest request, HttpServletResponse response,
021                        Object handler) throws Exception {
022                
023                String requestUri = request.getRequestURI();
024                
025                //disable user web service queries in admin mode is enabled
026                if(CPathSettings.getInstance().isAdminEnabled()                                 
027                        && !(  requestUri.contains("/resources") 
028                                || requestUri.contains("/help")
029                                || requestUri.contains("/admin")
030                                || requestUri.contains("/error")
031                                || requestUri.contains("/login")
032                                || requestUri.contains("/denied")
033                                || requestUri.contains("/home")
034                                || requestUri.contains("/datasources")
035                                || requestUri.contains("/metadata/")
036                                || requestUri.contains("/favicon")
037                                || requestUri.contains("/logback")
038                                || requestUri.contains("/log")
039                                || requestUri.contains("/formats")
040                                || requestUri.contains("/downloads")
041                                )
042                )
043                {
044                        response.sendError(503, CPathSettings
045                                .getInstance().property(CPathSettings.PROVIDER_NAME)
046                                + " service maintenance.");
047                        return false;
048                }
049                else
050                        return true; 
051        }
052
053}

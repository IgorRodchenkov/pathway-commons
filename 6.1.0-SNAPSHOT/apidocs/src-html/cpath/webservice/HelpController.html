060 * cPath2 Help REST Web Service.
061 * 
062 * Can return XML (default) or
063 * JSON (on request) serialized
064 * {@link Help} bean.
065 * 
066 * @author rodche

070        
071        public HelpController() {
072        }
073        
074        /**
075         * Customizes request parameters conversion to proper internal types,
076         * e.g., "network of interest" is recognized as GraphType.NETWORK_OF_INTEREST, etc.
077         * 
078         * @param binder
079         */
080        @InitBinder
081    public void initBinder(WebDataBinder binder) {
082        binder.registerCustomEditor(OutputFormat.class, new OutputFormatEditor());
083        binder.registerCustomEditor(GraphType.class, new GraphTypeEditor());
084        binder.registerCustomEditor(Cmd.class, new CmdEditor());
085        binder.registerCustomEditor(CmdArgs.class, new CmdArgsEditor());
086        binder.registerCustomEditor(Direction.class, new DirectionEditor());
087        binder.registerCustomEditor(Class.class, new BiopaxTypeEditor());
088    }

091        /*
092     * Using @Response with returning a bean
093     * makes it auto-generate xml or json, 
094     * depending on the client's http request
095     * (no extra coding required!)
096     */
097    @RequestMapping("/help")
098    public @ResponseBody Help getHelp() {
099        Help help = new Help();
100        help.setId("help");
101        help.setTitle("Help");
102        help.setInfo("Welcome to cPath2 Webservice Help");
103        help.setExample("help/commands");
104        // Help tree's five main branches:
105        help.addMember(getCommands()); // sub-tree for commands and their args info
106        help.addMember(getFormats());
107        help.addMember(getGraphTypes());
108        help.addMember(getBiopaxTypes());
109        return help;
110    }

113    /**
114     * Prints the XML schema.
115     * 
116     * @param writer
117     * @throws IOException
118     */
119    @RequestMapping("/help/schema")
120    public void getSchema(Writer writer, HttpServletResponse response) 
121                throws IOException 
122    {
123        BufferedReader bis = new BufferedReader(new InputStreamReader(
124                (new DefaultResourceLoader())
125                        .getResource("classpath:cpath/service/schema1.xsd")
126                                .getInputStream(), "UTF-8"));
127        
128        response.setContentType("application/xml");
129        
130        final String newLine = System.getProperty("line.separator");
131        String line = null;
132        while((line = bis.readLine()) != null) {
133                writer.write(line + newLine);
134        }
135    }

138    @RequestMapping("/help/commands")
139    public @ResponseBody Help getCommands() {
140        Help help = new Help();
141        for(Cmd c : Cmd.values()) {
142                help.addMember(getCommand(c));
143        }
144        help.setId("commands");
145        help.setInfo("cPath2 BioPAX L3 web service supports "
146                + Cmd.values().length + " commands");
147        help.setTitle("cPath2 Webservice Commands");
148        help.setExample("search?q=brca*&type=protein");
149        return help;
150    }

153    @RequestMapping("/help/commands/{cmd}")
154    public @ResponseBody Help getCommand(@PathVariable Cmd cmd) {
155        if(cmd == null) return getCommands();
156        Help help = new Help();
157        help.setId(cmd.name());
158                help.setTitle(cmd.name());
159                help.setInfo(cmd.getInfo());
160        help.setExample(cmd.getExample());
161        help.setOutput(cmd.getOutput());
162                for(CmdArgs a: cmd.getArgs()) {
163                        Help ah = new Help(a.name());
164                        ah.setTitle(a.name());
165                        ah.setInfo(a.getInfo());
166                        help.addMember(ah);
167                }
168        return help;
169    }

201        /**
202         * List of BioPAX L3 Classes
203         * 
204         * @return
205         */
206    @RequestMapping("/help/types")
207    public @ResponseBody Help getBiopaxTypes() {
208        Help help = new Help();
209        
210        for(Class<? extends BioPAXElement> t : 
211                SimpleEditorMap.L3.getKnownSubClassesOf(BioPAXElement.class)) 
212        {
213                if(BioPAXLevel.L3.getDefaultFactory().getImplClass(t) != null)
214                        help.addMember(new Help(t.getSimpleName()));
215        }
216        help.setId("types");
217        help.setTitle("BioPAX classes");
218        help.setInfo("Objects of the following BioPAX L3 class " +
219                "(and some abstract ones) " 
220                + System.getProperty("line.separator") +
221                "are persisted/indexed/searchable in the system " +
222                "(names are case insensitive):");
223        help.setExample("search?type=pathway&q=b*");
224        return help;
225    }

240    @RequestMapping("/help/types/{type}/properties") 
241    public @ResponseBody Help getBiopaxTypeProperties(@PathVariable Class<? extends BioPAXElement> type) {
242        final String id = type.getSimpleName() + " properties";
243        Help h = new Help(id);
244        h.setTitle(id);
245        h.setInfo("BioPAX properties " +
246                "for class: " + type.getSimpleName());
247        
248        EditorMap em = SimpleEditorMap.get(BioPAXLevel.L3);
249        for(PropertyEditor e : em.getEditorsOf(type)) 
250                h.addMember(new Help(e.toString()));
251        
252        return h;
253    }

001package cpath.converter.internal;
002
003import java.io.InputStream;
004import java.io.OutputStream;
005
006import org.biopax.paxtools.converter.psi.PsiToBiopax3Converter;
007
008public final class PsimiToComplexesConverterImpl extends BaseConverterImpl {
009
010        public void convert(InputStream is, OutputStream os) {
011                try {                           
012                        PsiToBiopax3Converter psimiConverter = new PsiToBiopax3Converter(xmlBase);
013                        //generate Complexes instead of MolecularInteractions
014                        psimiConverter.convert(is, os, true);
015                        os.close();
016                } catch (Exception e) {
017                        throw new RuntimeException("Failed to convert PSI-MI to BioPAX", e);
018                }
019        }
020
021}

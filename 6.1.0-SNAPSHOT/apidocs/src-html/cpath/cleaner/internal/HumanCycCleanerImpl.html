021        private static final Logger LOG = LoggerFactory.getLogger(HumanCycCleanerImpl.class);
022        
023        public void clean(InputStream data, OutputStream cleanedData)
024        {
025                try
026                {
027                        SimpleIOHandler simpleReader = new SimpleIOHandler(BioPAXLevel.L3);
028                        Model model = simpleReader.convertFromOWL(data);
029                        removeUnificationXrefInPhysicalEntities(model);
030                        cleanXrefIDs(model);
031                        cleanXrefDBName(model);
032                        cleanHtmlInDisplayNames(model);
033//                      fix_RDH14_NT5C1B_fusion(model); //the problem is not present in 17.1 HymanCyc biopax
034                        cleanMultipleUnificationXrefs(model);
035                        // set organism to all pathways, where it's null
036                        setOrganismHomoSapiens(model);
037                        simpleReader.convertToOWL(model, cleanedData);
038                }
039                catch (Exception e)
040                {
041                        throw new RuntimeException("Exception in HumanCycCleanerImpl.clean", e);
042                }
043        }
044
045        private void setOrganismHomoSapiens(Model model) {
046                Set<BioSource> organisms = model.getObjects(BioSource.class);
047                if(organisms.size() > 1) {
048                        LOG.error("Skip setting 'Homo sapiens' for HumanCyc Pathways, " +
049                                        "for there are several BioSources: " + organisms);
050                        return;
051                } else if(organisms.isEmpty()) {
052                        LOG.warn("Will create a new 'Homo sapiens' BioSource " +
053                                        "and set it for all no-organism Pathways in HumanCyc.");
054                }
055                
056                //there is only one BioSource (must be human); use it -
057                BioSource human = organisms.iterator().next();
058                for(Pathway pathway : model.getObjects(Pathway.class)) {
059                        if(pathway.getOrganism() == null)
060                                pathway.setOrganism(human);
061                }
062        }
063
064        /**
065         * Some UnificationXref ids starts with a blank, and some are like "CAS: 103-82-2".
066         * Xrefs to PDB contains small letters.
067         *  - not anymore in the 17.1 BioCyc release
068         */
069        protected void cleanXrefIDs(Model model)
070        {
071                for (Xref xr : model.getObjects(Xref.class))
072                {
073                        String id = xr.getId();
074
075                        if (id == null) continue;
076
077                        id = id.trim();
078
079                        if (id.contains(": "))
080                        {
081                                id = id.substring(id.indexOf(": ") + 2);
082                        }
083
084                        if (xr.getDb().equals("PDB") || xr.getDb().equals("Protein Data Bank"))
085                        {
086                                id = id.toUpperCase();
087                        }
088
089                        xr.setId(id);
090                }
091        }
092
093        /**
094         * HumanCyc refers to GenBank proteins as "Entrez".
095         * We change those xrefs to use "Protein GenBank Identifier"
096         * see {@code http://www.ebi.ac.uk/ontology-lookup/browse.do?ontName=MI&termId=MI%3A0851&termName=protein%20genbank%20identifier}
097         * 
098         */
099        protected void cleanXrefDBName(Model model)
100        {
101                for (Xref xr : model.getObjects(Xref.class))
102                {
103                        if(xr.getDb() == null)
104                                LOG.warn("Xref.db is NULL : " + xr.getRDFId());
105                        else if(xr.getDb().equals("Entrez")) 
106                                xr.setDb("Protein GenBank Identifier");
107                        else if(xr.getDb().equalsIgnoreCase("NCBI Taxonomy")) 
108                                xr.setDb("taxonomy");
109                }
110        }
111        
112        protected void removeUnificationXrefInPhysicalEntities(Model model)
113        {
114                for (PhysicalEntity pe : model.getObjects(PhysicalEntity.class))
115                {
116                        for (Xref xref : new HashSet<Xref>(pe.getXref()))
117                        {
118                                if (xref instanceof UnificationXref)
119                                {
120                                        pe.removeXref(xref);
121                                }
122                        }
123                }
124        }
125        
126        protected void cleanHtmlInDisplayNames(Model model)
127        {
128                for (Named named : model.getObjects(Named.class))
129                {
130                        String s = named.getDisplayName();
131                        if (s != null)
132                        {
133                                s = removeHTML(s);
134                                named.setDisplayName(s);
135                        }
136                        
137                        s = named.getStandardName();
138                        if (s != null)
139                        {
140                                s = removeHTML(s);
141                                named.setStandardName(s);
142                        }
143                }
144        }
145
146        /*no need to use this with the 17.1 BioCyc biopax release*/
147        private String removeHTML(String s)
148        {
149                s = s.replaceAll("</i>", "").replaceAll("<i>","");
150                s = s.replaceAll("</I>", "").replaceAll("<I>","");
151                s = s.replaceAll("</SUB>", "").replaceAll("<SUB>","");
152                s = s.replaceAll("</sub>", "").replaceAll("<sub>","");
153                s = s.replaceAll("</sup>", "").replaceAll("<sup>","");
154                s = s.replaceAll("</SUP>", "").replaceAll("<SUP>","");
155                return s;
156        }

158        /**
159         * The protein NT5C1B contains info of RDH14. This is due to an old incorrect prediction in 
160         * TrEMBL. It is fixed in Uniprot, but remained in HumanCyc.
161         * 
162         * @param model
163         */
164        protected void fix_RDH14_NT5C1B_fusion(Model model)
165        {
166                EntityReference er = (EntityReference) model.getByID(model.getXmlBase()+"ProteinReference147987");
167
168                if (er != null)
169                {
170                        clearRDH14(er);
171
172                        for (SimplePhysicalEntity pe : er.getEntityReferenceOf())
173                        {
174                                clearRDH14(pe);
175                        }
176                }
177        }
178
179        protected void clearRDH14(Named nd)
180        {
181                nd.setDisplayName("NT5C1B");
182                nd.setStandardName("NT5C1B");
183
184                // We need only one unification xref. Clear all others. 
185                for (Xref xref : new HashSet<Xref>(nd.getXref()))
186                {
187                        if (xref instanceof UnificationXref && !xref.getRDFId().equals("UnificationXref147997"))
188                        {
189                                nd.removeXref(xref);
190                        }
191                }
192
193                for (String name : new HashSet<String>(nd.getName()))
194                {
195                        if (name.startsWith("RETINOL") || name.startsWith("PANCREAS"))
196                        {
197                                nd.removeName(name);
198                        }
199                }
200        }

203        static

206                toKeep.put("Q9UKY3", "P23141");
207                toKeep.put("Q8N0Y7", "P18669");

214                toKeep.put("Q96K59", "Q99519");

217        protected void cleanMultipleUnificationXrefs(Model model)
218        {
219                for (EntityReference er : model.getObjects(EntityReference.class))
220                {
221                        for (String key : toKeep.keySet())
222                        {
223                                String val = toKeep.get(key);
224
225                                if (hasRef(er, key) && hasRef(er, val))
226                                {
227                                        keepOnlyThis(er, val);
228                                }
229                        }
230                }
231        }
232        
233        protected void keepOnlyThis(XReferrable ele, String ref)
234        {
235                Set<UnificationXref> removeSet = new HashSet<UnificationXref>();
236
237                for (Xref xref : ele.getXref())
238                {
239                        if (xref instanceof UnificationXref)
240                        {
241                                removeSet.add((UnificationXref) xref);
242
243                                if (!xref.getId().equals(ref)) 
244                                {
245                                        removeSet.add((UnificationXref) xref);
246                                }
247                        }
248                }
249
250                for (UnificationXref xref : removeSet)
251                {
252                        ele.removeXref(xref);
253                }
254        }
255        
256        protected boolean hasRef(XReferrable ele, String ref)
257        {
258                for (Xref xref : ele.getXref())
259                {
260                        if (xref.getId() != null && xref.getId().equals(ref)) return true;
261                }
262                return false;
263        }

001package cpath.dao;
002
003import java.io.File;
004import java.io.IOException;
005import java.text.DateFormat;
006import java.text.ParseException;
007import java.text.SimpleDateFormat;
008import java.util.Calendar;
009import java.util.Date;
010
011import com.maxmind.geoip.Location;
012import com.maxmind.geoip.LookupService;
013
014import cpath.config.CPathSettings;
015import cpath.jpa.Geoloc;
016
017/**
018 * @author rodche
019 *
020 */
021public final class LogUtils {
022        
023        static LookupService geolite; 
024        
025        public static final DateFormat ISO_DATE_FORMAT = new SimpleDateFormat("yyyy-MM-dd");
026        
027        static {
028                //will be downloaded if not exists already (one can delete the file to auto-update)
029                String localFileName = CPathSettings.getInstance().homeDir() + File.separator + "GeoLiteCity.dat";
030                try {
031                        CPathUtils.download(
032                                "http://geolite.maxmind.com/download/geoip/database/GeoLiteCity.dat.gz", 
033                                        localFileName, true, false); // - also gunzip                   
034                        geolite = new LookupService(
035                                localFileName,
036                                LookupService.GEOIP_MEMORY_CACHE
037                        );
038                } catch (IOException e) {
039                        throw new RuntimeException("Fauled initializing GeoLite LookupService", e);
040                }
041        }
042        
043        
044        protected LogUtils() {
045                throw new AssertionError("Not instantiable");
046        }
047        
048        
049        /**
050         * Gets a geographical location by IP address
051         * using the GeoLite database.
052         * 
053         * @param ipAddress
054         * @return location or null (when it cannot be found; e.g., if it's local IP)
055         */
056        public static Geoloc lookup(String ipAddress) {
057                Location geoloc = geolite.getLocation(ipAddress);
058                return (geoloc != null) 
059                        ? new Geoloc(geoloc.countryCode, geoloc.region, geoloc.city) 
060                                : null;
061        }
062        
063        
064        public static String today() {
065                return ISO_DATE_FORMAT.format(new Date());
066        }
067        
068        static String isoDate(Date date) {
069                return ISO_DATE_FORMAT.format(date);
070        }
071
072        
073        /**
074         * Get a new date (ISO) by adding or 
075         * subtracting (if days < 0) the number 
076         * of days to/from specified date.
077         * 
078         * @param date
079         * @param days
080         * @return
081         */
082        static String addIsoDate(Date date, int days) {
083                Calendar cal = Calendar.getInstance();    
084                cal.setTime(date);    
085                cal.add( Calendar.DATE, days );    
086                return ISO_DATE_FORMAT.format(cal.getTime()); 
087        }
088        
089        
090        /**
091         * Get a new date (ISO) by adding or 
092         * subtracting (if days < 0) the number 
093         * of days to/from specified date.
094         * 
095         * @param date
096         * @param days
097         * @return
098         */
099        public static String addIsoDate(String isoDate, int days) {
100                Date date;
101                try {
102                        date = ISO_DATE_FORMAT.parse(isoDate);
103                } catch (ParseException e) {
104                        throw new RuntimeException(e);
105                }
106                return addIsoDate(date, days);
107        }
108
109}

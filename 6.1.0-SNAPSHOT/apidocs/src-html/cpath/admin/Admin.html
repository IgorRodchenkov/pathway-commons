120    /**
121     * The big deal main.
122     * 
123     * @param params String[]
124     */    
125    public static void main(String[] params) throws Exception {
126        // "CPATH2_HOME" env. var must be set (mostly for logging)
127        cpath.property(HOME_DIR); //throws IllegalStateEx. is not set.
128
129        if(!Charset.defaultCharset().equals(Charset.forName("UTF-8")))
130                if(LOG.isWarnEnabled())
131                        LOG.warn("Default Charset, " + Charset.defaultCharset() 
132                                + " (is NOT 'UTF-8'...)");
133        
134        
135        // Cleanup arguments by removing empty/null strings from the end
136        // - possible result of calling this method from a script
137        List<String> argl = new ArrayList<String>();
138        for(String a : params) {
139                if(a == null || a.isEmpty() || a.equalsIgnoreCase("null"))
140                        break;
141                else
142                        argl.add(a);
143        }
144        final String[] args = argl.toArray(new String[]{});
145                argl = null;
146        
147        LOG.debug("Command-line arguments were: " + Arrays.toString(args));
148        
149        // sanity check
150        if (args.length == 0 || args[0].isEmpty()) {
151            System.err.println("Missing args to Admin.");
152                        System.err.println(Admin.usage());
153            System.exit(-1);
154        }
155        
156
157        // create the data dir. inside the home dir. if it does not exist
158                File dir = new File(cpath.dataDir());
159                if(!dir.exists()) {
160                        dir.mkdir();
161                }
162
163                if (args[0].equals(Cmd.INDEX.toString())) {
164                        
165                        createIndex();
166                        
167                } else if (args[0].equals(Cmd.FETCH_METADATA.toString())) {
168                        
169                        if (args.length == 1) {
170                                fetchMetadata("file:" + cpath.property(PROP_METADATA_LOCATION));
171                        } else {
172                                fetchMetadata(args[1]);
173                        }
174                        
175                } else if (args[0].equals(Cmd.CREATE_WAREHOUSE.toString())) {
176                        
177                                createWarehouse();
178                        
179                } else if (args[0].equals(Cmd.PREMERGE.toString())) {
180                        
181                        if (args.length > 1)
182                                runPremerge(args[1]);
183                        else
184                                // command without extra parameter
185                                runPremerge(null);
186                        
187                } else if (args[0].equals(Cmd.MERGE.toString())) {
188                        boolean force = false;
189                        for (int i = 1; i < args.length; i++) {
190                                if ("--force".equalsIgnoreCase(args[i])) {
191                                        force = true;
192                                }
193                        }
194
195                        runMerge(force);
196                        
197                } else if (args[0].equals(Cmd.EXPORT.toString())) {
198                        //(the first args[0] is the command name, the second - must be output file)
199                        if (args.length < 2)
200                                fail(args, "must provide at least an output file name (other parameters are optional).");
201                        else {
202                                String[] uris = new String[] {};
203                                String[] datasources = new String[] {};
204                                String[] types = new String[] {};
205                                boolean absoluteUris = false;
206                                                        
207                                for(int i=2; i < args.length; i++) {
208                                        if(args[i].equalsIgnoreCase("--output-absolute-uris"))
209                                                absoluteUris = true;
210                                        else if(args[i].toLowerCase().startsWith("--datasources="))
211                                                datasources = args[i].substring(14).split(",");
212                                        else if(args[i].toLowerCase().startsWith("--types="))
213                                                types = args[i].substring(8).split(",");
214                                        else if(args[i].toLowerCase().startsWith("--uris="))
215                                                uris = args[i].substring(7).split(",");
216                                        else 
217                                                LOG.error("Skipped unrecognized argument: " + args[i]);
218                                }
219                                
220                                exportData(args[1], uris, absoluteUris, datasources, types);
221                        }
222
223                        
224                } else if (args[0].equals(Cmd.CREATE_BLACKLIST.toString())) {
225                        
226                        createBlacklist();
227                        
228                } else if (args[0].equals(Cmd.CONVERT.toString())) {
229                        if (args.length < 4)
230                                fail(args, "provide at least three arguments.");
231                        OutputStream fos = new FileOutputStream(args[2]);
232                        OutputFormat outputFormat = OutputFormat.valueOf(args[3]);      
233                        Model model = (new SimpleIOHandler()).convertFromOWL(biopaxStream(args[1]));
234                        
235                        String idTypeOrLayout = (args.length >= 5) ? args[4] : null;
236                        
237                        if(args.length >= 6 && Boolean.parseBoolean(args[5])) {
238                                LOG.info("Merging equivalent interactions in the model...");
239                                ModelUtils.mergeEquivalentInteractions(model);
240                        }
241                        
242                        LOG.info("Start converting to " + outputFormat);
243                        convert(model, outputFormat, fos, idTypeOrLayout);
244                        
245                } else if (args[0].equals(Cmd.CREATE_DOWNLOADS.toString())) {
246                        
247                        createDownloads();
248                        
249                } else if (args[0].equals(Cmd.UPDATE_COUNTS.toString())) {
250                        
251                        updateCounts();
252                
253                } else if (args[0].equals(Cmd.EXPORT_LOG.toString())) { 
254                        if (args.length == 2 && "--clear".equalsIgnoreCase(args[1]))
255                                exportLog(true);
256                        else
257                                exportLog(false);       
258                } else if (args[0].equals(Cmd.IMPORT_LOG.toString())) { 
259                        
260                        if (args.length < 2) 
261                                fail(args, "No input file provided.");  
262                        else 
263                                importLog(args[1]);
264                        
265                } else if (args[0].equals(Cmd.ANALYSIS.toString())) {   
266                        
267                        if (args.length < 2) 
268                                fail(args, "No Analysis implementation class provided.");       
269                        else if(args.length == 2)
270                                executeAnalysis(args[1], true);
271                        else if(args.length > 2 && "--update".equalsIgnoreCase(args[2]))
272                                executeAnalysis(args[1], false);
273                
274                } else {
275                        System.err.println(usage());
276                }
277
278                // required because MySQL Statement
279                // Cancellation Timer thread is still running
280                System.exit(0);
281    }

284    /**
285     * Executes a code that uses or edits the main BioPAX model.
286     * 
287     * @param analysisClass a class that implements {@link Analysis} 
288     * @param readOnly
289     */
290    public static void executeAnalysis(String analysisClass, boolean readOnly) {
291        
292        if(!cpath.isAdminEnabled())
293                        throw new IllegalStateException("Maintenance mode is not enabled.");
294        
295        Analysis analysis = null;
296                try {
297                        Class<Analysis> c = (Class<Analysis>) Class.forName(analysisClass);
298                        analysis = c.newInstance();
299                } catch (Exception e) {
300                        throw new RuntimeException(e);
301                }
302        
303                Model model = CPathUtils.loadMainBiopaxModel();
304                analysis.execute(model);
305                
306                if(!readOnly) { //replace the main BioPAX model archive
307                        try {                   
308                                new SimpleIOHandler(BioPAXLevel.L3).convertToOWL(model, 
309                                        new GZIPOutputStream(new FileOutputStream(
310                                                        CPathSettings.getInstance().mainModelFile())));
311                        } catch (Exception e) {
312                                throw new RuntimeException("Failed updating the main BioPAX archive!", e);
313                        }
314                        
315                        LOG.warn("The main BioPAX model was modified; "
316                                + "do not forget to re-index, update counts, re-export other files, etc.");
317                }
318                        
319        }

323    public static void exportLog(boolean clear) throws IOException {
324                if(!cpath.isAdminEnabled())
325                        throw new IllegalStateException("Maintenance mode is not enabled.");
326                
327                //backup cpath2 access db to a CSV file
328                String filename = cpath.dataDir() + File.separator + "logentity.csv";
329                Connection conn = null;
330                try {
331                        Class.forName("org.h2.Driver");
332                        conn = DriverManager.getConnection("jdbc:h2:"+cpath.homeDir()+"/cpath2", "sa", "");
333                        new Csv()
334                                .write(conn, filename, "select * from logentity", "UTF-8");
335                        LOG.info("Saved current access log DB to " + filename);
336                        
337                        if(clear) {
338                                //purge existing access time and location history
339                                conn.createStatement().executeUpdate("delete from logentity;");
340                                conn.commit();
341                                LOG.info("Cleared access log DB");
342                        }
343                } catch (Exception e) {
344                        throw new RuntimeException(e);
345                } finally {
346                        try {conn.close();} catch (Exception e) {}
347                }
348        }

350    public static void importLog(String filename) throws IOException {
351                if(!cpath.isAdminEnabled())
352                        throw new IllegalStateException("Maintenance mode is not enabled.");
353                
354                //load cpath2 access db from a CSV file
355                String backup = cpath.dataDir() + File.separator + "logentity.csv.bak";
356                Connection conn = null;
357                try {
358                        Class.forName("org.h2.Driver");
359                        conn = DriverManager.getConnection("jdbc:h2:"+cpath.homeDir()+"/cpath2", "sa", "");
360                        
361                        //backup
362                        new Csv().write(conn, backup, "select * from logentity", "UTF-8");
363                        LOG.info("Saved current access log DB to " + backup);
364                        
365                        //clear all existing data
366                        conn.createStatement().executeUpdate("delete from logentity;");
367                        LOG.info("Cleared access log DB");
368                        
369                        conn.createStatement().executeUpdate("insert into logentity " +
370                                        "select * from CSVREAD('"+ filename +"')");                     
371                        conn.commit();
372                LOG.info("Imported access log entries from " + filename);
373                } catch (Exception e) {
374                        try {conn.rollback(); conn.close();} catch (Exception ex) {}
375                        throw new RuntimeException(e);
376                } finally {
377                        try {conn.close();} catch (Exception e) {}
378                }

409        /**
410         * Updates counts of pathways, etc. and saves in the Metadata table.
411         * 
412     * This depends on the full-text index, which must have been created already
413     * (otherwise, results will be wrong).
414     * 
415     * @throws IllegalStateException when not in maintenance mode
416     */
417        public static void updateCounts() {
418                if (!cpath.isAdminEnabled())
419                        throw new IllegalStateException("Maintenance mode is not enabled.");
420
421                ClassPathXmlApplicationContext context = new ClassPathXmlApplicationContext(
422                                new String[] { "classpath:META-INF/spring/applicationContext-jpa.xml" });
423
424                MetadataRepository metadataRepo = (MetadataRepository) context.getBean(MetadataRepository.class);
425                
426                //load the model
427                Model model = CPathUtils.loadMainBiopaxModel();
428                LOG.info("Loaded the BioPAX Model");
429                
430                // initialize the search engine
431                Searcher searcher = new SearchEngine(model, CPathSettings.getInstance().indexDir());
432
433                // prepare a list of all pathway type metadata to update
434                List<Metadata> pathwayMetadata = new ArrayList<Metadata>();
435                for (Metadata md : metadataRepo.findAll())
436                        if (!md.isNotPathwayData())
437                                pathwayMetadata.add(md);
438
439                // for each non-warehouse metadata entry, update counts of pathways, etc.
440                for (Metadata md : pathwayMetadata) {
441                        String name = md.standardName();
442                        String[] dsUrisFilter = new String[] { md.getUri() };
443
444                        SearchResponse sr = (SearchResponse) searcher.search("*", 0,
445                                        Pathway.class, dsUrisFilter, null);
446                        md.setNumPathways(sr.getNumHits());
447                        LOG.info(name + " - pathways: " + sr.getNumHits());
448
449                        sr = (SearchResponse) searcher.search("*", 0, Interaction.class,
450                                        dsUrisFilter, null);
451                        md.setNumInteractions(sr.getNumHits());
452                        LOG.info(name + " - interactions: " + sr.getNumHits());
453
454                        Integer count;
455                        sr = (SearchResponse) searcher.search("*", 0, PhysicalEntity.class,
456                                        dsUrisFilter, null);
457                        count = sr.getNumHits();
458                        sr = (SearchResponse) searcher.search("*", 0, Gene.class,
459                                        dsUrisFilter, null);
460                        count += sr.getNumHits();               
461                        md.setNumPhysicalEntities(count);
462                        LOG.info(name + " - molecules, complexes and genes: " + count);
463                }
464
465                metadataRepo.save(pathwayMetadata);
466
467                context.close();
468        }

497    /**
498     * Performs cpath2 Merge stage.
499     * @param force
500     * 
501     * @throws IllegalStateException when not maintenance mode
502     */
503    public static void runMerge(boolean force) {
504                if(!cpath.isAdminEnabled())
505                        throw new IllegalStateException("Maintenance mode is not enabled.");
506        
507                //disable 2nd level hibernate cache (ehcache)
508                // otherwise the merger eventually fails with a weird exception
509                // (this probably depends on the cache config. parameters)
510                System.setProperty("net.sf.ehcache.disabled", "true");
511                ClassPathXmlApplicationContext context = 
512                        new ClassPathXmlApplicationContext(new String[] {
513                                        "classpath:META-INF/spring/applicationContext-jpa.xml"
514                        });
515                
516                LOG.info("runMerge: --force=" + force);
517                
518                //create CPathService bean that provides access to JPA repositories (metadata, mapping)
519                CPathService service = context.getBean(CPathService.class);
520                Merger merger = new Merger(service, force);             
521                merger.merge(); //at the end, it saves the resulting integrated main biopax model to a special file at known location.
522                
523                context.close();                
524        }

604        /**
605         * Exports a cpath2 BioPAX sub-model or full model to the specified file.
606         * 
607         * @param output - output BioPAX file name (path)
608         * @param uris - optional, the list of valid (existing) URIs to extract a sub-model
609         * @param outputAbsoluteUris - if true, all URIs in the BioPAX elements 
610         * and properties will be absolute (i.e., no local relative URIs, 
611         * such as rdf:ID="..." or rdf:resource="#...", will be there in the output file.) 
612         * @param datasources filter by data source if 'uris' is not empty
613         * @param types filter by biopax type if 'uris' is not empty
614         * 
615         * @throws IOException, IllegalStateException (in maintenance mode)
616         */
617        public static void exportData(final String output, String[] uris, boolean outputAbsoluteUris, 
618                        String[] datasources, String[] types) throws IOException 
619        {       
620                if(uris == null) 
621                        uris = new String[]{};
622                if(datasources == null) 
623                        datasources = new String[]{};
624                if(types == null) 
625                        types = new String[]{};
626                
627                //load the model
628                Model model = CPathUtils.loadMainBiopaxModel();
629                LOG.info("Loaded the BioPAX Model");
630
631                if(uris.length == 0 && (datasources.length > 0 || types.length > 0)) {
632                                        
633                        // initialize the search engine
634                        Searcher searcher = new SearchEngine(model, CPathSettings.getInstance().indexDir());
635
636                        Collection<String> selectedUris = new HashSet<String>();
637                        
638                        if(types.length>0) {
639                                //collect biopax object URIs of the specified types and sub-types, and data sources if specified
640                                //(child biopax elements will be auto-included during the export to OWL)
641                                for(String bpInterfaceName : types) {
642                                        selectedUris.addAll(findAllUris(searcher, 
643                                                        biopaxTypeFromSimpleName(bpInterfaceName), datasources, null)); 
644                                }
645                        } else {
646                                //collect all Entity URIs filtered by the not empty data sources list
647                                //(child Gene, PhysicalEntity, UtilityClass biopax elements will be auto-included 
648                                // during the export to OWL; we do not want to export dangling Genes, PEs, etc., except for Complexes...)
649                                selectedUris.addAll(findAllUris(searcher, Pathway.class, datasources, null));
650                                selectedUris.addAll(findAllUris(searcher, Interaction.class, datasources, null)); 
651                                selectedUris.addAll(findAllUris(searcher, Complex.class, datasources, null)); 
652                        }
653                
654                        uris = selectedUris.toArray(new String[] {});
655                }
656                
657                OutputStream os = new FileOutputStream(output);
658                // export a sub-model from the main biopax database
659                SimpleIOHandler sio = new SimpleIOHandler(BioPAXLevel.L3);
660                sio.absoluteUris(outputAbsoluteUris);
661                sio.convertToOWL(model, os, uris);
662        }

665        private static Class<? extends BioPAXElement> biopaxTypeFromSimpleName(String type) 
666        {       
667                // 'type' (a BioPAX L3 interface class name) is case insensitive 
668                for(Class<? extends BioPAXElement> c : SimpleEditorMap.L3
669                                .getKnownSubClassesOf(BioPAXElement.class)) 
670                {
671                        if(c.getSimpleName().equalsIgnoreCase(type)) {
672                                if(c.isInterface() && BioPAXLevel.L3.getDefaultFactory().getImplClass(c) != null)
673                                        return c; // interface
674                        }
675                }
676                throw new IllegalArgumentException("Illegal BioPAX class name '" + type);
677        }

751    /**
752     * Create cpath2 downloads 
753     * (exports the db to various formats)
754     * 
755     * @throws IOException, IllegalStateException (when not in maintenance mode), InterruptedException
756     */
757        public static void createDownloads() throws IOException, InterruptedException
758    {   
759                if(!cpath.isAdminEnabled())
760                        throw new IllegalStateException("Maintenance mode is not enabled.");
761                
762                // create downloads directory inside the home dir. if not exist
763                File f = new File(cpath.downloadsDir());
764                if(!f.exists()) 
765                        f.mkdir();
766                
767                ClassPathXmlApplicationContext context = 
768                        new ClassPathXmlApplicationContext(new String[] {
769                                        "classpath:META-INF/spring/applicationContext-jpa.xml"});
770                
771                MetadataRepository metadataRepository = (MetadataRepository) context.getBean(MetadataRepository.class);
772                
773                // 1) create an imported data summary file.txt (issue#23)
774                PrintWriter writer = new PrintWriter(cpath.downloadsDir() + File.separator 
775                                + "datasources.txt");
776                String date = new SimpleDateFormat("d MMM yyyy").format(Calendar.getInstance().getTime());
777                writer.println(StringUtils.join(Arrays.asList(
778                        "#CPATH2:", getInstance().getName(), "version", getInstance().getVersion(), date), " "));
779                writer.println("#Columns:\t" + StringUtils.join(Arrays.asList(
780                        "ID", "DESCRIPTION", "TYPE", "HOMEPAGE", "PATHWAYS", "INTERACTIONS", "PARTICIPANTS"), "\t"));           
781                Iterable<Metadata> allMetadata = metadataRepository.findAll();
782                for(Metadata m : allMetadata) {
783                        writer.println(StringUtils.join(Arrays.asList(
784                                m.getUri(), m.getDescription(), m.getType(), m.getUrlToHomepage(), 
785                                m.getNumPathways(), m.getNumInteractions(), m.getNumPhysicalEntities()), "\t"));
786                }               
787                writer.flush();
788                writer.close(); 
789                // destroy the Spring context, release some resources
790                context.close(); context = null;                                
791                LOG.info("create-downloads: successfully generated the datasources summary file.");
792                        
793                //load the main model
794                LOG.info("create-downloads: loading the Main BioPAX Model...");
795                Model mainModel = CPathUtils.loadMainBiopaxModel();
796                LOG.info("create-downloads: successfully read the Main BioPAX Model");
797                
798                // generate/find all other BioPAX archives (by organism, etc.):
799                List<String> otherBiopaxArchives = createOrFindBiopaxArchives(mainModel, allMetadata);
800        
801                // 2) create all other format archives from the main model (still in memory)
802                String prefix =  cpath.mainModelFile().substring(0, cpath.mainModelFile().indexOf("BIOPAX."));
803                exportBiopaxToOutputFormats(mainModel, prefix);
804                mainModel = null; //release quite a few Gygabytes
805                
806                // 3) export the rest of biopax archives to all other formats           
807        for(String biopaxArchive : otherBiopaxArchives) {       
808                InputStream biopaxStream = null;
809                try {
810                        biopaxStream = biopaxStream(biopaxArchive);
811                } catch (IOException e) {
812                        LOG.error("Failed to read " + biopaxArchive + "; skipped (wasn't created before?)", e );
813                        continue;
814                }                               
815                //read the given Model (can be vary large!) from archive
816                LOG.info("Loading the BioPAX model from " + biopaxArchive + "...");
817                Model m = (new SimpleIOHandler()).convertFromOWL(biopaxStream);         
818                prefix =  biopaxArchive.substring(0, biopaxArchive.indexOf("BIOPAX."));
819                exportBiopaxToOutputFormats(m, prefix);
820        }
821        }

884        private static Collection<String> findAllUris(Searcher searcher, 
885                Class<? extends BioPAXElement> type, String[] ds, String[] org) 
886    {
887        Collection<String> uris = new ArrayList<String>();
888        
889        SearchResponse resp = (SearchResponse) searcher.search("*", 0, type, ds, org);
890        int page = 0;
891        while(!resp.isEmpty()) {
892                for(SearchHit h : resp.getSearchHit())
893                        uris.add(h.getUri());
894                //next page
895                resp = (SearchResponse) searcher.search("*", ++page, type, ds, org);
896        }
897        
898        LOG.info("findAllUris(in "+type.getSimpleName()+", ds: "+Arrays.toString(ds)+", org: "+Arrays.toString(org)+") "
899                        + "collected " + uris.size() + " URIs (and the last hits page number was " + (page-1));
900        
901        return uris;
902    }

905        private static void convertBiopaxToOtherFormatGzipped( 
906                        final Model m, final OutputFormat format, 
907                        final String archiveName, final Object... params) 
908        {       
909                if(format == OutputFormat.BIOPAX)
910                        throw new IllegalArgumentException(format.name() + " is not allowed here.");
911
912                if(format == OutputFormat.GSEA && m.getObjects(Pathway.class).isEmpty()) {
913                        LOG.info("create-downloads: corresponding BioPAX model has no Pathways; "
914                                        + "so I will not create " + archiveName);
915                        return;
916                }
917                
918        if(!(new File(archiveName)).exists()) {
919                LOG.info("create-downloads: generating new " + archiveName);
920                //note: extended SIF will be one file (edges, nodes)
921                try {
922                        GZIPOutputStream zos = new GZIPOutputStream(new FileOutputStream(archiveName));
923                        convert(m, format, zos, params);
924                        zos.flush();
925                        IOUtils.closeQuietly(zos); 
926                        LOG.info("create-downloads: successully created " + archiveName);
927                } catch (IOException e) {
928                        throw new RuntimeException(e);
929                }               
930        } else {
931                LOG.info("create-downloads: skip for existing " + archiveName);
932        }
933        }

936        private static List<String> createOrFindBiopaxArchives(final Model mainModel, 
937                        Iterable<Metadata> allMetadata) throws InterruptedException 
938        {
939                final List<String> files = new ArrayList<String>();
940
941                // initialize the search engine
942                LOG.info("create-downloads: init the full-text search engine...");
943                final Searcher searcher = new SearchEngine(mainModel, CPathSettings.getInstance().indexDir());
944                
945                //will run concurrently
946                final ExecutorService exec = Executors.newFixedThreadPool(10);
947                
948        // export by organism (name)
949        if(cpath.getOrganisms().length>1) {
950                LOG.info("create-downloads: preparing data 'by organism' archives...");
951                for(final String org :  cpath.getOrganisms()) {
952                        // generate archives for current organism
953                        // hack: org.toLowerCase() is to tell by-organism from by-datasource archives (for usage stats...) 
954                        String archiveName = cpath.biopaxExportFileName(org.toLowerCase());
955                        if(!files.contains(archiveName)) {
956                                exportBiopax(exec, mainModel, searcher, archiveName, null, new String[]{org});
957                                files.add(archiveName); 
958                        }
959                }
960        } else {
961                LOG.info("create-downloads: skipped for 'by organism' archives (there is only one)");
962        }
963        
964                // export by datasource
965        LOG.info("create-downloads: creating 'by datasource' archives...");         
966        
967        //collect BioPAX pathway data source names in this set
968        Set<String> pathwayDataSources = new HashSet<String>();        
969        //generate by provider BioPAX archives
970        for(Metadata md : allMetadata) {
971                // generate archives for current pathway datasource;
972                if(!md.isNotPathwayData()) {
973                        // use standard name and not the metadata ID in the file name, 
974                        // because we want all data from same provider merge into one file, e.g.,
975                        // all Reactome (species) or IntAct with IntAct Complex, despite having them in separate metadata.
976                        String archiveName = cpath.biopaxExportFileName(md.standardName());
977                        //skip previously done files (this metadata has the same std. name as previously processed one)
978                        if(!files.contains(archiveName)) {
979                                        exportBiopax(exec, mainModel, searcher, archiveName, new String[]{md.standardName()}, null);
980                                        files.add(archiveName);
981                        }
982                        
983                        //collect if BioPAX data
984                        if(md.getType() == METADATA_TYPE.BIOPAX)
985                        pathwayDataSources.add(md.standardName());
986                }
987        }
988        
989        //a separate export - only BioPAX pathway data sources, "Detailed":
990        LOG.info("create-downloads: creating 'Detailed' data archives..."); 
991        final String archiveName = cpath.biopaxExportFileName("Detailed");
992                if(!files.contains(archiveName)) {
993                        exportBiopax(exec, mainModel, searcher,  archiveName, pathwayDataSources.toArray(new String[]{}), null);
994                        files.add(archiveName);
995                }
996        
997        exec.shutdown();
998        exec.awaitTermination(24, TimeUnit.HOURS);      
999        //continue in this thread after the above tasks ended or terminated
1000                
1001        return files;
1002        }

1005        private static void exportBiopax(ExecutorService exec, 
1006                        final Model mainModel, final Searcher searcher,
1007                        final String biopaxArchive, final String[] datasources, 
1008                        final String[] organisms)
1009        {
1010        // check file exists
1011        if(!(new File(biopaxArchive)).exists()) {
1012                LOG.info("create-downloads: creating new " +    biopaxArchive);
1013                
1014                        exec.execute(new Runnable() {                   
1015                                public void run() {
1016                                        try {
1017                                        //find all entities (all child elements will be then exported too)              
1018                                        Collection<String> uris = new HashSet<String>();
1019                                        uris.addAll(findAllUris(searcher, Pathway.class, datasources, organisms));      
1020                                        uris.addAll(findAllUris(searcher, Interaction.class, datasources, organisms));
1021                                        uris.addAll(findAllUris(searcher, Complex.class, datasources, organisms));
1022                                        
1023                                        // export objects found above to a new biopax archive           
1024                                        if(!uris.isEmpty()) {
1025                                                OutputStream os = new GZIPOutputStream(new FileOutputStream(biopaxArchive));
1026                                                SimpleIOHandler sio = new SimpleIOHandler(BioPAXLevel.L3);
1027//                                              sio.absoluteUris(true);
1028                                                sio.convertToOWL(mainModel, os, uris.toArray(new String[]{}));  
1029                                                LOG.info("create-downloads: successfully created " +    biopaxArchive);
1030                                        } else {
1031                                                LOG.info("create-downloads: no pathways/interactions found; skipping " +        biopaxArchive);
1032                                        }
1033                                        } catch (IOException e) {
1034                                                throw new RuntimeException(e);
1035                                        }                                               
1036                                }
1037                        });             
1038        } else {
1039                LOG.info("create-downloads: found previously generated " + biopaxArchive);
1040        }               
1041        }

1044        private static String formatAndExt(OutputFormat format, String suffix) {
1045                
1046                suffix = (suffix==null || suffix.isEmpty())
1047                                ? "" : "." + suffix.toLowerCase();
1048                
1049                switch (format) {
1050                case GSEA:
1051                        return format + suffix + ".gmt.gz";
1052                case SBGN:
1053                        return format + suffix + ".xml.gz";
1054                case BINARY_SIF:
1055                        return format + suffix + ".sif.gz";
1056                case EXTENDED_BINARY_SIF:
1057                        return format + suffix + ".sif.gz";
1058                default://fail - biopax is treated specially, not here
1059                        throw new IllegalArgumentException(format.toString() + " not allowed.");
1060                }
1061        }

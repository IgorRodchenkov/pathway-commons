001package cpath.importer;
002
003import java.io.InputStream;
004import java.io.OutputStream;
005
006/**
007 * Converter interface,  
008 * to convert other format data to BioPAX.
009 */
010public interface Converter {
011        
012        /**
013         * Sets the xml:base (URI namespace) for
014         * the resulting BioPAX model.
015         * @param xmlBase
016         */
017        void setXmlBase(String xmlBase);
018        
019        /**
020         * Builds and saves a new BioPAX model from the input data.
021         * @param is
022         * @param os  - result; the stream must be closed inside this method.
023         */
024        void convert(InputStream is, OutputStream os);
025}

001package cpath.importer;
002
003import java.io.InputStream;
004import java.io.OutputStream;
005
006/**
007 * Cleaner interface.
008 * 
009 * Can be implemented and used for a particular
010 * biological data source, when some filtering 
011 * and fixing are required, not waiting for the next
012 * provider's official data release.
013 *
014 */
015public interface Cleaner {
016
017        /**
018         * To clean, filter, fix the data.
019         *
020         * @param data
021         * @param cleanedData - result; the stream must be closed inside this method.
022         */
023        void clean(InputStream data, OutputStream cleanedData);
024}

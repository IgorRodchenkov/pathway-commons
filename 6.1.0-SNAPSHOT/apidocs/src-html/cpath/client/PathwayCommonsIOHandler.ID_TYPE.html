109    private InputStream openURL(String urlName) throws IOException
110        {
111                        URL url = new URL(urlName);
112                        URLConnection urlConnection = url.openConnection();
113                        return urlConnection.getInputStream();
114        }
115
116        private Model readBioPAXfromURL(String urlName) throws IOException
117        {
118                return ioHandler.convertFromOWL(openURL(urlName));
119        }

121    /**
122     * Retrieves the records of an entity specified by its id, and returns
123     * the results as a model.
124     *
125     * Pathway commons definition:
126     * Retrieves details regarding one or more records, such as a pathway,
127     * interaction or physical entity. For example, get the complete
128     * Apoptosis pathway from Reactome.
129     *
130     * @param id    entity id
131     * @throws IOException if connection fails for some reason.
132     * @return      empty model in case of errors.
133     *
134     * @see #setPrefix(String)
135     */
136    public Model retrieveByID(String id) throws IOException
137        {
138                String urlName = prefix+command+"get_record_by_cpath_id"+
139                        "&"+
140                        version+versionID +
141                        "&"+
142                        "q="+id +
143                        "&"+
144                        output+ output_format;
145                return readBioPAXfromURL(urlName);
146        }
147
148    /**
149     * Get pathways to which the entity, specified by its id, is related
150     * and returns a two-dimensional ArrayList containing all results as rows
151     * and their subproperties as columns. First row contains the column types
152     * (Database:ID, Pathway_Name, Pathway_Database_Name, CPATH_ID).
153     *
154     * Pathway Commons definition:
155     * Retrieves all pathways involving a specified physical entity
156     * (e.g. protein or small molecule).
157     *
158     * For example, get all pathways involving BRCA2.
159     *
160     * @param id    entity id
161     * @return      An ArrayList containing "PHYSICAL_ENTITY_ID_NOT_FOUND" as the second column in case of no results
162     *
163     * @see #setDataSource(cpath.client.PathwayCommonsIOHandler.DATA_SRC)
164     * @see #setInputIdType(cpath.client.PathwayCommonsIOHandler.ID_TYPE)
165     * @see #setOutputIdType(cpath.client.PathwayCommonsIOHandler.ID_TYPE)
166     * @see #setPrefix(String)
167     *
168     */
169    public List<List<String>> getPathways(String id) throws IOException
170    {
171        /* "get_pathway_list" command will probably be replaced with "get_pathways"
172         * in the upcoming versions of the webservice. Please, see the comment below.
173         *
174         * String urlName = prefix+command+"get_pathways"+
175         */
176        String urlName = prefix+command+"get_pathways"+
177                        "&"+
178                        version+versionID +
179                        "&"+
180                        "q="+id +
181                        "&"+
182                        output+ output_format +
183                        "&"+
184            input_id_type + inputIdType.getTag() +
185                        "&"+
186            data_source + dataSource.getTag();
187
188        BufferedReader reader = new BufferedReader( new InputStreamReader(openURL(urlName)) );
189        String oneLine;
190        List<List<String>> allLines = new ArrayList<List<String>>();
191
192        while((oneLine = reader.readLine()) != null ) {
193            List<String> lineArray = new ArrayList<String>();
194            StringTokenizer tokenizer = new StringTokenizer(oneLine, "\t");
195            while(tokenizer.hasMoreTokens())
196            {
197                lineArray.add(tokenizer.nextToken());
198            }
199
200                        if (!lineArray.isEmpty())
201                        {
202                                allLines.add(lineArray);
203                        }
204        }
205
206        return allLines;
207    }

209    /**
210     * Gets neighbors of an entity specified by an id in biopax format
211     * and returns the results as a model.
212     *
213     * Pathway Commons definition:
214     * Retrieves the nearest neighbors of a given physical entity
215     * (e.g. gene, protein or small molecule).
216     *
217     * For example, get all the neighbors of BRCA2.
218     *
219     * @param id    entity id
220     * @return      empty model in case of errors.
221     *
222     * @see #setDataSource(cpath.client.PathwayCommonsIOHandler.DATA_SRC)
223     * @see #setInputIdType(cpath.client.PathwayCommonsIOHandler.ID_TYPE)
224     * @see #setOutputIdType(cpath.client.PathwayCommonsIOHandler.ID_TYPE)
225     * @see #setPrefix(String)
226     */
227    public Model getNeighbors(String id) throws IOException
228    {
229        String urlName = prefix+command+"get_neighbors"+
230                        "&"+
231                        version+versionID +
232                        "&"+
233                        "q="+id +
234                        "&"+
235                        output+ output_format +
236                        "&"+
237            input_id_type + inputIdType.getTag() +
238                        "&"+
239            output_id_type + outputIdType.getTag() +
240                        "&"+
241            data_source + dataSource.getTag();
242
243        return readBioPAXfromURL(urlName);
244        }

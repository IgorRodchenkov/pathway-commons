065    /**
066     * Instantiates the client using the cpath2 endpoint URL.
067     * 
068     * @param url cpath2 web service endpoint URL or null (to use defaults)
069     */
070    public static CPathClient newInstance(String url) {
071        CPathClient client = new CPathClient(); 
072        
073        // add custom cPath2 XML message converter as the first one (accepts 'application/xml' content type)
074        // because one of existing/default msg converters, XML root element based jaxb2, does not work for ServiceResponce types...
075        client.restTemplate.getMessageConverters().add(0, new ServiceResponseHttpMessageConverter());
076        // add BioPAX http message converter
077        client.restTemplate.getMessageConverters().add(1, new BioPAXHttpMessageConverter(new SimpleIOHandler()));
078        
079        // set the cpath2 server URL (or default one or from the java option)
080        if(url == null || url.isEmpty())
081                url = System.getProperty(JVM_PROPERTY_ENDPOINT_URL, DEFAULT_ENDPOINT_URL);  
082        
083        client.setEndPointUrlAndRedirect(url);
084
085        return client;
086    }

089        /**
090         * Sends a HTTP POST (preferred, more reliable with complex queries) 
091         * request to the server.
092         * 
093         * @param requestPath cpath2 web service command path (e.g., search, help/types, etc.)
094         * @param requestParams query parameters object.
095         * @param responseType result class (e.g., String.class, Model)
096         * @return
097         * @throws CPathException
098         */
099        public <T> T post(String requestPath, MultiValueMap<String, String> requestParams, Class<T> responseType)
100                throws CPathException 
101        {
102                final String url = actualEndPointURL + requestPath;
103                
104                if(name != null && requestParams != null)
105                        requestParams.put("client", Collections.singletonList(name));
106                
107                try {
108                        return restTemplate.postForObject(url, requestParams, responseType);
109                } catch (UnknownHttpStatusCodeException e) {
110                        if (e.getRawStatusCode() == 460) {
111                                return null; //empty result
112                        } else
113                                throw new CPathException(url + " and " + requestParams, e);
114                } catch (RestClientException e) {
115                        throw new CPathException(url + " and " + requestParams, e);
116                }
117        }
118    
119        
120        /**
121         * Sends a HTTP GET request to the cpath2 server.
122         * 
123         * Note: using {@link #post(String, Object, Class)} is the preferred 
124         * and more reliable method, especially with complex queries that use URIs or
125         * Lucene syntax.
126         * 
127         * @param requestPath cpath2 web service command path (e.g., search, help/types, etc.)
128         * @param requestParams query parameters map
129         * @param responseType result class (e.g., String.class, Model)
130         * @return
131         * @throws CPathException
132         */
133        public <T> T get(String requestPath, MultiValueMap<String, String> requestParams, Class<T> responseType)
134                throws CPathException 
135        {       
136                StringBuilder sb = new StringBuilder(actualEndPointURL);
137                sb.append(requestPath);
138                
139                if(requestParams != null) {
140                        sb.append("?");
141                        for (Entry<String, List<String>> entry : requestParams.entrySet()) {
142                                String params = join(entry.getKey() + "=", entry.getValue(), "&");
143                                sb.append(params).append("&");
144                        }
145                        if(name!=null)
146                                sb.append("client=").append(name);
147                }
148                
149                String url = sb.toString();
150                
151                try {
152                        return restTemplate.getForObject(url, responseType);
153                } catch (UnknownHttpStatusCodeException e) {
154                        if (e.getRawStatusCode() == 460) {
155                                return null; //empty result
156                        } else
157                                throw new CPathException(url, e);
158                } catch (RestClientException e) {
159                        throw new CPathException(url, e);
160                }
161        }

164    /**
165     * Retrieves information about available cPath2 commands and their parameters.
166     * 
167     * @param hpath relative (ie., as in 'help/[path]') REST query path variable; e.g.: null (all), "datasources", "commands/search", etc.
168     * 
169     * @return
170     * @throws CPathException 
171     */
172    public Help executeHelp(String hpath) throws CPathException {
173        return get("help/" + ((hpath != null) ? hpath : ""), null, Help.class);
174    }

177    /**
178     * Joins the collection of strings into one string 
179     * using the prefix and delimiter.
180     * 
181     * @param prefix 
182     * @param strings
183     * @param delimiter
184     * @return
185     */
186    private String join(String prefix, Collection<String> strings, String delimiter) {
187        List<String> prefixed = new ArrayList<String>();
188
189        for(String s: strings)
190                prefixed.add(prefix + s);
191
192        return StringUtils.join(prefixed, delimiter);
193    }

215    }

218    /**
219     * Sets the web service URL and then
220     * finds actual resource location (after HTTP redirects)
221     * for sending future HTTP requests.
222     * 
223     * @see #getEndPointURL()
224     * @see #getActualEndPointURL()
225     * @param endPointURL a cPath2 web service URL
226     */
227    public void setEndPointUrlAndRedirect(final String url) {
228        endPointURL = url;
229        actualEndPointURL = endPointURL; //initially the same
230        // discover actual resource location from the initial one (also, trying to avoid going in circles):
231        int i=0;
232        for(URI loc = URI.create(endPointURL); loc != null && i<5; i++ ) 
233        {                               
234                loc = restTemplate.postForLocation(loc, null);
235                
236                if(loc != null)
237                        actualEndPointURL = loc.toString();
238                
239                        LOGGER.info("Location: " + loc);
240        }
241    }

004package cpath.query;
005
006import java.util.Arrays;
007import java.util.Collection;
008
009import org.apache.commons.lang.ArrayUtils;
010import org.biopax.paxtools.controller.ModelUtils;
011import org.biopax.paxtools.model.Model;
012import org.springframework.util.LinkedMultiValueMap;
013import org.springframework.util.MultiValueMap;
014
015import cpath.client.CPathClient;
016import cpath.client.util.CPathException;
017import cpath.service.Cmd;
018import cpath.service.CmdArgs;
019import cpath.service.OutputFormat;
020
021/**
022 * A get by ID/URI query to be executed with {@link CPathClient}
023 * 
024 * @author rodche
025 */
026public final class CPathGetQuery extends BaseCPathQuery<Model> implements CPathQuery<Model> {
027        
028        private boolean mergeEquivalentInteractions = false;
029        private String[] source;
030        
031        
032        /**
033         * @return the request
034         */
035        protected MultiValueMap<String, String> getRequestParams() {
036                MultiValueMap<String, String> request = new LinkedMultiValueMap<String, String>();
037                if(source == null || source.length == 0)
038                        throw new IllegalArgumentException("Required list of URIs/IDs " +
039                                        "cannot be null or empty");
040                request.put(CmdArgs.uri.name(), Arrays.asList(source));
041                return request;
042        }
043
044        /**
045         * Constructor.
046         * @param client cpath2 client instance
047         */
048        public CPathGetQuery(CPathClient client) {
049                this.client = client;
050        }
051
052        /**
053         * A list of URIs (of biopax elements) 
054         * or IDs (e.g., gene symbols).
055         * 
056         * @param sources
057         * @return
058         */
059        public CPathGetQuery sources(String[] sources) {
060                this.source = sources;
061                return this;
062        }
063        
064        /**
065         * A collection of URIs (of biopax elements) 
066         * or IDs (e.g., gene symbols).
067         * 
068         * @param sources
069         * @return
070         */
071        public CPathGetQuery sources(Collection<String> sources) {
072                this.source = sources.toArray(ArrayUtils.EMPTY_STRING_ARRAY);
073                return this;
074        }       
075
076        
077        /**
078         * Sets the option to merge equivalent interactions in the result model.
079         * @param mergeEquivalentInteractions
080         */
081        public void mergeEquivalentInteractions(boolean mergeEquivalentInteractions)
082        {
083                this.mergeEquivalentInteractions = mergeEquivalentInteractions;
084        }
085
086        @Override
087        public String stringResult(OutputFormat format) throws CPathException {         
088                MultiValueMap<String, String> request = getRequestParams();               
089                if (format == null)
090                        format = OutputFormat.BIOPAX;
091                request.add(CmdArgs.format.name(), format.name());
092                
093                return client.post(Cmd.GET.toString(), request, String.class);
094        }
095
096        @Override
097        public Model result() throws CPathException {
098                Model model = client.post(Cmd.GET.toString(), getRequestParams(), Model.class);
099                
100                if (mergeEquivalentInteractions && model != null)
101                {
102                        ModelUtils.mergeEquivalentInteractions(model);
103                }
104                
105                return model;
106        }
107
108}

001package cpath.query;
002
003import java.util.Arrays;
004import java.util.Collection;
005
006import org.apache.commons.lang.ArrayUtils;
007import org.biopax.paxtools.controller.PathAccessor;
008import org.springframework.util.LinkedMultiValueMap;
009import org.springframework.util.MultiValueMap;
010
011import cpath.client.CPathClient;
012import cpath.client.util.CPathException;
013import cpath.service.Cmd;
014import cpath.service.CmdArgs;
015import cpath.service.OutputFormat;
016import cpath.service.jaxb.TraverseResponse;
017
018/**
019 * A property 'traverse' query 
020 * to be executed with a {@link CPathClient}
021 * on the cpath2 server's biopax db.
022 * 
023 * @author rodche
024 */
025public final class CPathTraverseQuery extends BaseCPathQuery<TraverseResponse> implements CPathQuery<TraverseResponse> {
026
027        private String path;
028        private String[] source;
029
030        /**
031         * @return the request
032         */
033        protected MultiValueMap<String, String> getRequestParams() {
034                MultiValueMap<String, String> request = new LinkedMultiValueMap<String, String>();
035                if(source == null || source.length == 0)
036                        throw new IllegalArgumentException("Required 'source' (URIs) parameter cannot be null or empty.");
037                request.put(CmdArgs.uri.name(), Arrays.asList(source));
038                if(path == null || path.isEmpty())
039                        throw new IllegalArgumentException("Property 'path' is null or empty.");
040                request.add(CmdArgs.path.name(), path);
041                return request;
042        }
043
044        /**
045         * Constructor.
046         * @param client instance of the cpath2 client
047         */
048        public CPathTraverseQuery(CPathClient client) {
049                this.client = client;
050        }
051
052        /**
053         * URIs of biopax elements to start with
054         * (i.e., to apply the path to each one and get property values).
055         * @param source
056         * @return
057         */
058        public CPathTraverseQuery sources(String[] source) {
059                this.source = source;
060                return this;
061        }
062        
063        /**
064         * URIs of biopax elements to start with
065         * (i.e., to apply the path to each one and get property values).
066         * @param source
067         * @return
068         */
069        public CPathTraverseQuery sources(Collection<String> sources) {
070                this.source = sources.toArray(ArrayUtils.EMPTY_STRING_ARRAY);
071                return this;
072        }       
073
074        /**
075         * A biopax properties path to follow
076         * in order to get values.
077         * 
078         * @see PathAccessor
079         * 
080         * @param path
081         * @return
082         */
083        public CPathTraverseQuery propertyPath(String path) {
084                this.path = path;
085                return this;
086        }
087
088        @Override
089        public String stringResult(OutputFormat outputFormat) throws CPathException {
090                throw new UnsupportedOperationException();
091        }
092        
093        @Override
094        public TraverseResponse result() throws CPathException {
095                return client.post(Cmd.TRAVERSE.toString(), getRequestParams(), TraverseResponse.class);
096        }
097}

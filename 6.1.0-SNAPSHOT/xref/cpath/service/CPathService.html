39  import cpath.jpa.Content;
40  import cpath.jpa.LogEntitiesRepository;
41  import cpath.jpa.LogEntity;
42  import cpath.jpa.LogEvent;
43  import cpath.jpa.Mapping;
44  import cpath.jpa.MappingsRepository;
45  import cpath.jpa.Metadata;
46  import cpath.jpa.MetadataRepository;

50  /**
51   * CPath^2 Service is an adapter between DAO and web controllers. 
52   * Can be used in a console application or integration tests 
53   * (web container is not required.)
54   * 
55   * This interface defines several middle-tier data access and analysis methods 
56   * that accept valid parameters, handle exceptions, and return results packed 
57   * in a ServiceResponse bean.
58   * 
59   * @author rodche
60   *
61   */
62  public interface CPathService {
63  
64  	/**
65  	 * Retrieves the BioPAX element(s) by URI or identifier (e.g., gene symbol)
66  	 * - a complete BioPAX sub-model with all available child elements and 
67  	 * properties - and then converts it to the specified output format 
68  	 * (if applicable), such as BioPAX (RDF/XML), SIF, GSEA (.gmt).
69  	 * 
70  	 * @param format
71  	 * @param uris the list of URIs to fetch
72  	 * @return
73  	 */
74  	ServiceResponse fetch(OutputFormat format, String... uris);
75  
76  	
77  	/**
78  	 * Full-text search for the BioPAX elements. 
79  	 * 
80  	 * @param queryStr
81  	 * @param page search results page no.
82  	 * @param biopaxClass
83  	 * @param dsources URIs of data sources
84  	 * @param organisms URIs of organisms
85  	 * @return
86  	 * 
87  	 * @see PaxtoolsDAO#search(String, Class[], String[], String[])
88  	 */
89  	ServiceResponse search(String queryStr, 
90  			int page, Class<? extends BioPAXElement> biopaxClass, String[] dsources, String[] organisms);
91  
92  	
93  	/**
94  	 * Runs a neighborhood query using the given parameters
95  	 * (returns a sub-model in the specified format, 
96  	 * wrapped as service object). 
97  	 *
98  	 * @param format output format
99  	 * @param sources IDs of seed of neighborhood
100 	 * @param limit search limit (integer value)
101 	 * @param direction flag 
102 	 * @param organisms optional filter
103 	 * @param datasources optional filter
104 	 * @return the neighborhood
105 	 */
106 	ServiceResponse getNeighborhood(OutputFormat format, 
107 		String[] sources, Integer limit, Direction direction,
108 			String[] organisms, String[] datasources);
109 
110 	/**
111 	 * Runs a paths-between query for the given sources	
112 	 * (returns a sub-model in the specified format, 
113 	 * wrapped as service object). 
114 	 *
115 	 * @param format output format
116 	 * @param sources IDs of source molecules
117 	 * @param limit search limit (integer value)
118 	 * @param organisms optional filter
119 	 * @param datasources optional filter
120 	 * @return paths between
121 	 */
122 	ServiceResponse getPathsBetween(OutputFormat format, String[] sources, 
123 		Integer limit, String[] organisms, String[] datasources);
124 
125 	/**
126 	 * Runs a POI query from the given sources to the given targets
127 	 * (returns a sub-model in the specified format, wrapped as service object). 
128 	 *
129 	 * @param format output format
130 	 * @param sources IDs of source molecules
131 	 * @param targets IDs of target molecules
132 	 * @param limit search limit (integer value)
133 	 * @param organisms optional filter
134 	 * @param datasources optional filter
135 	 * @return paths between
136 	 */
137 	ServiceResponse getPathsFromTo(OutputFormat format, String[] sources,
138 		String[] targets, Integer limit, String[] organisms, String[] datasources);
139 	
140 	
141 	/**
142 	 * Runs a common upstream or downstream query
143 	 * (returns a sub-model in the specified format, 
144 	 * wrapped as service object). 
145 	 *
146 	 * @param format output format
147 	 * @param sources IDs of query seed
148 	 * @param limit search limit
149 	 * @param direction - can be {@link Direction#DOWNSTREAM} or {@link Direction#UPSTREAM}
150 	 * @param organisms optional filter
151 	 * @param datasources optional filter
152 	 * @return common stream
153 	 */
154 	ServiceResponse getCommonStream(OutputFormat format, 
155 		String[] sources, Integer limit, Direction direction, 
156 			String[] organisms, String[] datasources);
157 
158 	//---------------------------------------------------------------------------------------------|
159 
160 	/**
161 	 * Collects BioPAX property values at the end of the property path
162 	 * applied to each BioPAX object in the list (defined by URIs), 
163 	 * where applicable.
164 	 *  
165 	 * @see PathAccessor
166 	 * 
167 	 * @param propertyPath
168 	 * @param sourceUris
169 	 * @return
170 	 */
171 	ServiceResponse traverse(String propertyPath, String... sourceUris);
172 	
173 	
174 	/**
175 	 * Gets top (root) pathways (URIs, names) in the current BioPAX model.
176 	 * 
177 	 * @param organisms filter values (URIs, names, or taxonomy IDs)
178 	 * @param datasources filter values (URIs, names)
179 	 * @return
180 	 */
181 	ServiceResponse topPathways(String[] organisms, String[] datasources);
182 	
183 	
184 	/**
185 	 * Guess the identifier type (chemical vs gene/protein) 
186 	 * and returns other Is it maps to. This method can be wrong
187 	 * about mapping, it's weak, designed to use primarily in
188 	 * BioPAX graph queries and not for data integration/merging.
189 	 * 
190 	 * @param identifier
191 	 * @return
192 	 */
193 	Set<String> map(String identifier);
194 	
195 	
196 	/**
197      * Maps an identifier to primary ID(s) of a given type.
198      * 
199      * Normally, the result set contains only one ID.
200      * If the result contains more than one value, which does not
201      * necessarily an error, then it's up to other methods to decide 
202      * how to proceed; e.g., one should not probably merge different 
203      * data objects if the mapping is known to be umbiguous,
204      * but it's usually ok to generate relationship xrefs 
205      * or use the resulting IDs in a BioPAX graph query.
206      * 
207      * @param fromDb data collection name or null (to use all source ID types)
208      * @param fromId the source ID
209      * @param toDb standard (MIRIAM) preferred name of the target ID type (e.g., 'UniProt')
210      * 
211      * @return a set of primary IDs of the type; normally one or none elements
212      */
213     Set<String> map(String fromDb, String fromId, String toDb);
214         
215     /**
216      * Test if this or equivalent record exists
217      * in the id-mapping db and if not, saves the new one.
218      * 
219      * @param mapping
220      * @return
221      */
222     void saveIfUnique(Mapping mapping);
223 
224 	/**
225 	 * Saves and counts a series of data access events 
226 	 * (usually associated with the same web request) 
227 	 * to the log db.
228 	 * 
229 	 * @param events
230 	 * @param ipAddr
231 	 */
232 	void log(Collection<LogEvent> events, String ipAddr);
233     
234 	/**
235 	 * Increases the number (counter) 
236 	 * of today user's requests of some 
237 	 * sort and location.
238 	 * 
239 	 * Right now, only country code there
240 	 * matters for location matching 
241 	 * (city and region are ignored).
242 	 * 
243 	 * @param date
244 	 * @param event
245 	 * @param ipAddr
246 	 * @return
247 	 */
248 	LogEntity count(String date, LogEvent event, String ipAddr);	
249 	
250     /**
251      * Persists or updates the given metadata object.
252      *
253      * @param metadata Metadata
254      * @return
255      */
256 	Metadata save(Metadata metadata);
257 	
258 	/**
259 	 * Deletes the provider Metadata and data sub-directory.
260 	 * 
261 	 * @param metadata
262 	 */
263 	void delete(Metadata metadata);
264 	
265 	/**
266 	 * Imports Metadata from a config file.
267 	 * 
268 	 * @param location
269 	 */
270 	void addOrUpdateMetadata(String location);
271 	
272     /**
273 	 * Removes from the system all entries and 
274 	 * previously converted / premerged / validated 
275 	 * files accociated with this data provider.
276 	 * 
277 	 * @param metadata
278 	 * @return updated/saved object
279 	 */
280 	Metadata init(Metadata metadata);		
281     
282 	/**
283 	 * Generates the BioPAX validation report for a pathway data file.
284 	 * 
285 	 * @param provider data source (Metadata) identifier, not null
286 	 * @param file - base filename as in {@link Content}, or null (for all files)
287 	 * @return
288 	 */
289 	ValidatorResponse validationReport(String provider, String file);
290 	
291 	
292 	//spring-data-jpa repositories
293 	
294 	MappingsRepository mapping();
295     
296     MetadataRepository metadata();
297     
298     LogEntitiesRepository log();
299 
300     /**
301      * Loads or re-loads the main BioPAX Model 
302      * and blacklist from archive.
303      */
304 	void init();
305 	
306 	/**
307 	 * Checks whether the service object is fully initialized, i.e,
308 	 * if all data repositories, in-memory BioPAX model and search
309 	 * engine are ready to go.
310 	 * 
311 	 * @return true if it's fully initialized
312 	 */
313 	boolean ready();
314 }

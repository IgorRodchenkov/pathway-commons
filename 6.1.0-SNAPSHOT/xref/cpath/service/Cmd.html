28  package cpath.service;
29  
30  import static cpath.service.CmdArgs.*;
31  
32  
33  /**
34   * cPath2 web service commands.
35   * 
36   * @author rodche
37   *
38   */
39  public enum Cmd {
40  	SEARCH("Full-text search for BioPAX objects. " +
41  			"It accepts up to five parameters " +
42  			"and returns the ordered list of search hits, " +
43  			"which are simplified description of BioPAX elements " +
44  			"matching the query and passing all filters. A hit's uri (same as the corresponding BioPAX " +
45  			"object's RDF ID) can be used with other webservice commands to " +
46  			"extract the corresponding sub-model to BioPAX or another supported format. ",
47  			"/search?q=brca*&organism=9606", //full PROVIDER_URL shouldn't be specified here (it depends on the server configuration)!
48  			"Search Response that lists Search Hits - XML (default) or JSON (when called as '/search.json?')",
49  			new CmdArgs[]{q, page, type, organism, datasource}),	
50  	GET("Gets a BioPAX element or sub-model " +
51          "by ID(s).  This command has two parameters.",
52          "/get?uri=http://identifiers.org/uniprot/P38398",
53          "BioPAX by default, other formats as specified by the format parameter.  " +
54          "See the <a href=\"#valid_output_parameter\">valid values for format parameter</a> below.",
55          new CmdArgs[]{uri, format}),
56  	GRAPH("Executes an advanced graph query on the data within pathway commons. " +
57            "Returns a sub-model as the result. This command has up to six parameters.",
58            "/graph?kind=neighborhood&source=URI1&source=URI2&...",
59            "BioPAX by default, other formats as specified by the format parameter. " +
60            "See the <a href=\"#valid_output_parameter\">valid values for format parameter</a> below.",
61            new CmdArgs[]{kind, source, target, format, limit, direction, organism, datasource}),
62      TOP_PATHWAYS("Gets Top Pathways. This command accepts optional filter by organism and by datasource values", 
63      	"/top_pathways",
64          "Search Response - XML (JSON, when called as '/top_pathways.json?') contains the list of all top pathways.", 
65          new CmdArgs[]{organism, datasource}),   
66      TRAVERSE("Gets data property values (or elements's URIs) " +
67      	"at the end of the property path.  This command has two parameters.",
68      	"/traverse?uri=http://identifiers.org/uniprot/P38398&path=ProteinReference/organism/displayName",
69      	"Traverse Response - XML (or JSON, when called as '/traverse.json?').", 
70      	new CmdArgs[]{path, uri})
71            
72      ;
73  	
74  	private final CmdArgs[] args; //Array is better for use in json/jsp than List/Set
75  	private final String info;
76      private final String example;
77      private final String output;
78  	
79  	public CmdArgs[] getArgs() {
80  		return args;
81  	}
82  	
83  	public String getInfo() {
84  		return info;
85  	}
86  
87      public String getExample() {
88          return example;
89      }
90      
91      public String getOutput() {
92          return output;
93      }
94  	
95  	private Cmd(String info, String example, String output, CmdArgs... args) {
96  		this.info = info;
97          this.example = example;
98          this.output = output;
99  		this.args = args;
100 	}
101 
102 	@Override
103 	public String toString() {
104 		return name().toLowerCase();
105 	}
106 }

75      /**
76       * Empties the directory.
77       * 
78       * @param path
79       * @return
80       */
81      public static void cleanupDirectory(File path) {
82          if( path.exists() ) {
83            File[] files = path.listFiles();
84            for(int i=0; i<files.length; i++) {
85               if(files[i].isDirectory()) {
86              	 cleanupDirectory(files[i]);
87               }
88               else {
89                 files[i].delete();
90               }
91            }
92          }
93      }

96      /**
97       *  For the given url, returns a collection of Metadata Objects.
98       *
99       * @param url String
100      * @return Collection<Metadata>
101      */
102     public static Collection<Metadata> readMetadata(final String url)
103     {
104         // order of lines/records in the Metadata table does matter (since 2013/03);
105 		// so List is used here instead of HashSet
106 		List<Metadata> toReturn = new ArrayList<Metadata>();
107 
108         // check args
109         if (url == null) {
110             throw new IllegalArgumentException("url must not be null");
111         }
112 
113         // get data from service
114         BufferedReader reader = null;
115         try {
116             // we'd like to read lines at a time
117             reader = new BufferedReader(new InputStreamReader(
118             	LOADER.getResource(url).getInputStream(), "UTF-8"));
119 
120             // are we ready to read?
121             while (reader.ready()) 
122             {
123                 // grab a line
124                 String line = reader.readLine();
125                 if("".equals(line.trim()))
126                 	continue;
127                 else if(line.trim().startsWith("#")) {
128 					LOGGER.info("readMetadata(), ignored line: " + line);
129                 	continue; //ignore/skip parsing
130                 }
131                 	
132                 /* for now, assume line is delimited into 9 columns by '\t' (tab);
133                  * empty strings in the middle (the result of using \t\t) and 
134                  * trailing empty string after the last tabulation (i.e., Converter 
135                  * class name, if any), will be added to the tokens array as well.
136                  */
137                 String[] tokens = line.split("\t",-1);
138                 
139 				if (LOGGER.isDebugEnabled()) {
140 					LOGGER.debug("readMetadata(), token size: " + tokens.length);
141 					for (String token : tokens) {
142 						LOGGER.debug("readMetadata(), token: " + token);
143 					}
144 				}
145 
146                 assert tokens.length == NUMBER_METADATA_ITEMS : "readMetadata(): " +
147                 		"wrong number of columns, " + tokens.length + " instead of "
148                 		+ NUMBER_METADATA_ITEMS + ", in the metadata record: " + line;
149 
150 				// get metadata type
151 				Metadata.METADATA_TYPE metadataType = Metadata.METADATA_TYPE.valueOf(tokens[METADATA_TYPE_INDEX]);
152 				
153 				LOGGER.debug("readMetadata(): make a Metadata bean.");
154 
155                 // create a metadata bean
156                 Metadata metadata = new Metadata(
157                 		tokens[METADATA_IDENTIFIER_INDEX], 
158                 		tokens[METADATA_NAME_INDEX],
159                 		tokens[METADATA_DESCRIPTION_INDEX], 
160                 		tokens[METADATA_DATA_URL_INDEX],
161                         tokens[METADATA_HOMEPAGE_URL_INDEX], 
162                         tokens[METADATA_ICON_URL_INDEX],
163                         metadataType,
164 						tokens[METADATA_CLEANER_CLASS_NAME_INDEX],
165 						tokens[METADATA_CONVERTER_CLASS_NAME_INDEX],
166 						tokens[METADATA_PUBMEDID_INDEX],		
167                 		tokens[METADATA_AVAILABILITY_INDEX]);
168                 
169 				if (LOGGER.isInfoEnabled()) {
170 					LOGGER.info("readMetadata(): adding Metadata: "
171 					+ "identifier=" + metadata.getIdentifier() 
172 					+ "; name=" + metadata.getName()
173 					+ "; date/comment=" + metadata.getDescription()
174 					+ "; location=" + metadata.getUrlToData()
175 					+ "; icon=" + metadata.getIconUrl()
176 					+ "; type=" + metadata.getType()
177 					+ "; cleaner=" + metadata.getCleanerClassname() 
178 					+ "; converter=" + metadata.getConverterClassname()
179 					+ "; pubmedId=" + metadata.getPubmedId() 
180 					+ "; availability=" + metadata.getAvailability()
181 					);
182 				}
183 					
184 				// add metadata object toc collection we return
185 				toReturn.add(metadata);
186             } 
187         } catch (java.io.UnsupportedEncodingException e) {
188         } catch (IOException e) {
189             throw new RuntimeException(e);
190         } finally {
191             closeQuietly(reader);
192         }
193 
194         return toReturn;
195     }

198     /**
199      * For the given Metadata, unpacks and reads the corresponding 
200      * original zip data archive, creating new {@link Content} objects 
201      * in the metadata's dataFile collection.
202      * Skips for system files/directory entries.
203      *
204      * @see Metadata#getDataArchiveName()
205 	 * @param metadata Metadata
206      * @throws RuntimeException if an IO error occurs
207      */
208     public static void analyzeAndOrganizeContent(final Metadata metadata) 
209     {
210 		Collection<Content> contentCollection = new HashSet<Content>();
211 		ZipInputStream zis = null;
212 		try {
213 			zis = new ZipInputStream((metadata.getUrlToData()
214 				.startsWith("classpath:")) //a hack for easy junit tests
215 					? LOADER.getResource(metadata.getUrlToData()).getInputStream() 
216 						: new FileInputStream(metadata.getDataArchiveName()));		
217 			// interate over zip entries
218 			ZipEntry entry = null;
219             while ((entry = zis.getNextEntry()) != null) 
220             {
221             	String entryName = entry.getName();
222            		LOGGER.info("analyzeAndOrganizeContent(), processing zip entry: " + entryName);
223 
224 				//skip some sys/tmp files (that MacOSX creates sometimes)
225 				if(entry.isDirectory() || entryName.contains("__MACOSX") 
226 						|| entryName.startsWith(".") || entryName.contains("/.") 
227 						|| entryName.contains("\\.")) 
228 				{
229             		LOGGER.info("analyzeAndOrganizeContent(), skipped " + entryName);
230 					continue;
231 				}
232 				
233 				// create pathway data object
234 				LOGGER.info("analyzeAndOrganizeContent(), adding new Content: " 
235 					+ entryName + " of " + metadata.getIdentifier());
236 				Content content = new Content(metadata, entryName);
237 				// add object to return collection
238 				contentCollection.add(content);
239 				
240 				OutputStream gzos = new GZIPOutputStream(new FileOutputStream(content.originalFile()));
241 				copy(zis, gzos);
242 				gzos.close();
243             }           
244 		} catch (IOException e) {
245 			throw new RuntimeException("analyzeAndOrganizeContent(), " +
246 					"failed reading from: " + metadata.getIdentifier() , e);
247 		} finally {
248 			closeQuietly(zis);
249 		}
250 		
251 		if(contentCollection != null && !contentCollection.isEmpty())
252 			metadata.getContent().addAll(contentCollection);
253 		else
254 			LOGGER.warn("analyzeAndOrganizeContent(), no data found for " + metadata);
255     }

258     /**
259      * Uncompresses the zip input stream,
260      * all entries if it's a multi-entry archive,
261      * and writes to the output stream.
262      * 
263      * Skips for system files/directory entries.
264      * 
265      * Does not close the streams.
266      * 
267      * @param is
268      * @param os
269      */
270     public static void unzip(ZipInputStream zis, OutputStream os) 
271     {
272 		try {		
273 			// interate over zip entries
274 			ZipEntry entry = null;
275             while ((entry = zis.getNextEntry()) != null) 
276             {
277             	String entryName = entry.getName();
278 				//skip some sys/tmp files (that MacOSX creates sometimes)
279 				if(entry.isDirectory() || entryName.contains("__MACOSX") 
280 						|| entryName.startsWith(".") || entryName.contains("/.") 
281 						|| entryName.contains("\\.")) 
282 				{
283 					continue;
284 				}
285 				
286 				copy(zis, os); //does not close os
287             }           
288 		} catch (IOException e) {
289 			throw new RuntimeException("unzip(), failed", e);
290 		} finally {
291 			closeQuietly(os);
292 		}
293     }

323      * @param src
324      * @param file
325      * @throws RuntimeException when there was an IO problem
326      */
327     public static void write(byte[] src, String file) {
328     	FileOutputStream os = null;
329     	try {
330     		os = new FileOutputStream(file);
331     		os.write(src);
332     		os.flush();

336     	} finally {closeQuietly(os);}
337     }

341 	 * Replaces the URI of a BioPAX object
342 	 * using java reflection. Normally, one should avoid this;
343 	 * please use when absolutely necessary and with great care. 
344 	 * 
345 	 * @param model
346 	 * @param el
347 	 * @param newRDFId
348 	 */
349 	public static  void replaceID(Model model, BioPAXElement el, String newRDFId) {
350 		if(el.getRDFId().equals(newRDFId))
351 			return; // no action required
352 		
353 		model.remove(el);
354 		try {
355 			Method m = BioPAXElementImpl.class.getDeclaredMethod("setRDFId", String.class);
356 			m.setAccessible(true);
357 			m.invoke(el, newRDFId);
358 		} catch (Exception e) {
359 			e.printStackTrace();
360 			throw new RuntimeException(e);
361 		}
362 		model.add(el);
363 	}

367 	 * Loads the BioPAX model from a Gzip archive 
368 	 * previously created by the same cpath2 instance.
369 	 * 
370 	 * @param archive
371 	 * @return big BioPAX model
372 	 */
373 	static Model importFromTheArchive(String archive) {
374 		
375 		Model model = null;
376 
377 		try {
378 			LOGGER.info("Loading the BioPAX Model from " + archive);
379 			model = (new SimpleIOHandler(BioPAXLevel.L3))
380 					.convertFromOWL(new GZIPInputStream(new FileInputStream(archive)));
381 		} 
382 		catch (IOException e) {
383 			LOGGER.error("Failed to import model from " + archive, e);
384 		}
385 
386 		return model;
387 	}

390 	/**
391 	 * Downloads a file (content) from a URI
392 	 * and saves in the cpath2 home directory. The content
393 	 * can be anything, but only single-file GZIP archives 
394 	 * can be optionally expanded with this method, before saved
395 	 * (e.g., this is how we grab GeoIP GeoLiteCity database).
396 	 * 
397 	 * @param srcUrl remote URL
398 	 * @param destFile name or relative path and name
399 	 * @param unpack if true, expands the archive
400 	 * @param replace
401 	 * @return bytes saved or 0 if existed before file weren't replaced
402 	 * @throws RuntimeException when there was an IOException
403 	 */
404 	public static long download(String srcUrl, String destFile, 
405 			boolean unpack, boolean replace) {
406 		
407 		File localFile = new File(destFile);
408 		
409 		if(localFile.exists() && !replace) {
410 			LOGGER.info("Keep existing " + destFile);
411 			return 0L;
412 		}
413 		
414 		Resource resource = LOADER.getResource(srcUrl);
415         long size = 0; 
416         
417         try {
418         
419         if(resource.isReadable()) {
420         	size = resource.contentLength();
421         	LOGGER.info(srcUrl + " content length= " + size);
422         }       
423         if(size < 0) 
424         	size = 100 * 1024 * 1024 * 1024;
425         
426         //downoad to a tmp file
427         ReadableByteChannel source = Channels.newChannel(resource.getInputStream());      
428        	File tmpf = File.createTempFile("cpath2_", ".download");
429        	tmpf.deleteOnExit();
430         FileOutputStream dest = new FileOutputStream(tmpf);        
431         size = dest.getChannel().transferFrom(source, 0, size);
432         dest.close();
433         LOGGER.info(size + " bytes downloaded from " + srcUrl);
434         
435         if(unpack) {
436         	GZIPInputStream ginstream = new GZIPInputStream(new FileInputStream(tmpf));
437         	FileOutputStream outstream = new FileOutputStream(localFile);
438         	byte[] buf = new byte[1024]; 
439         	int len;
440         	while ((len = ginstream.read(buf)) > 0) 
441         		outstream.write(buf, 0, len);
442         	ginstream.close();
443         	outstream.close();
444         } else {
445         	if(replace)
446         		if(localFile.exists() && !localFile.delete())
447             		throw new RuntimeException("Failed to delete old " 
448             			+ localFile.getAbsolutePath());
449         	if(!tmpf.renameTo(localFile))
450         		throw new RuntimeException("Failed to move " 
451         			+ tmpf.getAbsolutePath() + " to " + localFile.getAbsolutePath());
452         }
453         
454         } catch(IOException e) {
455         	throw new RuntimeException("download(). failed", e);
456         }
457               
458         return size;
459 	}

476 	/**
477 	 * For a warehouse (normalized) EntityReference's or CV's URI 
478 	 * gets the corresponding identifier (e.g., UniProt or ChEBI primary ID).
479 	 * (this depends on current biopax normalizer and cpath2 premerge
480 	 * that make/consume 'http://identifiers.org/*' URIs for those utility class biopax objects.)
481 	 * 
482 	 * @param uri
483 	 * @return
484 	 */
485 	public static String idfromNormalizedUri(String uri) {
486 		Assert.isTrue(uri.contains("http://identifiers.org/"));
487 		return uri.substring(uri.lastIndexOf('/')+1);
488 	}

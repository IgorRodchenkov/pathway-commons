1   package cpath.client.util;

3   import org.biopax.paxtools.io.BioPAXIOHandler;
4   import org.biopax.paxtools.model.Model;
5   import org.springframework.http.HttpInputMessage;
6   import org.springframework.http.HttpOutputMessage;
7   import org.springframework.http.MediaType;
8   import org.springframework.http.converter.HttpMessageConverter;
9   
10  import java.io.BufferedInputStream;
11  import java.io.IOException;
12  import java.util.ArrayList;
13  import java.util.Collections;
14  import java.util.List;
15  
16  /**
17   * Converts the cpath2 REST response into a BioPAX model.
18   */
19  public class BioPAXHttpMessageConverter implements HttpMessageConverter<Model> {
20      private final BioPAXIOHandler bioPAXIOHandler;
21      private static final List<MediaType> mediaList;
22      
23      static {
24      	mediaList = new ArrayList<MediaType>();
25          mediaList.add(MediaType.TEXT_PLAIN);       
26      }
27  
28      public BioPAXHttpMessageConverter(BioPAXIOHandler bioPAXIOHandler) {
29          this.bioPAXIOHandler = bioPAXIOHandler;
30      }
31  
32      public boolean canRead(Class<?> clazz, MediaType mediaType) {
33          return Model.class.equals(clazz);
34      }
35  
36      public boolean canWrite(Class<?> clazz, MediaType mediaType) {
37          return false;
38      }
39  
40      public List<MediaType> getSupportedMediaTypes() {
41          return Collections.unmodifiableList(mediaList);
42      }
43  
44      public Model read(Class<? extends Model> clazz, HttpInputMessage inputMessage) throws IOException
45      {
46          /* OK, a little bit of hacking here:
47  
48             The PC2 server returns either a valid BioPAX model or an error coded in XML.
49             So if the BioPAX IO Handler fails, we have to re-read the stream from the
50             beginning in order to parse the error details.
51  
52             This is why we copy the stream into a buffered one.
53           */    	
54      	
55          BufferedInputStream bis = new BufferedInputStream(inputMessage.getBody());
56          bis.mark(0);
57  
58          return bioPAXIOHandler.convertFromOWL(bis);
59      }
60  
61      public void write(Model model, MediaType contentType, HttpOutputMessage outputMessage) throws IOException
62      {
63          throw new UnsupportedOperationException("Not supported");
64      }
65  }

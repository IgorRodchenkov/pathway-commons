18  /**
19   * Using the ChEBI ontology (OBO),
20   * adds intra-ChEBI relationships to the BioPAX model.
21   * 
22   * This can go after the small molecule warehouse 
23   * is built from ChEBI data (or even after cPath2 Merge stage). 
24   */
25  final class ChebiOntologyAnalysis implements Analysis<Model>
26  {
27  	private InputStream inputStream; //ChEBI OBO data
28  	
29  	public void setInputStream(InputStream is) {
30  		this.inputStream = is;
31  	}
32  	
33  	private static final String CHEBI_OBO_ENTRY_START = "[Term]";	
34      private static final Logger log = LoggerFactory.getLogger(ChebiOntologyAnalysis.class);
35  	  
36  	private final Pattern CHEBI_OBO_ID_REGEX = Pattern.compile("^id: CHEBI:(\\w+)$");
37  	private final Pattern CHEBI_OBO_ISA_REGEX = Pattern.compile("^is_a: CHEBI:(\\w+)$");
38  	private final Pattern CHEBI_OBO_RELATIONSHIP_REGEX = Pattern.compile("^relationship: (\\w+) CHEBI:(\\w+)$");
39  	private final String _COLON = ":";
40  	
41  	
42  	/**
43  	 * Given a ChEBI - OBO entry, creates memberEntityReference link 
44  	 * between the parent and child, and adds ChEBI internal relationship xrefs.
45  	 * 
46  	 * @param entryBuffer
47  	 * @param model existing biopax Warehouse model
48  	 * @throws IOException
49  	 */
50  	private void processOBOEntry(StringBuilder entryBuffer, Model model) throws IOException {
51  		if (log.isDebugEnabled()) {
52  			log.debug("calling processOBOEntry()");
53  		}
54  		// get SMR for entry out of Warehouse
55  		Collection<String> childChebiIDs = getValuesByREGEX(entryBuffer, CHEBI_OBO_ID_REGEX);
56  		if (childChebiIDs.size() != 1) {
57  			log.error("processOBOEntry(), got none or >1 ID in: " + entryBuffer.toString() + "; skipped.");
58  			return;
59  		}
60  		final String thisID = childChebiIDs.iterator().next();
61  		SmallMoleculeReference thisSMR = (SmallMoleculeReference) model
62  			.getByID("http://identifiers.org/chebi/CHEBI:" + thisID);
63  		if (thisSMR == null) {
64  			log.debug("processOBOEntry(), Skipped (not found): " + thisID);
65  			return;
66  		}
67  
68  		// for each parent ChEBI, create a member entity reference to child
69  		Collection<String> parentChebiIDs = getValuesByREGEX(entryBuffer, CHEBI_OBO_ISA_REGEX);
70  		for (String parentChebiID : parentChebiIDs) {
71  			SmallMoleculeReference parentSMR = (SmallMoleculeReference) model
72  				.getByID("http://identifiers.org/chebi/CHEBI:" + parentChebiID);
73  			
74  			//always add a rel. xref and is_a comments:
75  			log.info("processOBOEntry(), CHEBI:" + thisID 
76  					+ " 'IS_A' already ignored CHEBI:" + parentChebiID 
77  					+ " (generic, no InChIKey); will add xref/comment instead of memberEntityRef.");
78  			RelationshipXref xref = PreMerger
79  				.findOrCreateRelationshipXref(RelTypeVocab.MULTIPLE_PARENT_REFERENCE, 
80  						"CHEBI", "CHEBI:"+parentChebiID, model);
81  			thisSMR.addComment("is_a CHEBI:" + parentChebiID);
82  			thisSMR.addXref(xref);
83  			
84  			//if the parent SMR exists in the model (has InChIKey) -
85  			if (parentSMR != null) {
86  				parentSMR.addMemberEntityReference(thisSMR);
87  			}
88  		}
89  
90  		// store horizontal relationships between ChEBI terms (has_part, has_role, is_conjugate_*,..)
91  		Collection<String> relationships = getValuesByREGEX(entryBuffer, CHEBI_OBO_RELATIONSHIP_REGEX);
92  		for (String relationship : relationships) {
93  			String[] parts = relationship.split(_COLON);
94  			RelationshipXref xref = PreMerger
95  				.findOrCreateRelationshipXref(RelTypeVocab.ADDITIONAL_INFORMATION, 
96  						"ChEBI", "CHEBI:"+parts[1], model);		
97  			thisSMR.addComment(parts[0].toLowerCase() + " CHEBI:" + parts[1]);
98  			thisSMR.addXref(xref);
99  		}
100 	}
101 
102 	/**
103 	 * Given an OBO entry, returns the values matched by the given regex. If
104 	 * regex contains more that one capture group, a ":" will be used to delimit
105 	 * them.
106 	 * 
107 	 * @param entryBuffer
108 	 * @param regex
109 	 * @return
110 	 * @throws IOException
111 	 */
112 	private Collection<String> getValuesByREGEX(StringBuilder entryBuffer,
113 			Pattern regex) throws IOException {
114 
115 		Collection<String> toReturn = new ArrayList<String>();
116 		BufferedReader reader = new BufferedReader (new StringReader(entryBuffer.toString()));
117 
118 		if (log.isDebugEnabled()) {
119 			log.debug("getValue(), key: " + regex.toString());
120 		}
121 
122 		String line = reader.readLine();
123 		while (line != null) {
124 			Matcher matcher = regex.matcher(line);
125 			if (matcher.find()) {
126 				String toAdd = "";
127 				for (int lc = 1; lc <= matcher.groupCount(); lc++) {
128 					toAdd += matcher.group(lc) + _COLON;
129 				}
130 				toReturn.add(toAdd.substring(0, toAdd.length() - 1));//to remove ending ':'
131 			}
132 			line = reader.readLine();
133 		}
134 
135 		return toReturn;
136 	}

139 	public void execute(Model model) {
140 		log.info("convert(), starting to read data...");
141 		
142 		if(inputStream == null)
143 			throw new IllegalArgumentException("The second parameter must be not null input stream");
144 		
145 		try {
146 			BufferedReader reader = new BufferedReader(
147 					new InputStreamReader(inputStream, "UTF-8"));			
148 			
149 			String line;
150 			while ((line = reader.readLine()) != null) {
151 				// start of entry
152 				if (line.startsWith(CHEBI_OBO_ENTRY_START)) {
153 					StringBuilder entryBuffer = new StringBuilder(line + "\n");
154 					while ((line = reader.readLine()) != null) {
155 						// keep reading until we reach last modified
156 						if (line.isEmpty())
157 							break;
158 						entryBuffer.append(line + "\n");
159 					}
160 
161 					processOBOEntry(entryBuffer, model);
162 				}
163 			}
164 
165 			if (reader != null)
166 				reader.close();
167 		} catch (IOException e) {
168 			log.error("is.close() failed." + e);
169 		}
170 		
171 		log.info("convert(), exiting.");
172 	}

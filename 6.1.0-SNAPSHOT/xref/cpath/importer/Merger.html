87  	 * Constructor.

90  	 * @param force whether to forcibly merge BioPAX data the validation reported critical about or skip.

93  	public Merger(CPathService service, boolean force) 
94  	{
95  		this.service = service;
96  		this.xmlBase = CPathSettings.getInstance().getXmlBase();
97  		this.force = force;		
98  		
99  		this.warehouseModel = CPathUtils.loadWarehouseBiopaxModel();
100 		Assert.notNull(warehouseModel, "No BioPAX Warehouse");
101 		log.info("Successfully imported Warehouse BioPAX archive.");	
102 		
103 		this.mainModel = BioPAXLevel.L3.getDefaultFactory().createModel();
104 		this.mainModel.setXmlBase(xmlBase);
105 		log.info("Created a new empty main BioPAX model.");
106 	}

109 	public void merge() {
110 		SimpleIOHandler simpleReader = new SimpleIOHandler(BioPAXLevel.L3);
111 
112 		// build models and merge from dataFile.premergeData
113 		Collection<Metadata> providersMetadata = new ArrayList<Metadata>();
114 		
115 		for(Metadata metadata : service.metadata().findAll())
116 			providersMetadata.add(metadata);
117 
118 		for (Metadata metadata : providersMetadata) {
119 			
120 			if(metadata.isNotPathwayData()) {
121 				log.info("Skip for warehouse data: " + metadata);
122 				continue;
123 			}
124 			
125 			log.info("Start merging " + metadata);
126 			for (Content pwdata : metadata.getContent()) {		
127 				final String description = pwdata.toString();
128 				if (pwdata.getValid() == null) {
129 					log.warn("Skipped " + description + " - haven't gone through the premerge yet");
130 					continue;
131 				} else if (pwdata.getValid() == false) {
132 					// has BioPAX errors
133 					log.warn("There were critical BioPAX errors in - " + description);
134 					if (!force) {
135 						log.warn("Skipped " + description + " (due to BioPAX errors)");
136 						continue;
137 					} else {
138 						log.warn("FORCE merging " + description + " (ignoring BioPAX errors)");
139 					}
140 				}
141 
142 				log.info("Merging: " + description);
143 				
144 				// import the BioPAX L3 pathway data into the in-memory paxtools model
145 				InputStream inputStream;
146 				try {
147 					inputStream = new GZIPInputStream(new FileInputStream(pwdata.normalizedFile()));
148 				} catch (IOException e) {
149 					log.error("Skipped " + description + " - " +
150 						"failed to read from " + pwdata.normalizedFile());
151 					continue;
152 				}
153 				
154 				Model pathwayModel = simpleReader.convertFromOWL(inputStream);
155 
156 				// merge the input biopax model
157 				merge(pwdata.toString(), pathwayModel);
158 			}
159 			
160 			log.info("Done merging " + metadata);
161 		}
162 		
163 		//create or replace the main BioPAX archive
164 		log.info("Saving or updating the Main BioPAX file...");
165 		save();
166 		
167 		log.info("Complete.");
168 	}

172 	 * in the cpath2 downloads (in production) or tmp (tests)

179 						CPathSettings.getInstance().mainModelFile())));
180 		} catch (Exception e) {
181 			throw new RuntimeException("Failed updating the main BioPAX archive!", e);
182 		}

186 	/**
187 	 * Integrates the source model into the main one using the 
188 	 * reference biopax objects (entity references, etc.) 
189 	 * in the target or warehouse models.
190 	 * As the result, the source model will (and the warehouse might) be broken 
191 	 * (ok to dispose), and the target - complete, self-integral,
192 	 * contain all the entities that were there before and new from the source
193 	 * model (where some original utility class objects are replaced with 
194 	 * canonical warehouse or previously existing target ones). 
195 	 * 
196 	 * @param description
197 	 * @param source
198 	 */
199 	void merge(String description, final Model source) {	
200 		
201 		final String srcModelInfo = "source: " + description;
202 		
203 		log.info("Converting all Xref.db values to upper case (" + srcModelInfo + "; "
204 				+ "helps when merging xrefs...");
205 		for(Xref x : source.getObjects(Xref.class))
206 			if(x.getDb()!=null) x.setDb(x.getDb().toUpperCase());
207 			
208 		log.info("Searching for canonical or existing EntityReference objects " +
209 				" to replace equivalent original objects ("+srcModelInfo+")...");
210 		final Map<EntityReference, EntityReference> replacements = new HashMap<EntityReference, EntityReference>();		
211 		// match some UtilityClass objects to existing ones (previously imported, warehouse)
212 		for (EntityReference bpe: new HashSet<EntityReference>(source.getObjects(EntityReference.class))) 
213 		{
214 			EntityReference replacement = null;
215 			
216 			// Find the best replacement ER in the Warehouse:
217 			if (bpe instanceof ProteinReference) {
218 				replacement = findOrCreateProteinReference((ProteinReference)bpe);
219 			} 
220 			else if (bpe instanceof SmallMoleculeReference) {
221 				replacement = findOrCreateSmallMoleculeReference((SmallMoleculeReference)bpe);
222 			} 
223 				
224 			if (replacement != null) {
225 				EntityReference r = (EntityReference) mainModel.getByID(replacement.getRDFId());
226 				if(r != null) // re-use previously merged one
227 					replacement = r;
228 				
229 				//save in the map to replace the source bpe later 
230 				replacements.put(bpe, replacement);
231 				
232 				if(!replacement.getRDFId().equals(bpe.getRDFId()))
233 					replacement.addComment("REPLACED " + bpe.getRDFId());
234 			}
235 		}
236 		
237 		//explicitly remove old (to be replaced) objects from the source model
238 		// this is important for the replacement (below) to work, esp. in case 
239 		// new URI is the same as original normalized URI...
240 		for(EntityReference old : replacements.keySet()) {
241 			source.remove(old);
242 		}
243 		
244 		// Replace objects in the source model
245 		log.info("Replacing objects ("+srcModelInfo+")...");	
246 		ModelUtils.replace(source, replacements);
247 
248 		// post-fix
249 		log.info("Migrate some properties, such as original entityFeature and xref ("+srcModelInfo+")...");
250 		for (EntityReference old : replacements.keySet()) {
251 			
252 			final EntityReference repl = replacements.get(old);	
253 			
254 			for (EntityFeature ef : new HashSet<EntityFeature>(old.getEntityFeature())) 
255 			{ // move entity features of the replaced ER to the new canonical one
256 				// remove the ef from the old ER
257 				old.removeEntityFeature(ef);
258 				// now, this ef should not belong to any other ER (no entityFeature can contain this ef, for all ERs)
259 				//TODO ideally, we'd check for all the ERs in both the original and target models, and none should contain this ef.
260 											
261 				// If there exist an equivalent, don't add original 'ef', 
262 				// but just replace with the equiv. one in all PEs of given old ER
263 				EntityFeature equivEf = null;
264 				for(EntityFeature f : repl.getEntityFeature()) {
265 					if(f.isEquivalent(ef)) {
266 						equivEf = f;
267 						equivEf.getComment().addAll(ef.getComment());
268 						break;
269 					}
270 				}			
271 				if(equivEf == null) //add new EF to the canonical ER
272 					repl.addEntityFeature(ef);
273 				else { //update PEs' feature and notFeature properties to use the existing equiv. feature
274 					for(PhysicalEntity pe : new HashSet<PhysicalEntity>(ef.getFeatureOf())) {
275 						pe.removeFeature(ef);
276 						pe.addFeature(equivEf);
277 					}
278 					for(PhysicalEntity pe : new HashSet<PhysicalEntity>(ef.getNotFeatureOf())) {
279 						pe.removeNotFeature(ef);
280 						pe.addNotFeature(equivEf);
281 					}	
282 				}
283 			}				
284 			// move new PublicationXrefs and RelationshipXrefs (otherwise we lost some of original xrefs...)
285 			for(Xref x : new HashSet<Xref>(old.getXref())) {
286 				if(!(x instanceof UnificationXref)) {
287 					((XReferrable) old).removeXref(x);				
288 					Xref equivX = null;
289 					for(Xref y : repl.getXref()) {
290 						if(y.isEquivalent(x)) {
291 							equivX = y;
292 							break;
293 						}
294 					}
295 					//if the repl. ER has neither same-id xrefs nor equivalent ones, add x:
296 					if(!repl.getXref().contains(x) && equivX == null)
297 							repl.addXref(x);
298 				}
299 			}
300 		}
301 		
302 		// cleaning up dangling objects (including the replaced above ones)
303 		log.info("Removing dangling objects ("+srcModelInfo+")...");
304 		ModelUtils.removeObjectsIfDangling(source, UtilityClass.class);
305 		
306 		/* 
307 		 * The following can improve graph queries and full-text search, 
308 		 * for generic and poorly defined physical entities (lacking entity reference)
309 		 * can eventually match a query.
310 		 * 
311 		 * Using existing xrefs and id-mapping, add primary uniprot/chebi RelationshipXref 
312 		 * to all simple PEs (SM, Protein, Dna, Rna,..) and Gene, if possible (skip Complexes).
313 		 * This might eventually result in mutually exclusive identifiers, 
314 		 * but we'll keep those xrefs and just log a warning for future (data) fix;
315 		 * - not a big deal as long as we do not merge data based on these new xrefs,
316 		 * but just index/search/query (this especially helps 
317 		 * when no entity references defined for a molecule).
318 		 */	
319 		log.info("Adding canonical UniProt/ChEBI RelationshipXrefs to physical"
320 			+ " entities by using existing xrefs and id-mapping (" + srcModelInfo + ")");
321 		for(Entity pe : new HashSet<Entity>(source.getObjects(Entity.class))) 
322 		{
323 			if(pe instanceof PhysicalEntity) {
324 				if(pe instanceof SimplePhysicalEntity) {
325 					// skip for SPE that got its ER just replaced (from Warehouse)
326 					EntityReference er = ((SimplePhysicalEntity) pe).getEntityReference();
327 					if(er != null && warehouseModel.containsID(er.getRDFId()))
328 						continue;
329 					
330 					if(pe instanceof SmallMolecule) {
331 						if(er == null)
332 							addCanonicalRelXrefs(mainModel, pe, "CHEBI");
333 						else
334 							addCanonicalRelXrefs(mainModel, er, "CHEBI");
335 					} else {//Protein, Dna*, Rna* type
336 						if(er == null)
337 							addCanonicalRelXrefs(mainModel, pe, "UNIPROT");
338 						else 
339 							addCanonicalRelXrefs(mainModel, er, "UNIPROT");
340 					}						
341 				} else if(pe instanceof Complex) {
342 					continue; // skip
343 				} else { // top PE class, i.e., pe.getModelInterface()==PhysicalEntity.class
344 					addCanonicalRelXrefs(mainModel, pe, "UNIPROT");
345 					addCanonicalRelXrefs(mainModel, pe, "CHEBI");
346 				}
347 			} else if(pe instanceof Gene) {
348 				addCanonicalRelXrefs(mainModel, pe, "UNIPROT");
349 			}
350 		}
351 		
352 		/* 
353 		 * Replace all not normalized so far URIs in the source model 
354 		 * with auto-generated new short ones (also add a bp:comment about original URIs)
355 		 */
356 		log.info("Assigning new URIs (xml:base=" + xmlBase + 
357 				"*) to all not normalized BioPAX elements (" + 
358 				srcModelInfo + ", xml:base=" + source.getXmlBase() + ")...");
359 		//wrap source.getObjects() in a new set to avoid concurrent modif. excep.
360 		for(BioPAXElement bpe : new HashSet<BioPAXElement>(source.getObjects())) {
361 			String currUri = bpe.getRDFId();
362 			
363 			// skip for previously normalized objects (we believe they are right and ready to merge)
364 			if(currUri.startsWith(xmlBase) || currUri.startsWith("http://identifiers.org/")) 
365 				continue; 
366 			
367 			// Generate new consistent URI for not generated not previously normalized objects:
368 			String newRDFId = Normalizer.uri(xmlBase, null, currUri, bpe.getModelInterface());
369 			
370 			// Avoid same URI - non-equivalent objects case/clash 
371 			// (some providers might mistakenly use the same URI 
372 			// for several non-equivalent biopax entities in different files):
373 			if(mainModel.containsID(newRDFId)) {
374 				if(!bpe.isEquivalent(mainModel.getByID(newRDFId))) {
375 					newRDFId = Normalizer.uri(xmlBase, null, description+currUri, bpe.getModelInterface());
376 					log.warn(description + " has a " + bpe.getModelInterface().getSimpleName() + 
377 						", uri: " + currUri + ", which also was the original URI of a " + 
378 						"not equivalent, previously merged object; fixed.");
379 				}
380 			}
381 			
382 			// Replace URI
383 			CPathUtils.replaceID(source, bpe, newRDFId);
384 			
385 			// save original URI in comments
386 			((Level3Element) bpe).addComment("REPLACED " + currUri);
387 		}		
388 		
389 		log.info("Merging into the main BioPAX model...");
390 		// merge all the elements and their children from the source to target model
391 		mainModel.merge(source);
392 		
393 		log.info("Merge ("+srcModelInfo+") is done.");
394 	}

397 	/**
398 	 * Performs id-mapping from the  
399 	 * unification and relationship xrefs 
400 	 * of a physical entity or gene to the primary/canonical
401 	 * id (uniprot or chebi), creates new relationship xrefs,
402 	 * and adds them back to the entity.
403 	 * 
404 	 * @param m where to add new xrefs (and who's xml:base to apply for new URIs)
405 	 * @param bpe a {@link Gene} or {@link PhysicalEntity}
406 	 * @param db map identifiers to (e.g., CHEBI, UNIPROT)
407 	 * @throws AssertionError when bpe is neither Gene nor PhysicalEntity
408 	 */
409 	private void addCanonicalRelXrefs(Model m, Named bpe, String db) 
410 	{
411 		if(!(bpe instanceof Gene || bpe instanceof PhysicalEntity || bpe instanceof EntityReference))
412 			throw new AssertionError("Not Gene or PE: " + bpe);
413 			
414 		// map and generate/add xrefs
415 		Set<String> mappingSet = idMappingByXrefs(bpe, db, UnificationXref.class, true);
416 		addCanonicalRelXrefs(m, bpe, db, mappingSet, RelTypeVocab.IDENTITY);
417 		
418 		mappingSet = idMappingByXrefs(bpe, db, RelationshipXref.class, true);
419 		addCanonicalRelXrefs(m, bpe, db, mappingSet, RelTypeVocab.ADDITIONAL_INFORMATION);
420 		
421 		//map by display and standard names
422 		if(bpe.getDisplayName()!=null && !bpe.getDisplayName().isEmpty()) {
423 			mappingSet = service.map(null, bpe.getDisplayName(), db);
424 			addCanonicalRelXrefs(m, bpe, db, mappingSet, RelTypeVocab.ADDITIONAL_INFORMATION);
425 		}
426 	}

429 	/**
430 	 * Finds or creates relationship xrefs
431 	 * using id-mapping results;
432 	 * adds them to the object and model.
433 	 * 
434 	 * @param model a biopax model where to find/create xrefs
435 	 * @param bpe a gene, physical entity or entity reference only
436 	 * @param db database name for all (primary/canonical) xrefs; 'uniprot' or 'chebi'
437 	 * @param mappingSet
438 	 * @param relType - term to use with the Xref
439 	 * @throws AssertionError when bpe is neither Gene nor PhysicalEntity nor EntityReference
440 	 */
441 	private void addCanonicalRelXrefs(Model model, XReferrable bpe, String db, Set<String> mappingSet, RelTypeVocab relType)
442 	{	
443 		if(!(bpe instanceof Gene || bpe instanceof PhysicalEntity || bpe instanceof EntityReference))
444 			throw new AssertionError("Not Gene or PE: " + bpe);
445 		
446 		ac: for(String ac : mappingSet) {
447 			// find or create
448 			RelationshipXref rx = PreMerger
449 				.findOrCreateRelationshipXref(relType, db, ac, model);
450 			
451 			//check if an equivalent rel. xref is already present (skip it then)
452 			for(Xref x : bpe.getXref()) {
453 				if(x instanceof RelationshipXref && x.isEquivalent(rx))
454 					continue ac; //break and go to next ac
455 			}
456 			
457 			bpe.addXref(rx);
458 		}
459 	}

462 	/**
463 	 * Finds previously created or generates (searching in the db) 
464 	 * a new {@link ProteinReference} BioPAX element that is equivalent 
465 	 * to the original one and has standard URI and properties, 
466 	 * which allows to simply merge it with other semantically equivalent ones, by ID (URI).
467 	 * 
468 	 * @param orig
469 	 * @return the replacement object or null if none can found
470 	 */
471 	private ProteinReference findOrCreateProteinReference(final ProteinReference orig) 
472 	{				
473 		ProteinReference toReturn = null;	
474 		
475 		final String standardPrefix = "http://identifiers.org/";
476 		final String canonicalPrefix = standardPrefix + "uniprot/";
477 		final String origUri = orig.getRDFId();
478 		
479 		// Try to re-use existing object
480 		if(origUri.startsWith(canonicalPrefix)) {
481 			toReturn = (ProteinReference) warehouseModel.getByID(origUri);
482 		}
483  
484 		// If nothing's found by URI right away, 
485 		// try id-mapping (of the normalized URI part to chebi primary accession) 
486 		if (toReturn == null && origUri.startsWith(standardPrefix)) {
487 			String id = origUri.substring(origUri.lastIndexOf('/')+1);	
488 			String db = null;				
489 			//a hack/shortcut for normalized PRs
490 			if(orig instanceof ProteinReference) {
491 				if(origUri.toLowerCase().contains("uniprot.isoform")) {
492 					db = "uniprot isoform";
493 				} else if(origUri.toLowerCase().contains("uniprot")) {
494 					db = "uniprot";
495 				} else if(origUri.toLowerCase().contains("refseq")) {
496 					db = "refseq";	
497 				} else if(origUri.toLowerCase().contains("kegg") && id.contains(":")) {
498 					db = "NCBI Gene"; //KEGG actually uses NCBI Gene (aka Entrez Gene)
499 				}
500 			}			
501 			if(db == null) 
502 				db = dbById(id, orig.getXref());	
503 
504 			Set<String> mp = service.map(db, id, "UNIPROT");
505 			Set<EntityReference> ers = findEntityRefUsingIdMappingResult("URI", mp, canonicalPrefix);
506 			if(ers.size()>1)
507 				log.warn(origUri + " maps to multiple warehouse ERs: " + ers);
508 			else if (!ers.isEmpty())
509 				toReturn = (ProteinReference) ers.iterator().next();
510 		}
511 				
512 		// if yet ambiguous mapping or nothing, 
513 		// try using (already normalized) Xrefs and id-mapping  
514 		if (toReturn == null)
515 			toReturn = (ProteinReference) mapByXrefs(orig, "UNIPROT", canonicalPrefix);
516 		
517 		// nothing/ambiguous?..
518 		//TODO map by names? (too RISKY, might replace wrong PR, for proteins/genes often share names!)
519 	
520 		return toReturn;
521 	}

523 	/**
524 	 * Using specified class xrefs of given object, 
525 	 * finds primary identifiers (can be many).
526 	 * 
527 	 * 
528 	 * @param orig
529 	 * @param mapTo
530 	 * @param xrefType
531 	 * @param isUnion tells to return either union or intersection of each Xref's id-mapping result
532 	 * @return
533 	 */
534 	private Set<String> idMappingByXrefs(final XReferrable orig,
535 			String mapTo, final Class<? extends Xref> xrefType, boolean isUnion) 
536 	{
537 		return (isUnion) ? idMappingByXrefsUnion(orig, mapTo, xrefType)
538 				: idMappingByXrefsIntersection(orig, mapTo, xrefType);
539 	}
540 	
541 
542 	private Set<String> idMappingByXrefsIntersection(final XReferrable orig,
543 			String mapTo, final Class<? extends Xref> xrefType) 
544 	{
545 		Set<String> xSet = new HashSet<String>();
546 		
547 		for (Xref x : orig.getXref()) {			
548 			if(x.getDb() == null || x.getDb().isEmpty() || x.getId() == null || x.getId().isEmpty()) {
549 				log.warn("Ignored bad " + xrefType.getSimpleName()
550 					+ " (" + x.getRDFId() + "), db: " + x.getDb() + ", id: " + x.getId());
551 				continue;
552 			}
553 						
554 			if (xrefType.isInstance(x)) {
555 				Set<String> mp = service.map(x.getDb(), x.getId(), mapTo);
556 				if(mp==null || mp.isEmpty()) 
557 					continue; //ignore xrefs that don't map to any primary IDs
558 				
559 				// mp is not empty
560 				if(mp.size() > 1) 
561 					log.debug("Ambiguous xref.id: " + x.getId() + " maps to: " + mp);
562 				
563 				if(!xSet.isEmpty()) {//and mp is not empty too -
564 					mp.retainAll(xSet);
565 					if(mp.isEmpty())
566 						break; //quit w/o trying other xrefs due to apparently empty intersection
567 				} 
568 				
569 				xSet = mp; //xSet now contains not empty intersection
570 				assert !xSet.isEmpty() : "Impossible: xSet is empty.";
571 			}
572 		}
573 
574 		if(xSet.isEmpty())
575 			return Collections.EMPTY_SET;
576 		else if(xSet.size()>1)
577 			return new TreeSet<String>(xSet);
578 		else {
579 			log.info(orig.getRDFId() + ", using its " + xrefType.getSimpleName() 
580 					+ "s, distinctly maps to: " + xSet.iterator().next());
581 			return xSet;
582 		}
583 	}

585 	private Set<String> idMappingByXrefsUnion(final XReferrable orig,
586 			String mapTo, final Class<? extends Xref> xrefType) 
587 	{
588 		final Set<String> mappedTo = new TreeSet<String>();
589 		
590 		for (Xref x : orig.getXref()) {			
591 			if(x.getDb() == null || x.getDb().isEmpty()
592 					|| x.getId() == null || x.getId().isEmpty()) {
593 				log.warn("Ignored bad " + xrefType.getSimpleName()
594 					+ " (" + x.getRDFId() + "), db: " + x.getDb() + ", id: " + x.getId());
595 				continue;
596 			}
597 						
598 			if (xrefType.isInstance(x)) {
599 				Set<String> mp = service.map(x.getDb(), x.getId(), mapTo);
600 				mappedTo.addAll(mp);
601 				if(mp.size() > 1) //one xref maps to several primary ACs
602 					log.debug("Ambiguous xref.id: " + x.getId() + " maps to: " + mp);
603 			}
604 		}
605 
606 		return mappedTo;
607 	}

628 	/**
629 	 * Finds previously created or generates (searching in the data warehouse) 
630 	 * a new {@link SmallMoleculeReference} BioPAX element that is equivalent 
631 	 * to the original one and has standard URI and properties, 
632 	 * which allows to simply merge it with other semantically equivalent ones, by ID (URI).
633 	 * 
634 	 * @param orig
635 	 * @param target
636 	 * @param type
637 	 * @return the replacement object or null if none can found
638 	 */
639 	private SmallMoleculeReference findOrCreateSmallMoleculeReference(final SmallMoleculeReference orig) 
640 	{				
641 		SmallMoleculeReference toReturn = null;	
642 		
643 		final String standardPrefix = "http://identifiers.org/";
644 		final String canonicalPrefix = standardPrefix + "chebi/";
645 		final String origUri = orig.getRDFId();
646 		
647 		// Try to re-use existing object
648 		if(origUri.startsWith(canonicalPrefix)) {
649 			toReturn = (SmallMoleculeReference) warehouseModel.getByID(origUri);
650 		}
651  
652 		// If nothing's found by URI right away, 
653 		// try id-mapping (of the normalized URI part to chebi primary accession) 
654 		if (toReturn == null && origUri.startsWith(standardPrefix)) {
655 			String id = origUri.substring(origUri.lastIndexOf('/')+1);	
656 			String db = dbById(id, orig.getXref()); //find by id			
657 			Set<String> mp = service.map(db, id, "CHEBI");
658 			Set<EntityReference> ers = findEntityRefUsingIdMappingResult("URI", mp, canonicalPrefix);
659 			if(ers.size()>1)
660 				log.warn(origUri + " URI maps to multiple canonical ChEBI SMRs: " + ers);
661 			else if (!ers.isEmpty())
662 				toReturn = (SmallMoleculeReference) ers.iterator().next();
663 		}
664 				
665 		// if yet ambiguous mapping or nothing, 
666 		// try using (already normalized) Xrefs and id-mapping  
667 		if (toReturn == null)
668 			toReturn = (SmallMoleculeReference) mapByXrefs(orig, "CHEBI", canonicalPrefix);
669 		
670 		// nothing/ambiguous? - keep trying, map by name (e..g, 'ethanol') to ChEBI ID
671 		if (toReturn == null) {		
672 			Set<String> mp = mapByName(orig, "CHEBI");
673 			Set<EntityReference> ers = findEntityRefUsingIdMappingResult("names", mp, canonicalPrefix);
674 			if(ers.size()>1)
675 				log.warn(origUri + ", its names match multiple canonical ChEBI SMRs: " + ers);
676 			else if (!ers.isEmpty())
677 				toReturn = (SmallMoleculeReference) ers.iterator().next();
678 		}
679 		
680 		return toReturn;
681 	}

684 	private EntityReference mapByXrefs(EntityReference orig,
685 			String dest, String canonicalUriPrefix) {
686 
687 		EntityReference toReturn = null; 
688 		
689 		Set<String> mappingSet = idMappingByXrefs(orig, dest, UnificationXref.class, false);
690 		
691 		Set<EntityReference> mapsTo = findEntityRefUsingIdMappingResult("UnificationXrefs", mappingSet, canonicalUriPrefix);
692 		if(mapsTo.size()>1) {
693 			log.warn(orig.getRDFId() + ", UnificationXrefs map to multiple: " + mapsTo);
694 		} else if(mapsTo.isEmpty()) {
695 			mappingSet = idMappingByXrefs(orig, dest, RelationshipXref.class, false);
696 			mapsTo = findEntityRefUsingIdMappingResult("RelationshipXrefs", mappingSet, canonicalUriPrefix);
697 			if(mapsTo.size()>1)
698 				log.warn(orig.getRDFId() + ", RelationshipXrefs map to multiple: " + mapsTo);
699 		}
700 		
701 		if(mapsTo.size()==1) 
702 			toReturn = mapsTo.iterator().next();
703 		
704 		return toReturn; //can be null
705 	}

708 	@SuppressWarnings("unchecked")
709 	private Set<String> mapByName(EntityReference orig, String toDb) {
710 		Set<String> mp = new TreeSet<String>();
711 
712 		String name = orig.getStandardName();
713 		if(name != null)
714 			mp.addAll(service.map(null, name.toLowerCase(), toDb));
715 
716 		if(mp.isEmpty()) {
717 			name = orig.getDisplayName();
718 			if(name != null)
719 				mp.addAll(service.map(null, name.toLowerCase(), toDb));
720 		}
721 		
722 		if(mp.isEmpty() && //and only for PRs and SMRs -
723 				(orig instanceof ProteinReference || orig instanceof SmallMoleculeReference)) 
724 		{
725 			//To find a warehouse SMR(s) with exactly the same name (case-insensitive).			
726 			//first, collect all orig names, lowercase
727 			Set<String> origNames = new HashSet<String>();
728 			for(String n : orig.getName()) 
729 				origNames.add(n.toLowerCase());
730 			
731 			//search for warehouse ERs of the same as orig's class
732 			for(EntityReference er : warehouseModel
733 					.getObjects((Class<? extends EntityReference>) orig.getModelInterface())) 
734 			{
735 				for(String s : er.getName()) {
736 					if(origNames.contains(s.toLowerCase())) {
737 						//extract the accession from URI, add
738 						mp.add(CPathUtils.idfromNormalizedUri(er.getRDFId()));
739 						break;
740 					}
741 				}
742 			}			
743 		}
744 		
745 		return mp;
746 	}

750 	{

752 		
753 		for(String id : mapsTo) {		
754 			String uri = uriPrefix + id;
755 			EntityReference er = (EntityReference) warehouseModel.getByID(uri);

760 		return toReturn;

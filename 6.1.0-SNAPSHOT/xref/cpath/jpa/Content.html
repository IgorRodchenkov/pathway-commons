56      /**
57       * Create a Content domain object (value object).
58       * 
59       * @param provider must be output provider for normalized data and validation reports
60       * @param filename file name base (prefix for the normalized data and validation report file names)
61       */
62      public Content(Metadata metadata, String filename) 
63      {    	
64          Assert.notNull(metadata);
65      	Assert.notNull(filename);
66      	this.provider = metadata.getIdentifier();
67          this.filename = filename.replaceAll("[^a-zA-Z0-9.-]", "_");
68      }

70  	
71  	public void saveValidationReport(Validation v) {		
72  		Writer writer;
73  		try {
74  			writer = new OutputStreamWriter(
75  				 new GZIPOutputStream(new FileOutputStream(validationXmlFile())));
76  			ValidatorUtils.write(v, writer, null);
77  			writer.flush();
78  			writer.close();//important
79  		} catch (IOException e) {
80  			log.error("saveValidationReport, failed to save the XML report", e);
81  		}
82  		
83  		// transform to html report and save (frankly, not needed) 
84  		// (fails if old saxon libs present in the classpath, e.g., those come with the psimi-converter...)
85  		try {
86  			writer = new OutputStreamWriter(
87  					 new GZIPOutputStream(new FileOutputStream(validationHtmlFile())));
88  			StreamSource xsl = new StreamSource((new DefaultResourceLoader())
89  				.getResource("classpath:html-result.xsl").getInputStream());
90  			ValidatorUtils.write(v, writer, xsl); 
91  			writer.flush();
92  			writer.close();//important
93  		} catch (Exception e) {
94  			log.error("saveValidationReport, failed to transform the XML to HTML report", e);
95  		}		
96  	}

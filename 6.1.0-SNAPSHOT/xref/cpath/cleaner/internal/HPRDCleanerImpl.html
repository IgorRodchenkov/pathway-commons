3   import java.io.IOException;
4   import java.io.InputStream;
5   import java.io.OutputStream;
6   import java.util.Scanner;
7   
8   import cpath.importer.Cleaner;
9   
10  
11  /**
12   * Implementation of Cleaner interface for HPRD ppi data.
13   */
14  final class HPRDCleanerImpl implements Cleaner {
15  
16  	public void clean(InputStream data, OutputStream cleanedData) {
17  		//HPRD data is less than 2Gb (max String length)
18  		Scanner sc = new Scanner(data);
19  	    StringBuilder sb = new StringBuilder();
20  	    while (sc.hasNextLine())
21  	        sb.append(sc.nextLine());
22  	    sc.close(); sc = null;
23  		
24  	    String pathwayDataString = sb.toString();
25  		// we want to add refType=identity to uniprot secondaryRef
26  		pathwayDataString = pathwayDataString.replaceAll("^(\\s*)<secondaryRef db=\"uniprot\" dbAc=\"(.*)\" id=\"(.*)\"\\/>\\s*$",
27  						 "$1<secondaryRef db=\"uniprot\" dbAc=\"$2\" id=\"$3\" refType=\"identity\"/>");
28  
29          // A quick and dirty fix for the latest export: HPRD_SINGLE_PSIMI_041210.xml
30          // Duplicate id error due to a trailing space error
31  		pathwayDataString = pathwayDataString.replaceAll("\"07467 \"", "\"074670\"");
32  		
33  		try {
34  			cleanedData.write(pathwayDataString.getBytes());
35  			cleanedData.close();
36  		} catch (IOException e) {
37  			throw new RuntimeException("Failed writing to output stream", e);
38  		}
39  	}
40  
41  }

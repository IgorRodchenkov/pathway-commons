11  /**
12   * Implementation of Cleaner interface for UniProt data.
13   * Specifically, the class removes all shared accession numbers
14   * between protein records.  The assumption is these shared accessions
15   * are obsolete and have been left in the uniprot export for backward
16   * compatability.  An example would be:
17   *
18   * ['A2A3T6', 'Q0VGC5', 'Q5VX65', 'Q5VX66', 'Q8IUU4']
19   * between records: C1T9B_HUMAN, id: C1T9A_HUMAN
20   *
21   * This class works on UniProt exports of the form:
22   * uniprot_sprot_xxxx.dat, where xxxx is species.
23   *
24   */
25  final class UniProtCleanerImpl implements Cleaner {
26  
27      // delimiter between accessions
28      private static final String AC_DELIMITER = "; ";
29  
30      // identifier for start of accessions line
31      private static final String AC_PREFIX = "AC   ";
32      
33      // regex to capture protein identifier
34      private static Pattern idPattern = Pattern.compile("^ID\\s*(\\w*).*$");
35      
36      private final Map<String, List<String>> acToIdMap = new HashMap<String, List<String>>();
37      private final Map<String, List<String>> idToAcMap = new HashMap<String, List<String>>();
38      
39      /**
40       * {@inheritDoc} 
41       */
42      public void clean(InputStream data, OutputStream cleanedData) {
43          try {    	     	
44          	//read all data from the input stream to a byte[] stream
45          	//(this will work, because usually UniProt dat file in less than 2Gb)
46          	ByteArrayOutputStream bos = new ByteArrayOutputStream();
47          	IOUtils.copy(data, bos);
48          	bos.close();
49          	data.close();
50          	
51          	populateAccessionsMaps(new ByteArrayInputStream(bos.toByteArray()));
52  
53          	cleanAccessions(new ByteArrayInputStream(bos.toByteArray()), cleanedData);      	
54          }
55          catch (Exception e) {
56          	throw new RuntimeException("clean(), Exception thrown while cleaning UniProt data", e);
57          }
58      }
59  
60      /**
61       * For a given uniprot file, creates a map of accession (key) to list of protein id (values)
62       * So, for each of these entries in the file:
63       *
64       * ID   BRCA1_HUMAN   Reviewed; 1863 AA.
65       * AC   P38398; O15129; Q3LRJ0; Q7KYU9;
66       *
67       * A map would contain:
68       *
69       * P38398 -> [BRCA1_HUMAN]
70       * O15129 -> [BRCA1_HUMAN]
71       * Q3LRJ0 -> [BRCA1_HUMAN]
72       * Q7KYU9 -> [BRCA1_HUMAN,..]
73       *  
74       * later on, if a list (value) contains >1 elements, 
75       * then the corresponding accession number is to be removed 
76       * from all proteins.
77       * 
78       *
79       * @param uniprotData String
80       * @return Map<String, List<String>>
81       */
82      private void populateAccessionsMaps(InputStream uniprotData) throws IOException {
83  
84          // create a buffered reader
85          BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(uniprotData));
86          String line;
87          while ((line = bufferedReader.readLine()) != null) {
88              // process each id
89              Matcher matcher = idPattern.matcher(line);
90              if (matcher.matches()) {
91                  // grab the id
92                  String id = matcher.group(1);
93                  List<String> accessions = getAccessionsList(bufferedReader);
94                  // insert entry into map
95                  for (String accession : accessions) {
96                      if (acToIdMap.containsKey(accession)) {
97                          acToIdMap.get(accession).add(id);
98                      }
99                      else {
100                         List<String> ids = new ArrayList<String>();
101                         ids.add(id);
102                         acToIdMap.put(accession, ids);
103                     }
104                     
105                     if (idToAcMap.containsKey(id)) {
106                         idToAcMap.get(id).add(accession);
107                     }
108                     else {
109                         List<String> accs = new ArrayList<String>();
110                         accs.add(accession);
111                         idToAcMap.put(id, accs);
112                     }
113                 }
114             }
115         }
116     }

118     /**
119      * For a given bufferedReader into a uniprot file, creates a list of accessions.  For example
120      * if the buffered reader is pointing to the following:
121      *
122      * AC   P0CAP2; Q6EER8; Q6EES2; Q6EEV3; Q6EF00; Q6EF01; Q6EF02; Q6EF46;
123      * AC   Q6EFN8; Q6EM48; Q6K046; Q6K050; 
124      *
125      * The following list is returned:
126      *
127      * [P0CAP2, Q6EER8, Q6EES2, Q6EEV3, Q6EF00, Q6EF01, Q6EF02, Q6EF46, Q6EFN8, Q6EM48, Q6K046, Q6K050]
128      *
129      * Its assumed that the bufferedReader argument is pointing to the line preceeding the accession list
130      *
131      * @param bufferedReader BufferedReader
132      * @return List<String>
133      */
134     private List<String> getAccessionsList(BufferedReader bufferedReader) throws IOException {
135 
136         List<String> toReturn = new ArrayList<String>();
137 
138         String line = bufferedReader.readLine();
139         while (line.startsWith(AC_PREFIX)) {
140             String accessions = line.substring(AC_PREFIX.length());
141             accessions = accessions.replaceAll(" ", "");
142             toReturn.addAll(Arrays.asList(accessions.split(";")));
143             // we may read a line after the last AC (usually DT)
144             // in which case, we want to rollback the last readLine
145             bufferedReader.mark(1024);
146             line = bufferedReader.readLine();
147         }
148         // we've read past the last AC line, rollback
149         bufferedReader.reset();
150 
151         return toReturn;
152     }

154     /**
155      * Given a list of accessions, creates a string in the uniprot output format.
156      * 
157      * For example, if this is argument:
158      *
159      * [P0CAP2, Q6EER8, Q6EES2]
160      *
161      * The following is returned:
162      *
163      * AC   P0CAP2; Q6EER8; Q6EES2;
164      *
165      * @param accessionsList List<String>
166      * @return String
167      */
168     private String getAccessionsListAsString(List<String> accessionsList) {
169         
170         StringBuilder toReturn = new StringBuilder(AC_PREFIX);
171 
172         for (String accession : accessionsList) {
173             toReturn.append(accession + AC_DELIMITER);
174         }
175 
176         return toReturn.toString().trim();
177     }

179     /**
180      * Given uniprot data and a map containing accessions
181      * to remove, method returns clean uniprot export with those accessions removed.
182      *
183      * As a side effect, protein records that have accessions spanning multiple lines
184      * will have the accessions all concatenated on the same line.
185      *
186      * For example, the following input:
187      *
188      * AC   P0CAP2; Q6EER8; Q6EES2; Q6EEV3; Q6EF00;
189      * AC   Q6EFN8; Q6EM48; Q6K046; Q6K050; 
190      *
191      * Will be converted to:
192      *
193      * AC   P0CAP2; Q6EER8; Q6EES2; Q6EEV3; Q6EF00; Q6EFN8; Q6EM48; Q6K046; Q6K050; 
194      *
195      * With any accessions removed determined by given map.
196      *
197      * @param uniprotData
198      * @param accessionsMap Map<String, List<String>>
199      * @param out where to write the result
200      * @return String
201      */
202     private void cleanAccessions(InputStream uniprotData, OutputStream out) 
203     				throws IOException 
204     {
205         // create a buffered reader
206     	BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(uniprotData));
207     	PrintWriter writer = new PrintWriter(out);
208     	String line;
209     	String id;
210     	while ((line = bufferedReader.readLine()) != null) {
211     		//skip AC lines (we've already collected all ACs)
212     		if(!line.startsWith(AC_PREFIX)) {
213     			// process each id
214     			Matcher matcher = idPattern.matcher(line);
215     			if (matcher.matches()) {
216     				id = matcher.group(1);        		
217     				List<String> currentAccessions = idToAcMap.get(id);
218     				// check if any of these accessions should be removed
219     				List<String> accessionsToKeep = new ArrayList<String>(currentAccessions);
220     				for (String accession : currentAccessions) {
221     					// get the list of protein id's associated w/this accession
222     					// if accessions points to more than 1 protein record, clobber it
223     					if (acToIdMap.get(accession).size() > 1) {
224     						accessionsToKeep.remove(accession);
225     					}
226     				}
227     				writer.println(line); //write as is (not ID line)
228     				writer.println(getAccessionsListAsString(accessionsToKeep));
229     			} else {
230     				writer.println(line); //write as is (not ID line)
231     			}
232     		}
233     	}
234     	
235     	writer.close();
236     }

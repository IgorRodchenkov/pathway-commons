004package cpath.query;
005
006import java.util.Arrays;
007import java.util.Collection;
008
009import org.apache.commons.lang.ArrayUtils;
010import org.biopax.paxtools.controller.ModelUtils;
011import org.biopax.paxtools.model.Model;
012import org.springframework.util.LinkedMultiValueMap;
013import org.springframework.util.MultiValueMap;
014
015import cpath.client.CPathClient;
016import cpath.client.CPathClient.Direction;
017import cpath.client.util.CPathException;
018import cpath.service.Cmd;
019import cpath.service.CmdArgs;
020import cpath.service.GraphType;
021import cpath.service.OutputFormat;
022
023/**
024 * A query to be executed with {@link CPathClient}
025 * 
026 * @author rodche
027 */
028public final class CPathGraphQuery extends BaseCPathQuery<Model> implements
029                CPathQuery<Model> {
030
031        private boolean mergeEquivalentInteractions = false;
032    private Integer limit = 1;
033    private Direction direction;
034    private GraphType graphType = GraphType.NEIGHBORHOOD;
035        private String[] source;
036        private String[] target;
037        private String[] organism; // filter by
038        private String[] datasource; // filter by
039
040        /**
041         * @return the request
042         */
043        protected MultiValueMap<String, String> getRequestParams() {
044                MultiValueMap<String, String> request = new LinkedMultiValueMap<String, String>();
045
046                // common options for all graph commands
047                request.add(CmdArgs.kind.name(), graphType.name());
048                request.add(CmdArgs.limit.name(), limit.toString());
049                
050                if(source == null || source.length == 0)
051                        throw new IllegalArgumentException("Required 'source' " +
052                                        "parameter (cannot be null or empty)");
053                request.put(CmdArgs.source.name(), Arrays.asList(source));
054                
055                if(organism != null)
056                        request.put(CmdArgs.organism.name(), Arrays.asList(organism));
057                if(datasource != null)
058                        request.put(CmdArgs.datasource.name(), Arrays.asList(datasource));
059
060                switch (graphType) {
061                case COMMONSTREAM:
062                        if (direction != null) {
063                                if (direction == Direction.BOTHSTREAM)
064                                        throw new IllegalArgumentException(
065                                                        "Direction of common-stream query should be either upstream or downstream.");
066                                else
067                                        request.add(CmdArgs.direction.name(), direction.name());
068                        }
069                        break;
070                case PATHSBETWEEN:
071                        break;
072                case PATHSFROMTO:
073                        request.put(CmdArgs.target.name(), Arrays.asList(target));
074                        break;
075                case NEIGHBORHOOD:
076                default:
077                        if (direction != null)
078                                request.add(CmdArgs.direction.name(), direction.name());
079                        break;
080                }
081
082                return request;
083        }
084
085        /**
086         * Constructor.
087         * @param client cpath2 client instance 
088         */
089        public CPathGraphQuery(CPathClient client) {
090                this.limit = 1;
091                this.client = client;
092                this.graphType = GraphType.NEIGHBORHOOD;
093        }
094
095        /**
096         * Biopax graph query type (kind)
097         * @param graphType
098         * @return
099         */
100        public CPathGraphQuery kind(GraphType graphType) {
101                this.graphType = graphType;
102                return this;
103        }
104
105        /**
106         * A list of URIs or IDs (e.g., gene symbols) 
107         * of <em>source</em> biopax elements (required  
108         * for all graph query types).
109         * @param sources
110         * @return
111         */
112        public CPathGraphQuery sources(String[] sources) {
113                this.source = sources;
114                return this;
115        }
116
117        /**
118         * A list of URIs or IDs (e.g., gene symbols) 
119         * of <em>target</em> elements used by PathsFromTo 
120         * queries.
121         * @param targets
122         * @return
123         */
124        public CPathGraphQuery targets(String[] targets) {
125                this.target = targets;
126                return this;
127        }
128
129        /**
130         * A set of URIs or IDs (e.g., gene symbols) 
131         * of <em>source</em> biopax elements (required  
132         * for all graph query types).
133         * @param sources
134         * @return
135         */
136        public CPathGraphQuery sources(Collection<String> sources) {
137                this.source = sources.toArray(ArrayUtils.EMPTY_STRING_ARRAY);
138                return this;
139        }
140
141        /**
142         * A set of URIs or IDs (e.g., gene symbols) 
143         * of <em>target</em> elements used by PathsFromTo 
144         * queries.
145         * @param targets
146         * @return
147         */
148        public CPathGraphQuery targets(Collection<String> targets) {
149                this.target = targets.toArray(ArrayUtils.EMPTY_STRING_ARRAY);
150                return this;
151        }       
152        
153        /**
154     * Graph query search distance limit (default = 1);
155     * required by all but PathsFromTo queries.
156         * @param limit
157         * @return
158         */
159        public CPathGraphQuery limit(Integer limit) {
160                this.limit = limit;
161                return this;
162        }
163
164        /**
165         * Graph query direction. 
166         * The default depends on the graph query type.
167         * It's usually BOTHSTREAM, except for CommonStream query type, 
168         * or DOWNSTREAM (Note: BOTHSTREAM direction is incompatible 
169         * with CommonStream graph queries)
170         * 
171         * @param direction
172         * @return
173         */
174        public CPathGraphQuery direction(Direction direction) {
175                this.direction = direction;
176                return this;
177        }
178
179        /**
180         * Sets the filter by organism.
181         * @param organisms a set of organism names/taxonomy or null (no filter)
182         * @return
183         */
184        public CPathGraphQuery organismFilter(String[] organisms) {
185                this.organism = organisms;
186                return this;
187        }
188        
189        /**
190         * Sets the filter by pathway data source.
191         * @param datasources a set of data source names/URIs, or null (no filter)
192         * @return
193         */
194        public CPathGraphQuery datasourceFilter(String[] datasources) {
195                this.datasource = datasources;
196                return this;
197        }       
198        
199        /**
200         * Sets the filter by organism.
201         * @param organisms a set of organism names/taxonomy or null (no filter)
202         * @return
203         */
204        public CPathGraphQuery organismFilter(Collection<String> organisms) {
205                this.organism = organisms.toArray(ArrayUtils.EMPTY_STRING_ARRAY);
206                return this;
207        }
208        
209        /**
210         * Sets the filter by pathway data source.
211         * @param datasources a set of data source names/URIs, or null (no filter)
212         * @return
213         */
214        public CPathGraphQuery datasourceFilter(Collection<String> datasources) {
215                this.datasource = datasources.toArray(ArrayUtils.EMPTY_STRING_ARRAY);
216                return this;
217        }               
218        
219        /**
220         * Sets the option to merge equivalent interactions in the result model.
221         * 
222         * @param mergeEquivalentInteractions
223         */
224        public void mergeEquivalentInteractions(boolean mergeEquivalentInteractions) {
225                this.mergeEquivalentInteractions = mergeEquivalentInteractions;
226        }
227
228        @Override
229        public String stringResult(OutputFormat format) throws CPathException {
230                MultiValueMap<String, String> request = getRequestParams();
231                if (format == null)
232                        format = OutputFormat.BIOPAX;
233                request.add(CmdArgs.format.name(), format.name());
234
235                return client.post(Cmd.GRAPH.toString(), request, String.class);
236        }
237
238        @Override
239        public Model result() throws CPathException {
240                Model model = client.post(Cmd.GRAPH.toString(), getRequestParams(),
241                                Model.class);
242
243                if (mergeEquivalentInteractions && model != null) {
244                        ModelUtils.mergeEquivalentInteractions(model);
245                }
246
247                return model;
248        }
249
250}

028package cpath.service;
029
030import java.util.ArrayList;
031
032
033public enum Status {
034    /**
035     * Status Code:  Bad Request, Missing Arguments.
036     */
037    BAD_REQUEST(452, "Bad Request (illegal or no arguments)"),
038
039    /**
040     * Status Code:  No Results Found.
041     */
042    NO_RESULTS_FOUND(460, "No Results Found"),
043
044
045    /**
046     * Status Code:  Internal Server Error.
047     */
048    INTERNAL_ERROR(500, "Internal Server Error"),
049
050    
051    /**
052     * Status Code:  Internal Server Error.
053     */
054    MAINTENANCE(503, "Server is temporarily unavailable due to regular maintenance");
055
056    private final Integer errorCode;
057    private final String errorMsg;
058    
059
060        /**
061         * Private Constructor.
062         * 
063         * @param errorCode
064         * @param msg
065         */
066    private Status(int errorCode, String msg) {
067                this.errorCode = Integer.valueOf(errorCode);
068                this.errorMsg = msg;
069    }    
070        
071        
072    /**
073     * Gets Error Code.
074     *
075     * @return Error Code.
076     */
077    public int getErrorCode() {
078        return errorCode.intValue();
079    }
080
081    /**
082     * Gets Error Message.
083     *
084     * @return Error Message.
085     */
086    public String getErrorMsg() {
087        return errorMsg;
088    }
089
090    /**
091     * Gets Complete List of all Status Codes.
092     *
093     * @return ArrayList of Status Objects.
094     */
095    public static ArrayList<String> getAllStatusCodes() {
096        ArrayList<String> list = new ArrayList<String>();
097        for(Status statusCode : Status.values()) {
098                list.add(statusCode.name());
099        }
100        return list;
101    }
102
103    
104        public static Status fromCode(int code) {
105                for(Status type : Status.values()) {
106                        if(type.getErrorCode() == code)
107                                return type;
108                }
109                return null;
110        } 
111
112}

001package cpath.service.jaxb;
002
003import java.util.ArrayList;
004import java.util.HashSet;
005import java.util.List;
006import java.util.Set;
007
008import javax.xml.bind.annotation.*;
009
010
011@XmlRootElement(name="searchResponse")
012@XmlAccessorType(XmlAccessType.FIELD)
013@XmlType(name = "SearchResponse")
014public class SearchResponse extends ServiceResponse {
015    @XmlAttribute
016    private Integer numHits;
017    
018    @XmlAttribute
019    private Integer maxHitsPerPage;
020    
021    private List<SearchHit> searchHit; // count to get actual no. hits!
022    
023    @XmlAttribute
024    private Integer pageNo; //search result page number
025
026    @XmlAttribute
027    private String comment;
028    
029    @XmlTransient
030    private Set<String> providers; //pathway data provider standard names (for logging/stats)
031        
032        public SearchResponse() {
033        }
034    
035    public String getComment() {
036                return comment;
037        }
038        
039        public void setComment(String comment) {
040                this.comment = comment;
041        }
042    
043    public List<SearchHit> getSearchHit() {
044        if (searchHit == null) {
045            searchHit = new ArrayList<SearchHit>();
046        }
047        return this.searchHit;
048    }
049    
050    public void setSearchHit(List<SearchHit> searchHit) {
051                this.searchHit = searchHit;
052        }
053
054    /**
055     * The total number of hits
056     * @return
057     */
058        public Integer getNumHits() {
059                return numHits;
060        }
061
062        public void setNumHits(Integer numHits) {
063                this.numHits = numHits;
064        }
065
066        public Integer getPageNo() {
067                return pageNo;
068        }
069
070        public void setPageNo(Integer pageNo) {
071                if(pageNo >= 0)
072                        this.pageNo = pageNo;
073                else 
074                        throw new IllegalArgumentException("Negative values are not supported");
075        }
076
077        @Override
078        @XmlTransient
079        public boolean isEmpty() {
080                return getSearchHit().isEmpty();
081        }
082
083    /**
084     * The number of hits per page.
085     * @return
086     */
087        public Integer getMaxHitsPerPage() {
088                return maxHitsPerPage;
089        }
090
091        public void setMaxHitsPerPage(Integer maxHitsPerPage) {
092                this.maxHitsPerPage = maxHitsPerPage;
093        }
094        
095
096        /**
097         * Calculates the total number of search result
098         * pages using current  {@link #getMaxHitsPerPage()} and {@link #numHits}
099         * 
100         * @return no. pages or 0 (if there're no hits yet, or it's a wrong state)
101         */
102        public int numPages() {
103                if(numHits > 0 && maxHitsPerPage > 0)
104                        return (numHits-1)/maxHitsPerPage + 1;
105                else 
106                        return 0;
107        }
108        
109
110        @XmlTransient
111        public Set<String> getProviders() {
112                if(providers == null)
113                        providers = new HashSet<String>();
114                
115                return providers;
116        }
117        public void setProviders(Set<String> providers) {
118                this.providers = providers;
119        }
120        
121        
122        public Set<String> provenanceUris() {
123                Set<String> provUris = new HashSet<String>();
124                
125                if(searchHit != null) {
126                        for(SearchHit hit : searchHit) {
127                                //for Prov. there are names instead of URIs
128                                if(!"Provenance".equalsIgnoreCase(hit.getBiopaxClass()))
129                                        provUris.addAll(hit.getDataSource());
130                                else 
131                                        provUris.add(hit.getUri());
132                        }
133                }
134                
135                return provUris;
136        }
137}

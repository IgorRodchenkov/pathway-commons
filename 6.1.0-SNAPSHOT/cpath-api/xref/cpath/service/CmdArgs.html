28  package cpath.service;
29  
30  /**
31   * cPath2 web service command arguments.
32   * 
33   * @author rodche
34   *
35   */
36  public enum CmdArgs {
37  	uri("an identifier, usually a BioPAX element URI (default); multiple '&id=' arguments per query are supported"),
38      q("query string (full-text search)."),
39      page("search results page number (>=0)."),
40  	type("a BioPAX class - see the <a href=\"#valid_biopax_parameter\">valid values for type parameter</a> below"),
41  	kind("graph query type - see the <a href=\"#valid_graph_parameter\">valid values for kind parameter</a> below"),
42  	format("output format - see the <a href=\"#valid_output_parameter\">valid values for format parameter</a> below"),
43  	organism("filter by organism (multiple 'organism=' are allowed) - see the <a href=\"#valid_organism_parameter\">valid values for organism parameter</a> below"),
44  	datasource("filter by data sources (multiple 'datasource=' are allowed) - see the <a href=\"#valid_datasource_parameter\">valid values for datasource parameter</a> below"),
45  	source("graph query source URI (multiple 'source=' are allowed)"),
46  	target("graph query destination URI (multiple 'target=' are allowed)"),
47      limit("graph query search distance limit"),
48      biopax("a BioPAX OWL to convert"),
49      path("a BioPAX property path expression (like xPath)"),
50      direction("graph query parameter 'direction'"),
51      //alg("a user-defined BioPAX data analysis (byte code) to run."), //TODO
52  	;
53  	
54  	private final String info;
55  	
56  	public String getInfo() {
57  		return info;
58  	}
59  
60  	private CmdArgs(String info) {
61  		this.info = info;
62  	}
63  	
64  	@Override
65  	public String toString() {
66  		return name().toLowerCase();
67  	}	
68  }
